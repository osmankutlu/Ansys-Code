def _reg_get(name):
    try:
        key = Win32.Registry.CurrentUser.OpenSubKey(REG_KEY)
        if key is None: return ""
        val = key.GetValue(name); key.Close()
        return str(val) if val else ""
    except: return ""

def _reg_set(name, val):
    try:
        key = Win32.Registry.CurrentUser.CreateSubKey(REG_KEY)
        key.SetValue(name, val); key.Close()
    except: pass

def get_ns_groups(model):
    """Donus: {group_name: [ns_name, ...]}"""
    groups = {}
    try:
        root = model.NamedSelections
        for child in root.Children:
            # Grup nesnesi: Children property'si var ve bos degil
            try:
                sub = list(child.Children)
                if len(sub) > 0:
                    names = []
                    for ns in sub:
                        try: names.append(ns.Name)

def parse_out(path):
    with open(path, "r") as f:
        lines = f.readlines()

    results = {}
    cur_hi  = None

    def detect_dir(line):
        u = line.upper()
        if "PARTICIPATION FACTOR CALCULATION" not in u: return None
        tail = u.split("*****")[-1]
        for d in ["ROTX","ROTY","ROTZ"]:
            if d in tail: return d
        m = re.search(r"\b([XYZ])\s+DIRECTION", tail)
        if m: return m.group(1)
        return None

    i = 0
    while i < len(lines):
        u = lines[i].upper()
        m = re.search(r"CYCLIC SYMMETRY HARMONIC INDEX\s*[.\s]*(\d+)", u)
        if m:
            cur_hi = int(m.group(1))
            results.setdefault(cur_hi, {})
            i += 1; continue
        d = detect_dir(lines[i])
        if d is not None and d in ["X","Y","Z"]:
            if cur_hi is None: cur_hi = 0
            results.setdefault(cur_hi, {})
            j = i + 1
            while j < len(lines):
                if "MODE" in lines[j].upper() and "FREQUENCY" in lines[j].upper():
                    j += 1; break
                j += 1
            while j < len(lines):
                lj = lines[j].strip()
                if not lj: j += 1; continue
                if lj.startswith("---") or lj.lower().startswith("sum"): break
                parts = lj.split()
                try:
                    mode = int(parts[0])
                    freq = float(parts[1])
                    frac = float(parts[7]) if len(parts) >= 8 else None
                    results[cur_hi].setdefault(mode, {"freq": freq})
                    if frac is not None:
                        results[cur_hi][mode][d]     = frac
                        results[cur_hi][mode]["freq"] = freq
                except: pass
                j += 1
            i = j; continue
        i += 1

    out = []
    for hi in sorted(results.keys()):
        modes_dict = {}
        for mode in sorted(results[hi].keys()):
            rec = results[hi][mode]
            modes_dict[mode] = {
                "freq": rec.get("freq", 0.0),
                "X":    rec.get("X", 0.0),
                "Y":    rec.get("Y", 0.0),
                "Z":    rec.get("Z", 0.0),
            }
        out.append({"hi": hi, "modes": modes_dict})
    return out

# =============================================================================
# MECHANICAL'DAN HI/MOD/FREKANS OKUMA (full sweep icin, .OUT'a bagimsiz)
# Debug ile teyit edildi:
#   - rd.ResultSetCount / AnalysisSettings.MaximumModesToFind = HI sayisi
#   - rd.ListTimeFreq, HI=0..N-1 sirasiyla, her HI icin ardisik 'modes_per_hi'
#     adet frekans iceren bloklar halinde gelir.
# Donus: parse_out ile AYNI format -> [{"hi": n, "modes": {mode: {"freq","X","Y","Z"}}}]
# X/Y/Z burada hep 0.0 (effective mass bilgisi yok, full sweep'te kullanilmiyor)
# =============================================================================
def get_model_hi_modes(analysis):
    rd = analysis.GetResultsData()
    total_sets    = rd.ResultSetCount
    modes_per_hi  = analysis.AnalysisSettings.MaximumModesToFind
    if modes_per_hi <= 0 or total_sets <= 0:
        return []
    if total_sets % modes_per_hi != 0:
        # Tam bolunmuyor -- guvenli tarafta kal, tek HI=0 blogu gibi davranma,
        # bulabildigimiz kadarini yine de dondur (asagi yuvarla)
        print("UYARI: ResultSetCount ({0}) modes_per_hi'ye ({1}) tam bolunmuyor.".format(
            total_sets, modes_per_hi))
    n_hi  = total_sets // modes_per_hi
    freqs = list(rd.ListTimeFreq)

    out = []
    for hi in range(n_hi):
        start = hi * modes_per_hi
        end   = start + modes_per_hi
        modes_dict = {}
        for idx, f in enumerate(freqs[start:end]):
            mode = idx + 1
            modes_dict[mode] = {"freq": f, "X": 0.0, "Y": 0.0, "Z": 0.0}
        out.append({"hi": hi, "modes": modes_dict})
    return out

def eliminate_twins(parsed, twin_hz):
    """Her HI icinde frekans farki <= twin_hz olan modlardan birini eler.
    Kural: twin ciftinden max effective mass'e sahip olanı tut.
    EM esit veya EM veri yoksa: dusuk mod numarasini tut.
    Donus: parsed ile ayni yapi ama twin'ler elenmiş.
    Ayrica twin bilgisi loglanır: {hi: [(kept_mode, elim_mode, freq_diff), ...]}
    """
    twin_log = {}
    out = []
    for block in parsed:
        hi    = block["hi"]
        modes = block["modes"]
        # Frekansa gore sirala
        sorted_modes = sorted(modes.keys(), key=lambda m: modes[m]["freq"])
        eliminated   = set()
        twin_log[hi] = []

        for idx in range(len(sorted_modes) - 1):
            m1 = sorted_modes[idx]
            m2 = sorted_modes[idx + 1]
            if m1 in eliminated or m2 in eliminated:
                continue
            f1 = modes[m1]["freq"]
            f2 = modes[m2]["freq"]
            if abs(f2 - f1) <= twin_hz:
                # Twin bulundu: hangisini tut?
                em1 = max(modes[m1].get("X",0), modes[m1].get("Y",0), modes[m1].get("Z",0))
                em2 = max(modes[m2].get("X",0), modes[m2].get("Y",0), modes[m2].get("Z",0))
                if em2 > em1:
                    kept = m2; elim = m1
                else:
                    kept = m1; elim = m2  # esitse veya m1 buyukse dusuk no (m1) kalir
                eliminated.add(elim)
                twin_log[hi].append((kept, elim, abs(f2 - f1)))
                print("  Twin: HI{0} M{1}({2:.1f}Hz) ~ M{3}({4:.1f}Hz) diff={5:.3f}Hz -> kept M{6}".format(
                    hi, m1, f1, m2, f2, abs(f2-f1), kept))

        # Elenenleri cikar
        new_modes = {m: modes[m] for m in modes if m not in eliminated}
        out.append({"hi": hi, "modes": new_modes})

        if eliminated:
            print("  HI{0}: {1} twin eliminated, {2} modes remain".format(
                hi, len(eliminated), len(new_modes)))

    return out, twin_log

def select_modes(parsed, thr_pct):
    thr = thr_pct / 100.0
    out = {}
    for block in parsed:
        hi = block["hi"]
        for mode, rec in block["modes"].items():
            best = max(["X","Y","Z"], key=lambda d: rec[d])
            if rec[best] >= thr:
                out.setdefault(hi, {})[mode] = dict(rec, maxdir=best, maxval=rec[best])
    return out

def find_ns(model, ns_name):
    # Ust seviye ve grup icindeki NS'leri ara
    try:
        for item in model.NamedSelections.Children:
            try:
                sub = list(item.Children)
                if len(sub) > 0:
                    for ns in sub:
                        try:
                            if ns.Name == ns_name: return ns
                        except: pass
                else:
                    if item.Name == ns_name: return item
            except:
                if item.Name == ns_name: return item
    except: pass
    return None

def _sum_plotdata(obj):

    try:

def _sum_file(path):
    s = 0.0
    try:
        with open(path, "r") as f: lines = f.readlines()
    except: return 0.0
    for line in lines[1:]:
        line = line.strip()
        if not line: continue
        p = line.split("\t")
        if len(p) < 2: p = line.split()
        if len(p) < 2: continue
        try: s += float(p[-1].strip().replace(",","."))
        except: continue
    return s

def _move_to_group(model, items, group_name):
    """items listesini group_name adli bir TreeGroupingFolder'a tasir.
    Donus: olusturulan grup nesnesi (basarisizsa None)."""
    if not items:
        return None
    try:
        grp = model.AddTreeGroupingFolder(items)
        grp.Name = group_name
        print("{0} gruplama: {1} sonuc".format(group_name, len(items)))
        return grp
    except Exception as ex:
        try:
            grp = model.AddTreeGroupingFolder(items[0])
            grp.Name = group_name
            for item in items[1:]:
                try: item.GroupingFolder = grp
                except: pass
            print("{0} gruplama (tek tek): {1} sonuc".format(group_name, len(items)))
            return grp
        except Exception as ex2:
            print("{0} gruplama hatasi: {1}".format(group_name, ex2))
            return None

def group_results(sol, model, group_se=True, group_ms=True,
                  ms_prefix="_MS_", ms_group_name="Mode Shape Results"):
    """_SE_* sonuclarini tek gruba, {ms_prefix}* sonuclarini ana grup altinda
    3 alt gruba (Max Over Cyclic Phase / Cyclic Phase Of Maximum / Final) tasir.
    model.AddTreeGroupingFolder(list) kullanir (debug ile teyit edildi).
    """
    try:
        if group_se:
            se_items = []
            for child in list(sol.Children):
                try:
                    if str(child.Name).startswith("_SE_"):
                        se_items.append(child)
                except: pass
            _move_to_group(model, se_items, "Strain Energy Results")

        if group_ms:
            max_items = []
            phz_items = []
            fnl_items = []
            for child in list(sol.Children):
                try:
                    name = str(child.Name)
                except:
                    continue
                if not name.startswith(ms_prefix):
                    continue
                if "_MAX_HI" in name:
                    max_items.append(child)
                elif "_PHZ_HI" in name:
                    phz_items.append(child)
                elif "_HI0_M" in name or "_FNL_HI" in name:
                    fnl_items.append(child)

            sub_groups = []
            g1 = _move_to_group(model, max_items, "Max Over Cyclic Phase")
            if g1 is not None: sub_groups.append(g1)
            g2 = _move_to_group(model, phz_items, "Cyclic Phase Of Maximum")
            if g2 is not None: sub_groups.append(g2)
            g3 = _move_to_group(model, fnl_items, "Final (Cyclic Phase)")
            if g3 is not None: sub_groups.append(g3)

            # Alt gruplari ana grubun altina topla
            if sub_groups:
                _move_to_group(model, sub_groups, ms_group_name)

    except Exception as ex:
        print("Gruplama hatasi (kritik degil):", ex)

def _parse_phase_deg(quantity_str):
    """'349.99 [deg]' veya '350.0 [mm]' gibi stringden sayiyi cek."""
    try:
        s = str(quantity_str).strip()
        # "[" dan oncesini al
        num = s.split("[")[0].strip()
        return float(num)
    except:
        return 0.0

def _get_max_node_from_export(td, sol):
    """Total Deformation'i export edip max node ID'yi bul.
    Export formatı: NodeID  X  Y  Z  Value (tab ayracli)
    En yüksek Value satirindaki NodeID dondurulur."""
    _TMP[0] += 1
    tmp = get_temp_path("_td_export_{0}.txt".format(_TMP[0]))
    try:
        if os.path.exists(tmp): os.remove(tmp)
    except: pass
    try:
        td.ExportToTextFile(tmp)
    except:
        return None
    max_val  = -1e300
    max_node = None
    try:
        with open(tmp, "r") as f:
            lines = f.readlines()
        for line in lines[1:]:   # ilk satir header
            line = line.strip()
            if not line: continue
            parts_line = line.split("\t")
            if len(parts_line) < 2:
                parts_line = line.split()
            if len(parts_line) < 2: continue
            try:
                node_id = int(parts_line[0].strip())
                val     = float(parts_line[-1].strip().replace(",","."))
                if val > max_val:
                    max_val  = val
                    max_node = node_id
            except: continue
    except: pass
    try: os.remove(tmp)
    except: pass
    return max_node

def add_mode_shapes(sol, selected, prog=None, name_prefix="_MS"):
    """Tum secili (HI, mode) icin mode shape olusturur -- 3 TOPLU EVALUATE:
       ADIM 1 (toplu): Her (HI,mode) icin TD(MaximumOfCyclicPhase) olustur
                        -> TEK EvaluateAllResults
                        -> her birinden export ile max node bul
       ADIM 2 (toplu): Her (HI,mode) icin TD(CyclicPhaseOfMaximum, max_node scope) olustur
                        -> TEK EvaluateAllResults
                        -> her birinden phase acisini oku
       ADIM 3 (toplu): Her (HI,mode) icin TD(ResultSet, CyclicPhase=phase) olustur
                        -> TEK EvaluateAllResults
       HI=0 icin ADIM1/2/3 atlanir, sadece standart TD(ResultSet) olusturulur,
       bu da ADIM 3'un toplu evaluate'ine dahil edilir (ekstra evaluate gerekmez).
       name_prefix: UDR/TD isimlerinin onu (varsayilan "_MS", full sweep icin "_MSF" kullanilir)
    """
    def _upd(step, detail=""):
        print(step + ("  " + detail if detail else ""))
        if prog: prog.update(step, detail)

    def _cancelled():
        Application.DoEvents()
        return _CANCEL[0]

    def _make_td(name, hi, mode):
        td = sol.AddTotalDeformation()
        set_hindex(td, hi)
        set_mode(td, mode)
        td.Name = name
        return td

    combos_hi0 = [(hi, m) for hi in sorted(selected.keys()) if hi == 0
                  for m in sorted(selected[hi].keys())]
    combos_hin = [(hi, m) for hi in sorted(selected.keys()) if hi != 0
                  for m in sorted(selected[hi].keys())]
    total = len(combos_hi0) + len(combos_hin)
    if total == 0:
        return

    # ---- ADIM 1 (toplu): MaximumOfCyclicPhase icin tum HI>=1 TD'lerini olustur ----
    td_max_objs = {}
    if combos_hin:
        _upd("Mode Shapes [1/3] Creating Max-Over-Phase TDs...",
             "{0} combinations".format(len(combos_hin)))
        for (hi, mode) in combos_hin:
            if _cancelled(): return
            td = _make_td("{0}_MAX_HI{1}_M{2}".format(name_prefix, hi, mode), hi, mode)
            td.By = _SDS.MaximumOfCyclicPhase
            td_max_objs[(hi, mode)] = td

        if _cancelled(): return
        _upd("Mode Shapes [1/3] Evaluating (single pass)...",
             "{0} TDs".format(len(td_max_objs)))
        sol.EvaluateAllResults()

        if _cancelled(): return
        _upd("Mode Shapes [1/3] Finding max nodes...")
        max_nodes = {}
        for (hi, mode), td in td_max_objs.items():
            max_nodes[(hi, mode)] = _get_max_node_from_export(td, sol)
            print("  HI{0} M{1}: max node = {2}".format(hi, mode, max_nodes[(hi, mode)]))
    else:
        max_nodes = {}

    if _cancelled(): return

    # ---- ADIM 2 (toplu): CyclicPhaseOfMaximum icin tum HI>=1 TD'lerini olustur ----
    td_phz_objs = {}
    if combos_hin:
        _upd("Mode Shapes [2/3] Creating Phase-of-Maximum TDs...",
             "{0} combinations".format(len(combos_hin)))
        for (hi, mode) in combos_hin:
            if _cancelled(): return
            td = _make_td("{0}_PHZ_HI{1}_M{2}".format(name_prefix, hi, mode), hi, mode)
            td.By = _SDS.CyclicPhaseOfMaximum
            max_node = max_nodes.get((hi, mode))
            if max_node is not None:
                try:
                    sel_info = ExtAPI.SelectionManager.CreateSelectionInfo(
                        SelectionTypeEnum.MeshNodes)
                    sel_info.Ids = [max_node]
                    td.Location = sel_info
                except:
                    pass
            td_phz_objs[(hi, mode)] = td

        if _cancelled(): return
        _upd("Mode Shapes [2/3] Evaluating (single pass)...",
             "{0} TDs".format(len(td_phz_objs)))
        sol.EvaluateAllResults()

        if _cancelled(): return
        _upd("Mode Shapes [2/3] Reading phase angles...")
        phases = {}
        for (hi, mode), td in td_phz_objs.items():
            phases[(hi, mode)] = _parse_phase_deg(td.Maximum)
            print("  HI{0} M{1}: phase = {2:.2f} deg".format(hi, mode, phases[(hi, mode)]))
    else:
        phases = {}

    if _cancelled(): return

    # ---- ADIM 3 (toplu): Final TD'leri olustur (HI=0 dahil) ----
    _upd("Mode Shapes [3/3] Creating final TDs...",
         "{0} combinations".format(total))
    td_fnl_objs = {}

    for (hi, mode) in combos_hi0:
        if _cancelled(): return
        td = _make_td("{0}_HI0_M{1}".format(name_prefix, mode), hi, mode)
        td_fnl_objs[(hi, mode)] = td

    for (hi, mode) in combos_hin:
        if _cancelled(): return
        td = _make_td("{0}_FNL_HI{1}_M{2}".format(name_prefix, hi, mode), hi, mode)
        td.By = _SDS.ResultSet
        phase_deg = phases.get((hi, mode), 0.0)
        try:
            td.CyclicPhase = Quantity("{0} [deg]".format(phase_deg))
        except Exception as ex:
            print("  HI{0} M{1}: CyclicPhase atanamadi: {2}".format(hi, mode, ex))
        td_fnl_objs[(hi, mode)] = td

    if _cancelled(): return
    _upd("Mode Shapes [3/3] Evaluating (single pass)...",
         "{0} TDs".format(len(td_fnl_objs)))
    sol.EvaluateAllResults()

    for (hi, mode), td in td_fnl_objs.items():
        print("  HI{0} M{1}: final TD evaluated  Maximum={2}".format(hi, mode, td.Maximum))

def compute(analysis, parts, selected, prog=None):
    def _upd(step, detail=""):
        print(step + ("  " + detail if detail else ""))
        if prog: prog.update(step, detail)

    def _cancelled():
        Application.DoEvents()
        return _CANCEL[0]

    model = ExtAPI.DataModel.Project.Model
    sol   = analysis.Solution

    _upd("Resolving Named Selections...")
    ns_map = {}
    for p in parts:
        ns = find_ns(model, p)
        if ns is None:
            MessageBox.Show("Named Selection not found: " + p, "Error")
            return None
        ns_map[p] = ns

    total_hi_mode = sum(len(selected[hi]) for hi in selected)
    energy = {}

    for hi in sorted(selected.keys()):
        if _cancelled(): return None
        energy[hi] = {}

    # ---- TUM UDR'LERI OLUSTUR ----
    _upd("Creating all UDRs...",
         "{0} combinations".format(
             sum(len(selected[hi]) for hi in selected) * (len(parts) + 1)))

    # all_objs[(hi,mode)]    = all-bodies UDR
    # body_objs[(hi,mode,p)] = body UDR
    all_objs  = {}
    body_objs = {}
    obj_count = [0]

    for hi in sorted(selected.keys()):
        for mode in sorted(selected[hi].keys()):
            if _cancelled(): return None

            # All Bodies UDR
            o_all = sol.AddUserDefinedResult()
            o_all.Expression = SE_EXPR
            set_hindex(o_all, hi)
            set_mode(o_all, mode)
            o_all.Name = "_SE_ALL_HI{0}_M{1}".format(hi, mode)
            try: o_all.DisplayOption = ResultAveragingType.Unaveraged
            except: pass
            all_objs[(hi, mode)] = o_all
            obj_count[0] += 1

            # Her body icin UDR
            for p in parts:
                o = sol.AddUserDefinedResult()
                o.Expression = SE_EXPR
                o.Location   = ns_map[p].Location
                set_hindex(o, hi)
                set_mode(o, mode)
                o.Name = "_SE_{0}_HI{1}_M{2}".format(sanitize(p)[:14], hi, mode)
                try: o.DisplayOption = ResultAveragingType.Unaveraged
                except: pass
                body_objs[(hi, mode, p)] = o
                obj_count[0] += 1

    # ---- TEK EVALUATE ----
    if _cancelled(): return None
    _upd("Evaluating all results...",
         "{0} UDRs - single pass".format(obj_count[0]))
    sol.EvaluateAllResults()

    # ---- DEGERLERI OKU ----
    if _cancelled(): return None
    _upd("Reading results...")

    for hi in sorted(selected.keys()):
        energy[hi] = {}
        for mode in sorted(selected[hi].keys()):
            all_val   = read_udr_value(all_objs[(hi, mode)])
            body_vals = {p: read_udr_value(body_objs[(hi, mode, p)]) for p in parts}
            ratios    = {p: (body_vals[p] / all_val * 100.0) if all_val != 0 else 0.0
                         for p in parts}
            energy[hi][mode] = {
                "all":    all_val,
                "bodies": body_vals,
                "ratio":  ratios,
            }
            ratio_str = "  ".join("{0}={1:.1f}%".format(p, ratios[p]) for p in parts)
            print("  HI{0} M{1}: all={2:.3e}  {3}".format(hi, mode, all_val, ratio_str))

    # UDR sonuclari tamamlandi -> SE grubuna al
    if not _cancelled():
        _upd("Grouping Strain Energy results...")
        group_results(sol, model, group_se=True, group_ms=False)

    _upd("Done.", "Strain energy UDRs remain in tree.")
    return energy

def write_txt(path, parts, em_thr, se_thr, twin_hz, parsed, twin_log, selected, energy):
    SEP  = "=" * 76
    SEP2 = "-" * 76

    se_flagged = set()
    for hi in sorted(energy.keys()):
        for mode in sorted(energy[hi].keys()):
            ratios = energy[hi][mode]["ratio"]
            if any(ratios[p] >= se_thr for p in parts):
                se_flagged.add((hi, mode))

    with open(path, "w") as f:
        f.write("Cyclic Modal Part Identifier - Report\n")
        f.write("Effective mass threshold : {0:.2f}%\n".format(em_thr))
        f.write("Strain energy threshold  : {0:.2f}%\n".format(se_thr))
        f.write("Twin mode tolerance      : {0:.3f} Hz\n".format(twin_hz))
        f.write("SE expression            : {0}\n".format(SE_EXPR))
        f.write("Ratio formula            : body / all_bodies * 100\n")
        f.write(SEP + "\n\n")

        # BOLUM 0: Twin mod eliminasyonu ozeti
        total_twins = sum(len(twin_log.get(hi,[])) for hi in twin_log)
        f.write("SECTION 0: TWIN MODE ELIMINATION  (tolerance: {0:.3f} Hz)\n".format(twin_hz))
        f.write(SEP + "\n\n")
        if total_twins == 0:
            f.write("  No twin modes found.\n\n")
        else:
            f.write("  {0:>6}  {1:>8}  {2:>8}  {3:>10}  {4}\n".format(
                "HI","Kept","Elim","Diff(Hz)","Note"))
            f.write("  {0}\n".format(SEP2[:55]))
            for hi in sorted(twin_log.keys()):
                for (kept, elim, diff) in twin_log[hi]:
                    f.write("  {0:>6}  {1:>8}  {2:>8}  {3:>10.4f}  lower EM eliminated\n".format(
                        hi, kept, elim, diff))
            f.write("\n  Total eliminated: {0}\n\n".format(total_twins))

        # BOLUM 1: Effective Mass
        f.write("SECTION 1: EFFECTIVE MASS  (all harmonic indices)\n")
        f.write("(*) values exceeding threshold ({0:.2f}%)\n".format(em_thr))
        f.write(SEP + "\n\n")

        for block in parsed:
            hi    = block["hi"]
            modes = block["modes"]
            f.write("  Harmonic Index {0}\n".format(hi))
            f.write("  {0}\n".format(SEP2[:64]))
            f.write("  {0:>6}  {1:>12}  {2:>11}  {3:>11}  {4:>11}  {5}\n".format(
                "Mode", "Freq (Hz)", "X (%)", "Y (%)", "Z (%)", "Max"))
            f.write("  {0}\n".format(SEP2[:76]))
            for mode in sorted(modes.keys()):
                rec  = modes[mode]
                xp   = rec["X"] * 100
                yp   = rec["Y"] * 100
                zp   = rec["Z"] * 100
                best = max(["X","Y","Z"], key=lambda d: rec[d])
                def _fm(v):
                    s = "{0:>10.3f}".format(v)
                    return s + "*" if v >= em_thr else s + " "
                f.write("  {0:>6}  {1:>12.2f}  {2}  {3}  {4}  {5}({6:.2f}%)\n".format(
                    mode, rec["freq"], _fm(xp), _fm(yp), _fm(zp), best, rec[best]*100))
            f.write("\n")

        # BOLUM 2: Strain Energy
        f.write(SEP + "\n")
        f.write("SECTION 2: STRAIN ENERGY RATIO  (ENERGYPOTENTIAL)\n")
        f.write("Modes passing effective mass threshold ({0:.2f}%)\n".format(em_thr))
        f.write("Ratio = body / all_bodies * 100\n")
        f.write("(**) any body ratio >= SE threshold ({0:.2f}%)\n".format(se_thr))
        f.write(SEP + "\n\n")

        col_w = max(14, max(len(p) for p in parts) + 2)
        # Baslik: body isimleri
        hdr = "   {0:>6}  {1:>10}  {2:>16}  {3}  {4}".format(
            "Mode", "Freq(Hz)", "All Bodies",
            "  ".join(p.ljust(col_w) for p in parts), "Dominant")
        # Birim satiri
        hdr_unit = "   {0:>6}  {1:>10}  {2:>16}  {3}".format(
            "", "", "[mJ]",
            "  ".join("[mJ]".ljust(col_w) for p in parts))
        # Oran satiri
        hdr_r = "   {0:>6}  {1:>10}  {2:>16}  {3}".format(
            "", "", "(raw)",
            "  ".join("ratio(%)".ljust(col_w) for p in parts))

        for hi in sorted(energy.keys()):
            f.write("  Harmonic Index {0}\n".format(hi))
            f.write("  {0}\n".format(SEP2[:64]))
            f.write(hdr + "\n")
            f.write(hdr_unit + "\n")
            f.write(hdr_r + "\n")
            f.write("  {0}\n".format(SEP2[:min(len(hdr), 76)]))
            for mode in sorted(energy[hi].keys()):
                rec    = energy[hi][mode]
                ratios = rec["ratio"]
                best   = max(parts, key=lambda p: ratios[p]) if parts else "-"
                flag   = "**" if (hi, mode) in se_flagged else "  "
                em_str = "{0:.1f}% [{1}]".format(
                    selected[hi][mode]["maxval"]*100, selected[hi][mode]["maxdir"])
                # Deger satirlari: sadece sayilar, hizalama bozulmuyor
                body_cols  = "  ".join("{0:<{1}.4E}".format(rec["bodies"][p], col_w) for p in parts)
                ratio_cols = "  ".join("{0:<{1}}".format(
                    "{0:.2f}%".format(ratios[p]), col_w) for p in parts)
                f.write("{0} {1:>6}  {2:>10.2f}  {3:>16.4E}  {4}  {5}\n".format(
                    flag, mode, selected[hi][mode]["freq"], rec["all"], body_cols, best))
                f.write("   {0:>6}  {1:>10}  {2:>16}  {3}  Eff. Mass Ratio: {4}\n\n".format(
                    "", "", "", ratio_cols, em_str))
            f.write("\n")

        # BOLUM 3: SE esigini gecenler
        f.write(SEP + "\n")
        f.write("SECTION 3: MODES EXCEEDING STRAIN ENERGY THRESHOLD ({0:.2f}%)\n".format(se_thr))
        f.write(SEP + "\n\n")
        if not se_flagged:
            f.write("  No modes exceed the strain energy threshold.\n\n")
        else:
            f.write("  {0:>6}  {1:>6}  {2:>10}  {3:>12}  {4}\n".format(
                "HI","Mode","Freq(Hz)","MaxRatio(%)","Body@Max"))
            f.write("  {0}\n".format(SEP2[:60]))
            for (hi, mode) in sorted(se_flagged):
                ratios = energy[hi][mode]["ratio"]
                best   = max(parts, key=lambda p: ratios[p])
                f.write("  {0:>6}  {1:>6}  {2:>10.2f}  {3:>12.2f}%  {4}\n".format(
                    hi, mode, selected[hi][mode]["freq"], ratios[best], best))
            f.write("\n")

        # BOLUM 4: Her body icin dominant oldugu modlar (frekans + oran detayli)
        # NOT: "Dominant" = bu modda en yuksek SE oranina sahip body.
        # (**) = bu body'nin orani SE esigini ({0:.2f}%) de geciyor -- yani sadece
        # dominant degil, ayni zamanda guclu/belirleyici bir katkisi var.
        f.write(SEP + "\n")
        f.write("SECTION 4: DOMINANT MODES PER BODY  (** marks ratio >= SE threshold {0:.2f}%)\n".format(se_thr))
        f.write(SEP + "\n\n")
        for p in parts:
            # Bu body'nin dominant oldugu (hi, mode) ciftlerini frekans bilgisiyle topla
            rows = []
            for hi in sorted(energy.keys()):
                for mode in sorted(energy[hi].keys()):
                    ratios = energy[hi][mode]["ratio"]
                    best = max(parts, key=lambda pp: ratios[pp]) if parts else None
                    if best == p:
                        rows.append((hi, mode, selected[hi][mode]["freq"], ratios[p]))
            f.write("  {0}\n".format(p))
            if not rows:
                f.write("    -> No dominant modes\n\n")
                continue
            f.write("    {0:>6}  {1:>6}  {2:>10}  {3:>10}\n".format(
                "HI", "Mode", "Freq(Hz)", "Ratio(%)"))
            f.write("    {0}\n".format(SEP2[:44]))
            for (hi, mode, freq, ratio) in rows:
                flag = "**" if ratio >= se_thr else "  "
                f.write("  {0}{1:>6}  {2:>6}  {3:>10.2f}  {4:>9.2f}%\n".format(
                    flag, hi, mode, freq, ratio))
            f.write("    ({0} total)\n\n".format(len(rows)))

# ---- MAC hesap yardimcilari ----
def _mac_qty_to_float(q):
    for attr in ("Value", "Magnitude"):
        try: return float(getattr(q, attr))
        except: pass
    try: return float(str(q).split()[0].replace(",", "."))
    except: return 0.0

def _mac_dot(a, b):
    return sum(a[i] * b[i] for i in range(len(a)))

def _mac_value(phi_i, phi_j):
    num = _mac_dot(phi_i, phi_j)
    den = _mac_dot(phi_i, phi_i) * _mac_dot(phi_j, phi_j)
    if den < 1e-30: return 0.0
    return (num * num) / den

def _mac_col_letter(n):
    result = ""
    while n > 0:
        n, r = divmod(n - 1, 26)
        result = chr(65 + r) + result
    return result

def _mac_xml_escape(s):
    s = str(s)
    return s.replace("&","&amp;").replace("<","&lt;").replace(">","&gt;").replace('"',"&quot;")

def _mac_cs(col, row, text, style):
    return ('<c r="' + str(col) + str(row) + '" t="inlineStr" s="' + str(style) +
            '"><is><t>' + _mac_xml_escape(str(text)) + '</t></is></c>')

def _mac_cn(col, row, value, style, decimals=3):
    fmt = "{:." + str(decimals) + "f}"
    return ('<c r="' + str(col) + str(row) + '" t="n" s="' + str(style) +
            '"><v>' + fmt.format(float(value)) + '</v></c>')

def _mac_dpf_discover_modes(dpf, ds, model):
    """TimeFreqSupport'tan (set_id, hi, freq) okur."""
    tfs = model.TimeFreqSupport
    try: nsets = int(tfs.NumberSets)
    except: nsets = 0
    freqs = []
    try:
        ff = tfs.TimeFreqs
        freqs = [_mac_qty_to_float(x) for x in list(ff.Data)]
    except: freqs = []
    his = []
    try:
        hf = tfs.GetHarmonicIndices()
        his = [int(round(_mac_qty_to_float(x))) for x in list(hf.Data)]
    except:
        try:
            hf = tfs.HarmonicIndices
            his = [int(round(_mac_qty_to_float(x))) for x in list(hf.Data)]
        except: his = []
    if nsets == 0 and freqs: nsets = len(freqs)
    results = []
    for i in range(nsets):
        sid  = i + 1
        freq = freqs[i] if i < len(freqs) else 0.0
        hi   = abs(his[i]) if i < len(his) else 0
        results.append((sid, hi, freq))
    return results

def _mac_dpf_extract_phi(dpf, ds, set_id, node_ids):
    op = dpf.operators.result.displacement()
    op.inputs.data_sources.Connect(ds)
    op.inputs.time_scoping.Connect([set_id])
    try: op.inputs.read_cyclic.Connect(1)
    except: pass
    fc = op.outputs.fields_container.GetData()
    vals = {}
    try: nf = fc.FieldCount
    except:
        try: nf = len(fc)
        except: nf = 1
    for fi in range(nf):
        try: field = fc[fi]
        except: field = fc.GetFieldByIndex(fi)
        try: ids = list(field.ScopingIds)
        except: ids = list(field.Scoping.Ids)
        data = list(field.Data)
        try: comp = int(field.ComponentCount)
        except: comp = 3
        for k in range(len(ids)):
            nid = ids[k]
            if nid in vals: continue
            ux = data[k*comp + 0] if comp > 0 else 0.0
            uy = data[k*comp + 1] if comp > 1 else 0.0
            uz = data[k*comp + 2] if comp > 2 else 0.0
            vals[nid] = (ux, uy, uz)
    if node_ids is None:
        node_ids = sorted(vals.keys())
    phi = []
    for nid in node_ids:
        u = vals.get(nid, (0.0, 0.0, 0.0))
        phi.append(u[0]); phi.append(u[1]); phi.append(u[2])
    return phi, node_ids

def _mac_build_phi_matrix(dpf, ds, model, mode_list, prog=None, labels=None):
    n = len(mode_list)
    print("  MAC: {0} mod RST'den cekiliyor (DPF base sektor)...".format(n))
    phis = []; freqs = []; cur = None
    for idx, (sid, hi, fe) in enumerate(mode_list):
        lbl = labels[idx] if (labels and idx < len(labels)) else "Set {0}".format(sid)
        if prog:
            prog.update("MAC: reading mode shapes ({0}/{1})".format(idx+1, n),
                        "{0}  -  {1:.1f} Hz".format(lbl, fe))
        try:
            phi, cur = _mac_dpf_extract_phi(dpf, ds, sid, cur)
            phis.append(phi); freqs.append(fe)
        except Exception as e:
            print("    MAC HATA: {0}".format(e))
            phis.append([0.0] * (len(cur)*3 if cur else 0))
            freqs.append(fe)
    return phis, freqs, cur

def _mac_compute(phis):
    n = len(phis)
    mac = [[0.0]*n for _ in range(n)]
    for i in range(n):
        for j in range(i, n):
            v = _mac_value(phis[i], phis[j])
            mac[i][j] = v; mac[j][i] = v
    return mac

def _mac_make_labels(mode_list):
    counter = {}; labels = []
    for (sid, hi, fr) in mode_list:
        counter[hi] = counter.get(hi, 0) + 1
        labels.append("HI{0}_M{1}".format(hi, counter[hi]))
    return labels

def _mac_eliminate_twins_simple(mode_list, twin_hz):
    """mode_list: [(set_id, hi, freq), ...]. Ayni HI icinde frekansi yakin
    olan komsu ciftlerden dusuk set_id'yi tutar (MAC icin EM verisi yok).
    Donus: (kept_list, twin_log)"""
    by_hi = {}
    for (sid, hi, fr) in mode_list:
        by_hi.setdefault(hi, []).append((sid, fr))
    for hi in by_hi:
        by_hi[hi].sort(key=lambda x: x[1])

    twin_log = {}
    kept = []
    for hi in sorted(by_hi.keys()):
        items = by_hi[hi]
        eliminated = set()
        twin_log[hi] = []
        for idx in range(len(items) - 1):
            sid1, f1 = items[idx]
            sid2, f2 = items[idx + 1]
            if sid1 in eliminated or sid2 in eliminated: continue
            if abs(f2 - f1) <= twin_hz:
                eliminated.add(sid2)
                twin_log[hi].append((sid1, sid2, abs(f2-f1)))
                print("  MAC Twin: HI{0} Set{1}({2:.3f}Hz) ~ Set{3}({4:.3f}Hz) -> kept Set{5}".format(
                    hi, sid1, f1, sid2, f2, sid1))
        for (sid, fr) in items:
            if sid not in eliminated:
                kept.append((sid, hi, fr))
    kept.sort(key=lambda x: x[2])
    return kept, twin_log

def run_mac_module(rst_path, threshold, twin_hz, out_dir, prog=None):
    """Ana akistan cagrilir. RST'den DPF ile mode shape okur, MAC hesaplar,
    MAC_results.xlsx yazar. Kendi twin elemesini yapar (EM verisi yoktur)."""
    out_path = os.path.join(out_dir, "MAC_results.xlsx")
    if prog: prog.update("MAC: loading RST (DPF)...", os.path.basename(rst_path))
    dpf, ds, model = _mac_dpf_load(rst_path)

    if prog: prog.update("MAC: reading modes from TimeFreqSupport...")
    raw_modes = _mac_dpf_discover_modes(dpf, ds, model)
    print("MAC: bulunan set sayisi: {0}".format(len(raw_modes)))
    if not raw_modes:
        raise Exception("No modes found in RST.")
    raw_modes.sort(key=lambda x: x[2])

    if prog: prog.update("MAC: eliminating twin modes...", "tolerance {0} Hz".format(twin_hz))
    mode_list, twin_log = _mac_eliminate_twins_simple(raw_modes, twin_hz)
    print("MAC: twin sonrasi kalan mod: {0}".format(len(mode_list)))
    labels = _mac_make_labels(mode_list)

    if prog: prog.update("MAC: reading mode shapes (DPF)...", "{0} modes".format(len(mode_list)))
    phis, freqs, node_ids = _mac_build_phi_matrix(dpf, ds, model, mode_list, prog=prog, labels=labels)
    if not node_ids:
        raise Exception("Node list is empty.")

    if prog: prog.update("MAC: computing MAC matrix...", "{0}x{0}".format(len(mode_list)))
    mac = _mac_compute(phis)

    if prog: prog.update("MAC: writing Excel...", os.path.basename(out_path))
    _mac_build_xlsx(mac, mode_list, freqs, threshold, out_path, labels, twin_log, twin_hz)

    n_warn = sum(1 for i in range(len(mac)) for j in range(i+1, len(mac)) if mac[i][j] >= threshold)
    return out_path, len(mode_list), n_warn

def _safe_esc(s):
    return s.replace("&", "&amp;").replace("<", "&lt;").replace(">", "&gt;")

def _safe_col_letter(n):
    s = ""
    while n > 0:
        n, r = divmod(n - 1, 26)
        s = chr(65 + r) + s
    return s

def _safe_cellref(row, col):
    return "%s%d" % (_safe_col_letter(col), row)

def _safe_fold(eo, h2):
    period = 2.0 * h2
    m = eo % period
    if m > h2:
        return period - m
    return m

def _safe_parse_out_all(modes_by_hi_input):
    """parse_out() ana koddan gelen formati [{"hi":n,"modes":{m:{"freq":f,...}}}]
    -> SAFE'in bekledigi {hi: [(mode, freq), ...]} formatina cevirir."""
    out = {}
    for block in modes_by_hi_input:
        hi = block["hi"]
        modes = block["modes"]
        out[hi] = [(m, modes[m]["freq"]) for m in sorted(modes.keys())]
    return out

def run_safe_module(modes_by_hi_raw, blade_count, sector, max_hi,
                    idle_rpm, rl_rpm, margin_pct, critical_eos, out_path):
    """Ana akistan cagrilir. modes_by_hi_raw: ana koddan gelen parsed_clean
    (twin elenmis, [{"hi":n,"modes":{m:{"freq":f,...}}}] formatinda).
    SAFE_Diagram.xlsx yazar."""
    MODES_BY_HI = _safe_parse_out_all(modes_by_hi_raw)

    BLADE_COUNT = blade_count
    SECTOR = sector
    MAX_HI = max_hi
    IDLE_RPM = idle_rpm
    RL_RPM = rl_rpm
    MARGIN_PCT = margin_pct
    CRITICAL_EOS = critical_eos

    H2 = float(MAX_HI)
    MARGIN_FRAC = MARGIN_PCT / 100.
    IDLE_MARGIN_RPM = IDLE_RPM * (1. - MARGIN_FRAC)
    RL_MARGIN_RPM = RL_RPM * (1. + MARGIN_FRAC)

    eo_info = []
    for eo in CRITICAL_EOS:
        nd = _safe_fold(float(eo), H2)
        t = eo * IDLE_MARGIN_RPM / 60.
        u = eo * RL_MARGIN_RPM / 60.
        eo_info.append((eo, nd, t, u))

    max_eo = max(CRITICAL_EOS)
    N_ZZ = max(6, int(_safe_math.ceil(max_eo / H2)) + 1)

    def build_zigzag(rpm):
        pts = [(0., 0.)]
        for n in range(1, N_ZZ + 1):
            nd = 0. if (n % 2 == 0) else H2
            f = n * H2 * rpm / 60.
            pts.append((nd, f))
        return pts

    ZZ_IDLE = build_zigzag(IDLE_RPM)
    ZZ_RL = build_zigzag(RL_RPM)
    ZZ_IDLE_MARGIN = build_zigzag(IDLE_MARGIN_RPM)
    ZZ_RL_MARGIN = build_zigzag(RL_MARGIN_RPM)

    def crossing_eos_for(hi, freq):
        matches = []
        for (eo, nd, t, u) in eo_info:
            if abs(float(hi) - nd) > 0.5:
                continue
            lo, hif = (t, u) if t <= u else (u, t)
            if lo <= freq <= hif:
                matches.append(eo)
        return matches

    HI_LIST = sorted(MODES_BY_HI.keys())
    mode_points = []
    crossing_points = []
    crossing_details = []
    all_mode_freqs = []
    for hi in HI_LIST:
        for (mode_no, freq) in MODES_BY_HI[hi]:
            all_mode_freqs.append(freq)
            matches = crossing_eos_for(hi, freq)
            if matches:
                crossing_points.append((hi, mode_no, freq))
                crossing_details.append((hi, mode_no, freq, matches))
            else:
                mode_points.append((hi, mode_no, freq))

# MAIN
# =============================================================================
def main():
    global FONT_TITLE
    _CANCEL[0] = False   # her calistirmada sifirla

    model = ExtAPI.DataModel.Project.Model
    base  = Form().Font
    FONT_TITLE = Font(base, FontStyle.Bold)

    analyses = list(model.Analyses)
    modal    = [a for a in analyses if "modal" in str(a.AnalysisType).lower()]
    cands    = modal if modal else analyses
    if not cands:
        MessageBox.Show("No analysis found.", "Error"); return

    if len(cands) == 1:
        analysis = cands[0]
    else:
        d = AnalysisSelectDialog(cands)
        if d.ShowDialog() != DialogResult.OK: return
        analysis = d.selected_analysis

    # NS isimleri ve gruplar
    ns_groups = get_ns_groups(model)
    ns_names  = get_ns_names(model)
    if not ns_names:
        MessageBox.Show("At least 1 Named Selection is required.", "Error"); return

    dlg = ModalIdDialog(ns_names, ns_groups)
    if dlg.ShowDialog() != DialogResult.OK: return

    # ---- MAC popup (secildiyse) ----
    mac_dlg = None
    if dlg.want_mac:
        mac_dlg = MacPopupDialog(dlg.twin_hz)
        if mac_dlg.ShowDialog() != DialogResult.OK:
            return

    # ---- SAFE popup (secildiyse) ----
    safe_dlg = None
    if dlg.want_safe:
        safe_dlg = SafeDialog()
        if safe_dlg.ShowDialog() != DialogResult.OK:
            return

    # solve.out oku
    try:
        parsed = parse_out(dlg.out_file)
    except Exception as ex:
        MessageBox.Show("Failed to read solve.out:\n" + str(ex), "Error"); return
    if not parsed:
        MessageBox.Show("No participation factor table found in solve.out.", "Error"); return

    print("HI blocks found:", [b["hi"] for b in parsed])

    # Twin mod eliminasyonu (her iki akis icin de ortak)
    print("Twin elimination (tolerance: {0} Hz)...".format(dlg.twin_hz))
    parsed_clean, twin_log = eliminate_twins(parsed, dlg.twin_hz)
    total_twins = sum(len(twin_log[hi]) for hi in twin_log)
    if total_twins > 0:
        print("Total twins eliminated: {0}".format(total_twins))
    else:
        print("No twin modes found.")

    # ---- Section 1-4 akisi: effective mass filtreli ----
    run_main_flow = len(dlg.parts) > 0
    selected = {}
    nsel = 0
    if run_main_flow:
        selected = select_modes(parsed_clean, dlg.em_thr)
        nsel = sum(len(selected[hi]) for hi in selected)
        if nsel == 0:
            MessageBox.Show(
                "No modes exceed the effective mass threshold ({0:.1f}%).".format(dlg.em_thr),
                "Info")
            run_main_flow = False
        else:
            print("Selected modes (EM {0}%):".format(dlg.em_thr))
            for hi in sorted(selected.keys()):
                print("  HI {0}: modes {1}".format(hi, sorted(selected[hi].keys())))

    # ---- Full sweep akisi: sadece twin filtreli, effective mass yok ----
    # Veri kaynagi: .OUT degil, dogrudan Mechanical (analysis.GetResultsData())
    # Cunku .OUT bazen tum HI'lerin participation tablosunu icermez,
    # ama cozulen TUM HI/mod/frekans bilgisi her zaman Mechanical'da mevcuttur.
    full_selected = {}
    n_full = 0
    full_twin_log = {}
    if dlg.full_sweep:
        print("Full sweep: reading HI/mode/freq directly from Mechanical (not .OUT)...")
        try:
            model_hi_modes = get_model_hi_modes(analysis)
        except Exception as ex:
            MessageBox.Show("Failed to read HI/mode data from model:\n" + str(ex), "Error")
            model_hi_modes = []
        print("Model HI blocks found: {0}".format([b["hi"] for b in model_hi_modes]))
        for b in model_hi_modes:
            print("  HI {0}: {1} modes".format(b["hi"], len(b["modes"])))

        full_clean, full_twin_log = eliminate_twins(model_hi_modes, dlg.twin_hz)
        full_total_twins = sum(len(full_twin_log[hi]) for hi in full_twin_log)
        print("Full sweep twin elimination (tol {0} Hz): {1} eliminated".format(
            dlg.twin_hz, full_total_twins))

        for block in full_clean:
            hi = block["hi"]
            full_selected[hi] = dict(block["modes"])
        n_full = sum(len(full_selected[hi]) for hi in full_selected)
        print("Full sweep modes (twin-filtered only): {0} total".format(n_full))

    if not run_main_flow and not dlg.full_sweep:
        return
    if not run_main_flow and dlg.full_sweep and n_full == 0:
        MessageBox.Show("No modes found for full sweep.", "Info"); return

    # progress popup
    import System
    _t_start = System.DateTime.Now
    prog = ProgressDialog()
    prog.Show()
    prog.update("Starting...",
                "{0} modes (main) / {1} modes (full sweep)".format(nsel, n_full))

    energy = None
    if run_main_flow:
        try:
            energy = compute(analysis, dlg.parts, selected, prog)
        except Exception as ex:
            prog.Close()
            MessageBox.Show("Computation error:\n" + str(ex), "Error"); return

        if _CANCEL[0]:
            prog.Close()
            MessageBox.Show("Cancelled by user.", "Cancelled"); return

        if energy is None:
            prog.Close(); return

        # Mode shapes - sadece main flow'daki secili modlar icin
        prog.update("Creating mode shapes (main selection)...")
        add_mode_shapes(analysis.Solution, selected, prog, name_prefix="_MS")
        if not _CANCEL[0]:
            group_results(analysis.Solution, ExtAPI.DataModel.Project.Model,
                          group_se=False, group_ms=True,
                          ms_prefix="_MS_", ms_group_name="Mode Shape Results")

    # Full sweep - bagimsiz, ayrica calisir
    if dlg.full_sweep and n_full > 0 and not _CANCEL[0]:
        prog.update("Full sweep: creating mode shapes for ALL modes...",
                    "{0} modes".format(n_full))
        add_mode_shapes(analysis.Solution, full_selected, prog, name_prefix="_MSF")
        if not _CANCEL[0]:
            group_results(analysis.Solution, ExtAPI.DataModel.Project.Model,
                          group_se=False, group_ms=True,
                          ms_prefix="_MSF_", ms_group_name="Mode Shape Results (Full Sweep)")

    if _CANCEL[0]:
        prog.Close()
        MessageBox.Show("Cancelled by user.", "Cancelled"); return

    txt_path = None
    if run_main_flow and energy is not None:
        prog.update("Writing report...")
        if not os.path.exists(dlg.out_dir): os.makedirs(dlg.out_dir)
        txt_path = os.path.join(dlg.out_dir,
            "ModalPartID_{0}.txt".format(sanitize(analysis.Name)))
        write_txt(txt_path, dlg.parts, dlg.em_thr, dlg.se_thr, dlg.twin_hz,
                  parsed_clean, twin_log, selected, energy)

    # ---- MAC (secildiyse) ----
    mac_path = None; mac_n = 0; mac_warn = 0; mac_error = None
    if dlg.want_mac and mac_dlg is not None:
        try:
            prog.update("MAC: starting...", "")
            mac_path, mac_n, mac_warn = run_mac_module(
                mac_dlg.rst_path, mac_dlg.threshold, dlg.twin_hz, dlg.out_dir, prog=prog)
        except Exception as ex:
            mac_error = str(ex)
            print("MAC HATA:", mac_error)

    # ---- SAFE (secildiyse) ----
    safe_path = None; safe_cross = 0; safe_error = None
    if dlg.want_safe and safe_dlg is not None:
        try:
            prog.update("SAFE: writing diagram...", "")
            safe_out_path = os.path.join(dlg.out_dir, "SAFE_Diagram.xlsx")
            safe_path, safe_cross = run_safe_module(
                parsed_clean, safe_dlg.blade_count, safe_dlg.sector, safe_dlg.max_hi,
                safe_dlg.idle_rpm, safe_dlg.rl_rpm, safe_dlg.margin_pct,
                safe_dlg.critical_eos, safe_out_path)
        except Exception as ex:
            safe_error = str(ex)
            print("SAFE HATA:", safe_error)

    prog.done(); prog.Close()

    # Sure hesapla
    _t_end     = System.DateTime.Now
    _elapsed   = int((_t_end - _t_start).TotalSeconds)
    if _elapsed < 60:
        _dur_str = "{0} sec".format(_elapsed)
    else:
        _dur_str = "{0} min {1} sec".format(_elapsed // 60, _elapsed % 60)

    # ozet popup
    lines = []
    if run_main_flow and energy is not None:
        lines.append("Done.  {0} modes processed (main selection).\n".format(nsel))
        for hi in sorted(energy.keys()):
            lines.append("HI {0}:".format(hi))
            for mode in sorted(energy[hi].keys()):
                ratios = energy[hi][mode]["ratio"]
                best   = max(dlg.parts, key=lambda p: ratios[p]) if dlg.parts else "-"
                flag   = "**" if any(ratios[p] >= dlg.se_thr for p in dlg.parts) else "  "
                lines.append("{0}  Mode {1}  f={2:.0f}Hz -> {3} ({4:.1f}%)".format(
                    flag, mode, selected[hi][mode]["freq"], best, ratios[best]))
        if len(lines) > 45: lines = lines[:45] + ["... (see report)"]
    if dlg.full_sweep:
        lines.append("\nFull sweep: {0} mode shapes generated (twin-filtered only).".format(n_full))

    if dlg.want_mac:
        lines.append("")
        if mac_error:
            lines.append("MAC: FAILED -- " + mac_error)
        else:
            lines.append("MAC: {0} modes, {1} warnings >= threshold.".format(mac_n, mac_warn))

    if dlg.want_safe:
        lines.append("")
        if safe_error:
            lines.append("SAFE: FAILED -- " + safe_error)
        else:
            lines.append("SAFE: {0} modes crossing a Critical Engine Order.".format(safe_cross))

    msg  = "\n".join(lines) if lines else "Done."
    if txt_path:
        msg += "\n\nReport: " + txt_path
    if mac_path:
        msg += "\nMAC: " + mac_path
    if safe_path:
        msg += "\nSAFE: " + safe_path
    msg += "\nDuration: " + _dur_str
    MessageBox.Show(msg, "Modal Part Identifier - Summary")
    return analysis, dlg, parsed_clean, twin_log


main()
