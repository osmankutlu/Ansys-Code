REG_KEY = r"Software\ModalPartID"

def _reg_get(name):
    try:
        key = Win32.Registry.CurrentUser.OpenSubKey(REG_KEY)
        if key is None: return ""
        val = key.GetValue(name); key.Close()
        return str(val) if val else ""
    except: return ""

def _reg_set(name, val):
    try:
        key = Win32.Registry.CurrentUser.CreateSubKey(REG_KEY)
        key.SetValue(name, val); key.Close()
    except: pass

def sanitize(name):
    out = ""
    for ch in str(name):
        out += "_" if ch in '\\/:*?"<>|' else ch
    return out.strip()

def get_ns_groups(model):
    """Donus: {group_name: [ns_name, ...]}"""
    groups = {}
    try:
        root = model.NamedSelections
        for child in root.Children:
            # Grup nesnesi: Children property'si var ve bos degil
            try:
                sub = list(child.Children)
                if len(sub) > 0:
                    names = []
                    for ns in sub:
                        try: names.append(ns.Name)
                        except: pass
                    if names:
                        groups[child.Name] = names
            except:
                pass
    except: pass
    return groups

def get_ns_names(model):
    """Tum NS isimlerini dondurur (gruplar dahil)."""
    names = []
    try:
        for item in model.NamedSelections.Children:
            # Ust seviye NS
            try:
                sub = list(item.Children)
                if len(sub) > 0:
                    # Bu bir grup, icerigini ekle
                    for ns in sub:
                        try: names.append(ns.Name)
                        except: pass
                else:
                    names.append(item.Name)
            except:
                names.append(item.Name)
    except: pass
    return names

def parse_out(path):
    with open(path, "r") as f:
        lines = f.readlines()

    results = {}
    cur_hi  = None

    def detect_dir(line):
        u = line.upper()
        if "PARTICIPATION FACTOR CALCULATION" not in u: return None
        tail = u.split("*****")[-1]
        for d in ["ROTX","ROTY","ROTZ"]:
            if d in tail: return d
        m = re.search(r"\b([XYZ])\s+DIRECTION", tail)
        if m: return m.group(1)
        return None

    i = 0
    while i < len(lines):
        u = lines[i].upper()
        m = re.search(r"CYCLIC SYMMETRY HARMONIC INDEX\s*[.\s]*(\d+)", u)
        if m:
            cur_hi = int(m.group(1))
            results.setdefault(cur_hi, {})
            i += 1; continue
        d = detect_dir(lines[i])
        if d is not None and d in ["X","Y","Z"]:
            if cur_hi is None: cur_hi = 0
            results.setdefault(cur_hi, {})
            j = i + 1
            while j < len(lines):
                if "MODE" in lines[j].upper() and "FREQUENCY" in lines[j].upper():
                    j += 1; break
                j += 1
            while j < len(lines):
                lj = lines[j].strip()
                if not lj: j += 1; continue
                if lj.startswith("---") or lj.lower().startswith("sum"): break
                parts = lj.split()
                try:
                    mode = int(parts[0])
                    freq = float(parts[1])
                    frac = float(parts[7]) if len(parts) >= 8 else None
                    results[cur_hi].setdefault(mode, {"freq": freq})
                    if frac is not None:
                        results[cur_hi][mode][d]     = frac
                        results[cur_hi][mode]["freq"] = freq
                except: pass
                j += 1
            i = j; continue
        i += 1

    out = []
    for hi in sorted(results.keys()):
        modes_dict = {}
        for mode in sorted(results[hi].keys()):
            rec = results[hi][mode]
            modes_dict[mode] = {
                "freq": rec.get("freq", 0.0),
                "X":    rec.get("X", 0.0),
                "Y":    rec.get("Y", 0.0),
                "Z":    rec.get("Z", 0.0),
            }
        out.append({"hi": hi, "modes": modes_dict})
    return out

def eliminate_twins(parsed, twin_hz):
    """Her HI icinde frekans farki <= twin_hz olan ARDISIK (frekansa gore
    sirali) mod ciftlerinden HER ZAMAN ILKINI (dusuk numarali/dusuk
    frekansli olani) tutar, ikincisini eler. Bu, MAC modulundeki
    _mac_eliminate_twins_simple() ile AYNI kurali kullanir -- boylece
    kalan mod numaralari HER ZAMAN tutarli bir desende ilerler (ornegin
    1,3,5,7,... veya 2,4,6,8,...), EM oranina gore degisken/karisik bir
    secim yapilmaz.
    Donus: parsed ile ayni yapi ama twin'ler elenmiş.
    Ayrica twin bilgisi loglanır: {hi: [(kept_mode, elim_mode, freq_diff), ...]}
    """
    twin_log = {}
    out = []
    for block in parsed:
        hi    = block["hi"]
        modes = block["modes"]
        # Frekansa gore sirala
        sorted_modes = sorted(modes.keys(), key=lambda m: modes[m]["freq"])
        eliminated   = set()
        twin_log[hi] = []

        for idx in range(len(sorted_modes) - 1):
            m1 = sorted_modes[idx]
            m2 = sorted_modes[idx + 1]
            if m1 in eliminated or m2 in eliminated:
                continue
            f1 = modes[m1]["freq"]
            f2 = modes[m2]["freq"]
            if abs(f2 - f1) <= twin_hz:
                # HER ZAMAN ilkini (m1) tut, ikincisini (m2) ele --
                # tutarli bir mod numarasi deseni icin.
                kept = m1; elim = m2
                eliminated.add(elim)
                twin_log[hi].append((kept, elim, abs(f2 - f1)))
                print("  Twin: HI{0} M{1}({2:.1f}Hz) ~ M{3}({4:.1f}Hz) diff={5:.3f}Hz -> kept M{6}".format(
                    hi, m1, f1, m2, f2, abs(f2-f1), kept))

        # Elenenleri cikar
        new_modes = {m: modes[m] for m in modes if m not in eliminated}
        out.append({"hi": hi, "modes": new_modes})

        if eliminated:
            print("  HI{0}: {1} twin eliminated, {2} modes remain".format(
                hi, len(eliminated), len(new_modes)))

    return out, twin_log

def select_modes(parsed, thr_pct):
    thr = thr_pct / 100.0
    out = {}
    for block in parsed:
        hi = block["hi"]
        for mode, rec in block["modes"].items():
            best = max(["X","Y","Z"], key=lambda d: rec[d])
            if rec[best] >= thr:
                out.setdefault(hi, {})[mode] = dict(rec, maxdir=best, maxval=rec[best])
    return out

def find_ns(model, ns_name):
    # Ust seviye ve grup icindeki NS'leri ara
    try:
        for item in model.NamedSelections.Children:
            try:
                sub = list(item.Children)
                if len(sub) > 0:
                    for ns in sub:
                        try:
                            if ns.Name == ns_name: return ns
                        except: pass
                else:
                    if item.Name == ns_name: return item
            except:
                if item.Name == ns_name: return item
    except: pass
    return None

def set_mode(obj, m):
    # InternalObject.CyclicMode en guvenilir yol (debug ile teyit edildi)
    try:
        obj.InternalObject.CyclicMode = m
        return True
    except: pass
    for attr in ["CyclicMode", "Mode", "SetNumber"]:
        try: setattr(obj, attr, m); return True
        except: pass
    return False

def set_hindex(obj, hi):
    # InternalObject.CyclicHarmonicIndex en guvenilir yol (debug ile teyit edildi)
    try:
        obj.InternalObject.CyclicHarmonicIndex = hi
        return True
    except: pass
    for attr in ["HarmonicIndex", "CyclicHarmonicIndex"]:
        try: setattr(obj, attr, hi); return True
        except: pass
    return False

def _move_to_group(model, items, group_name):
    """items listesini group_name adli bir TreeGroupingFolder'a tasir.

    [DUZELTME] Eger AYNI isimde bir grup ONCEDEN VARSA (bir onceki
    calistirmadan kalma), yeni bir ALT grup acmak yerine (ic ice/nested
    buyume) ESKI grubu SILIP yerine TEPE SEVIYEDE, ayni isimde TEMIZ bir
    grup olusturur -- boylece arac tekrar tekrar calistirildikca gruplar
    ic ice buyumez, her calistirmada duz kalir.

    Donus: olusturulan grup nesnesi (basarisizsa None)."""
    if not items:
        return None
    try:
        parent_children = items[0].Parent.Children
        for child in list(parent_children):
            try:
                if str(child.Name) == group_name:
                    child.Delete()
                    break
            except Exception:
                pass
    except Exception:
        pass
    try:
        grp = model.AddTreeGroupingFolder(items)
        grp.Name = group_name
        print("{0} grouping: {1} result(s)".format(group_name, len(items)))
        return grp
    except Exception as ex:
        try:
            grp = model.AddTreeGroupingFolder(items[0])
            grp.Name = group_name
            for item in items[1:]:
                try: item.GroupingFolder = grp
                except: pass
            print("{0} grouping (one by one): {1} result(s)".format(group_name, len(items)))
            return grp
        except Exception as ex2:
            print("{0} grouping error: {1}".format(group_name, ex2))
            return None

def group_results(sol, model, ms_prefix="_MS_", ms_group_name="Mode Shape Results"):
    """{ms_prefix}* ile baslayan mode shape sonuclarini TEK DUZ bir gruba
    tasir. model.AddTreeGroupingFolder(list) kullanir (debug ile teyit
    edildi). Strain energy artik RST'den DPF ile okunuyor (Mechanical
    agacinda hicbir sonuc nesnesi olusturmuyor), bu yuzden SE gruplama
    mantigina artik gerek yok."""
    try:
        ms_items = []
        for child in list(sol.Children):
            try:
                name = str(child.Name)
            except:
                continue
            if name.startswith(ms_prefix):
                ms_items.append(child)
        _move_to_group(model, ms_items, ms_group_name)

    except Exception as ex:
        print("Grouping error (non-critical):", ex)

def add_mode_shapes(sol, selected, prog=None, name_prefix="_MS"):
    """Tum secili (HI, mode) icin mode shape (Total Deformation) olusturur --
    TEK ADIM, TEK EvaluateAllResults.

    ONCEDEN 3 ADIMDA yapiliyordu (MaximumOfCyclicPhase ile max nodu bul ->
    o nodda CyclicPhaseOfMaximum ile faz acisini bul -> o fazda final
    ResultSet olustur). Bu artik GEREKSIZ: TD.By varsayilan olarak
    SetDriverStyle.ResultSet'tir ve td.CyclicPhase atanmazsa varsayilan
    olarak 0 derece kullanilir -- yani sadece bir TD olusturup Name
    vermek yeterli, HI=0 ve HI!=0 arasinda fark yok."""
    def _upd(step, detail=""):
        print(step + ("  " + detail if detail else ""))
        if prog: prog.update(step, detail)

    def _cancelled():
        Application.DoEvents()
        return _CANCEL[0]

    combos = [(hi, m) for hi in sorted(selected.keys()) for m in sorted(selected[hi].keys())]
    if not combos:
        return

    _upd("Creating mode shape TDs...", "{0} combinations".format(len(combos)))
    td_objs = {}
    for (hi, mode) in combos:
        if _cancelled(): return
        td = sol.AddTotalDeformation()
        set_hindex(td, hi)
        set_mode(td, mode)
        td.Name = "{0}_HI{1}_M{2}".format(name_prefix, hi, mode)
        td_objs[(hi, mode)] = td

    if _cancelled(): return
    _upd("Evaluating mode shapes (single pass)...", "{0} TDs".format(len(td_objs)))
    sol.EvaluateAllResults()

    for (hi, mode), td in td_objs.items():
        print("  HI{0} M{1}: mode shape evaluated  Maximum={2}".format(hi, mode, td.Maximum))

def _rst_qty_to_float(q):
    for attr in ("Value", "Magnitude"):
        try: return float(getattr(q, attr))
        except: pass
    try: return float(str(q).split()[0].replace(",", "."))
    except: return 0.0

def _rst_flatten_region_elements(raw):
    """GetElementIdsFromRegionIds(geo_ids) TEK bir flat element ID listesi
    DEGIL, her geometri ID'si icin ayri bir KeyValuePair<int, IList<int>>
    (region_id -> element_id_listesi) dondurur -- standalone testlerle
    KESIN TEYIT EDILDI. Bu fonksiyon hepsini TEK bir flat, tekrarsiz (set)
    element ID listesine acar. 'raw' bazen None de donebilir."""
    elem_ids = set()
    if raw is None:
        return elem_ids
    try:
        raw_list = list(raw)
    except Exception:
        return elem_ids
    for item in raw_list:
        try:
            values = item.Value
            if values is not None:
                for v in values:
                    elem_ids.add(int(v))
            continue
        except AttributeError:
            pass
        try:
            values = item[1]
            if values is not None:
                for v in values:
                    elem_ids.add(int(v))
            continue
        except (TypeError, IndexError):
            pass
        try:
            elem_ids.add(int(item))
        except (TypeError, ValueError):
            pass
    return elem_ids

def resolve_body_element_ids_rst(model, parts, prog=None):
    """Her body (Named Selection) icin gercek MESH ELEMENT ID listesini
    Mechanical'in KENDI mesh<->geometri eslemesini kullanarak cozer
    (ns.Location.Ids -> GetElementIdsFromRegionIds)."""
    mesh_data = ExtAPI.DataModel.MeshDataByName("Global")
    if mesh_data is None:
        raise Exception("MeshDataByName('Global') failed.")

    n = len(parts)
    body_elem_ids = {}
    for i, p in enumerate(parts):
        pct = (100.0 * i / n) if n else 100.0
        if prog: prog.update("Strain Energy: Resolving body elements... {0:.0f}%".format(pct),
                             "{0}/{1}: {2}".format(i + 1, n, p))
        ns = find_ns(model, p)
        if ns is None:
            raise Exception("Named Selection not found: " + p)
        try:
            geo_ids = list(ns.Location.Ids)
        except Exception as ex:
            raise Exception("Failed to read ns.Location.Ids for '{0}': {1}".format(p, ex))
        if not geo_ids:
            print("  Strain Energy: WARNING - '{0}' has an EMPTY geometry ID list.".format(p))
            body_elem_ids[p] = set()
            continue
        try:
            raw = mesh_data.GetElementIdsFromRegionIds(geo_ids)
        except Exception as ex:
            raise Exception("GetElementIdsFromRegionIds failed for '{0}': {1}".format(p, ex))
        elem_ids = _rst_flatten_region_elements(raw)
        body_elem_ids[p] = elem_ids
        print("  Strain Energy: '{0}' -> {1} geometry region(s), {2} element ID(s)".format(
            p, len(geo_ids), len(elem_ids)))
    return body_elem_ids

def build_hi_mode_to_setid(dpf, ds, dpf_model):
    """RST'deki TimeFreqSupport'tan (HI, mode_local) -> set_id eslemesi
    kurar. mode_local numaralandirmasi, her HI blogu icinde RST'deki
    sirayla 1'den baslar -- bu, parse_out() ile
    AYNI mantiktir, yani selected[hi][mode] sozlugundeki 'mode'
    numaralari ile birebir eslesir."""
    tfs = dpf_model.TimeFreqSupport
    try: nsets = int(tfs.NumberSets)
    except: nsets = 0
    his = []
    try:
        hf = tfs.GetHarmonicIndices()
        his = [int(round(_rst_qty_to_float(x))) for x in list(hf.Data)]
    except:
        try:
            hf = tfs.HarmonicIndices
            his = [int(round(_rst_qty_to_float(x))) for x in list(hf.Data)]
        except:
            his = []
    mapping  = {}
    counters = {}
    for i in range(nsets):
        sid = i + 1
        hi  = abs(his[i]) if i < len(his) else 0
        counters[hi] = counters.get(hi, 0) + 1
        mapping[(hi, counters[hi])] = sid
    return mapping

def _rst_field_container_nonzero_info(fc, nf):
    total_elem = 0; nonzero_elem = 0
    for fi in range(nf):
        try: f = fc[fi]
        except: f = fc.GetFieldByIndex(fi)
        try:
            d = f.Data
            vals = list(d) if d is not None else []
        except:
            vals = []
        total_elem += len(vals)
        for v in vals:
            try:
                if abs(float(v)) > 1e-30:
                    nonzero_elem += 1
            except: pass
    return total_elem, nonzero_elem

def _rst_try_energy_operator(dpf, ds, cand, set_ids, diag):
    """Verilen TEK aday operator ismini dener (yuksek seviye once, dusuk
    seviye yedek). Basarili olursa (fields_container, nf) dondurur,
    olmazsa (None, 0)."""
    op = None; op_is_highlevel = False
    try:
        factory = getattr(dpf.operators.result, cand, None)
        if factory is not None:
            op = factory(); op_is_highlevel = True
    except Exception:
        pass
    if op is None:
        try:
            op = dpf.Operator(cand); op_is_highlevel = False
        except Exception as ex:
            diag("  '{0}' could not be created.".format(cand), str(ex))
            return None, 0

    fc = None
    try:
        if op_is_highlevel:
            op.inputs.data_sources.Connect(ds)
            op.inputs.time_scoping.Connect(list(set_ids))
            if hasattr(op.inputs, "read_cyclic"):
                op.inputs.read_cyclic.Connect(1)
            fc = op.outputs.fields_container.GetData()
        else:
            op.Connect(0, list(set_ids))
            op.Connect(4, ds)
            for out_fn in (lambda: op.GetOutput(0),
                          lambda: op.outputs.fields_container.GetData(),
                          lambda: op.Get_Output(0)):
                try:
                    fc = out_fn(); break
                except Exception:
                    continue
    except Exception as ex:
        diag("  '{0}' connect/read failed.".format(cand), str(ex))
        return None, 0

    if fc is None:
        return None, 0
    try: nf = fc.FieldCount
    except:
        try: nf = len(fc)
        except: nf = 0
    return fc, nf

def _rst_read_energy_bulk(dpf, ds, dpf_model, set_ids, prog=None):
    """Verilen TUM set_id'ler icin element strain energy'sini TEK DPF
    cagrisiyla okur. Donus: ({set_id: {elem_id: value}}, kullanilan_operator)

    Standalone testlerle (RST_SE_Test) KESIN TEYIT EDILDI: dogru operator
    'stiffness_matrix_energy' (= 0.5*u^T*K*u = toplam elastik strain
    energy, MAPDL SENE / ENERGYPOTENTIAL UDR ile AYNI buyukluk). Bu yuzden
    artik dir()/result_info taramasi yapip onlarca alakasiz aday
    denemiyoruz -- sadece bu ve kisa bir yedek listesi denenir.

    Aday deneme detaylari (hangi operator, kac field, kac sifirdan farkli
    vb.) SADECE konsola yazilir -- progress penceresi tek, sabit bir
    mesaj gosterir (aday isimleriyle spam yapmaz)."""
    def _console_diag(msg, detail=""):
        print("  Strain Energy: " + msg + ((" -- " + detail) if detail else ""))

    if prog: prog.update("Strain Energy: Locating energy data in RST...", "")

    candidates = ["stiffness_matrix_energy", "ENG_SE"]

    for cand in candidates:
        if _CANCEL[0]: return None, None
        _console_diag("Trying energy operator '{0}'...".format(cand))

        fc, nf = _rst_try_energy_operator(dpf, ds, cand, set_ids, _console_diag)
        if fc is None or nf == 0:
            _console_diag("'{0}' returned no data, trying next candidate.".format(cand))
            continue

        total_elem_count, nonzero_count = _rst_field_container_nonzero_info(fc, nf)
        if total_elem_count == 0:
            _console_diag("'{0}' fields are empty, trying next candidate.".format(cand))
            continue
        if nonzero_count == 0:
            _console_diag("'{0}' returned data but all values are zero, trying next candidate.".format(cand))
            continue

        _console_diag("Success: '{0}' -> {1} field(s), {2} element value(s), {3} nonzero.".format(
            cand, nf, total_elem_count, nonzero_count))

        result = {}
        for fi in range(nf):
            if _CANCEL[0]: return None, None
            if prog and (nf > 1):
                pct = 100.0 * fi / nf
                prog.update("Strain Energy: Reading element values... {0:.0f}%".format(pct),
                           "{0}/{1} set(s)".format(fi, nf))
            try: field = fc[fi]
            except: field = fc.GetFieldByIndex(fi)
            try: sid = int(fc.GetLabelSpace(fi)["time"])
            except:
                try: sid = int(field.TimeId)
                except: sid = set_ids[fi] if fi < len(set_ids) else (fi + 1)
            try:
                ids_raw = field.ScopingIds
                ids = list(ids_raw) if ids_raw is not None else []
            except:
                try:
                    ids_raw = field.Scoping.Ids
                    ids = list(ids_raw) if ids_raw is not None else []
                except:
                    ids = []
            try:
                data_raw = field.Data
                data = list(data_raw) if data_raw is not None else []
            except:
                data = []
            result[sid] = dict(zip(ids, data))

        if prog: prog.update("Strain Energy: Reading element values... 100%", "{0}/{0} set(s)".format(nf))
        return result, cand

    raise Exception("Neither 'stiffness_matrix_energy' nor 'ENG_SE' returned usable data. "
                    "Verify 'Output Controls > Volume and Energy = Yes' was enabled before "
                    "solving, then re-solve and try again.")

def compute_se_from_rst(analysis, parts, selected, rst_path, prog=None):
    """Strain energy'yi dogrudan cozulmus RST'den DPF ile okur (UDR/
    AddUserDefinedResult/EvaluateAllResults KULLANMAZ -- Mechanical
    agacinda hicbir sonuc nesnesi olusturmaz). Donus formati: energy[hi]
    [mode] = {"all","bodies","ratio"}. Standalone testlerle (RST_SE_Test,
    SE_Experiment OLD-vs-NEW) eski UDR yonteminin sonuclarina BIREBIR
    ESLESTIGI teyit edildi."""
    def _upd(step, detail=""):
        print(step + ("  " + detail if detail else ""))
        if prog: prog.update(step, detail)

    def _cancelled():
        Application.DoEvents()
        return _CANCEL[0]

    model = ExtAPI.DataModel.Project.Model

    body_elem_ids = resolve_body_element_ids_rst(model, parts, prog=prog)
    if _cancelled(): return None

    _upd("Strain Energy: Loading RST file...", os.path.basename(rst_path))
    dpf, ds, dpf_model = _rst_dpf_load(rst_path)
    if _cancelled(): return None

    _upd("Strain Energy: Building HI/mode mapping...")
    hi_mode_to_sid = build_hi_mode_to_setid(dpf, ds, dpf_model)

    needed_sids = []
    for hi in sorted(selected.keys()):
        for mode in sorted(selected[hi].keys()):
            sid = hi_mode_to_sid.get((hi, mode))
            if sid is None:
                raise Exception(
                    "No matching RST set found for HI{0} Mode{1} "
                    "(HI/mode numbering may not match the RST).".format(hi, mode))
            needed_sids.append(sid)
    if _cancelled(): return None

    energy_by_set, op_used = _rst_read_energy_bulk(dpf, ds, dpf_model, needed_sids, prog=prog)
    if _cancelled(): return None

    total_combos = sum(len(selected[hi]) for hi in selected)
    done = 0
    energy = {}
    for hi in sorted(selected.keys()):
        energy[hi] = {}
        for mode in sorted(selected[hi].keys()):
            done += 1
            pct = (100.0 * done / total_combos) if total_combos else 100.0
            _upd("Strain Energy: Summing per body... {0:.0f}%".format(pct),
                 "{0}/{1} mode(s)".format(done, total_combos))
            sid = hi_mode_to_sid[(hi, mode)]
            elem_vals = energy_by_set.get(sid, {})
            all_val   = sum(elem_vals.values(), 0.0)
            body_vals = {}
            for p in parts:
                ids = body_elem_ids[p]
                body_vals[p] = sum((elem_vals.get(eid, 0.0) for eid in ids), 0.0)
            ratios = {p: (body_vals[p] / all_val * 100.0) if all_val != 0 else 0.0
                      for p in parts}
            energy[hi][mode] = {"all": all_val, "bodies": body_vals, "ratio": ratios}
            ratio_str = "  ".join("{0}={1:.1f}%".format(p, ratios[p]) for p in parts)
            print("  HI{0} M{1} (Set{2}, op={3}): all={4:.3e}  {5}".format(
                hi, mode, sid, op_used, all_val, ratio_str))

    _upd("Strain Energy: Done.", "Read from RST (operator: {0}).".format(op_used))
    return energy

def write_txt(path, parts, em_thr, se_thr, twin_hz, parsed, twin_log, selected, energy,
             em_enabled=True, se_enabled=True, em_mode="filter"):
    SEP  = "=" * 76
    SEP2 = "-" * 76

    se_flagged = set()
    if energy is not None:
        for hi in sorted(energy.keys()):
            for mode in sorted(energy[hi].keys()):
                ratios = energy[hi][mode]["ratio"]
                if any(ratios[p] >= se_thr for p in parts):
                    se_flagged.add((hi, mode))

    with open(path, "w") as f:
        f.write("Cyclic Modal Part Identifier - Report\n")
        if em_enabled and em_mode == "info":
            f.write("Effective mass threshold : {0:.2f}%  (INFORMATIONAL ONLY -- "
                    "no elimination, all twin-filtered modes included)\n".format(em_thr))
        elif em_enabled:
            f.write("Effective mass threshold : {0:.2f}%\n".format(em_thr))
        else:
            f.write("Effective mass threshold : DISABLED (all modes included)\n")
        if se_enabled:
            f.write("Strain energy threshold  : {0:.2f}%\n".format(se_thr))
        else:
            f.write("Strain energy threshold  : DISABLED (strain energy not computed)\n")
        f.write("Twin mode tolerance      : {0:.3f} Hz\n".format(twin_hz))
        f.write("SE method                : RST direct (DPF, stiffness_matrix_energy)\n")
        f.write("Ratio formula            : body / all_bodies * 100\n")
        f.write(SEP + "\n\n")

        # BOLUM 0: Twin mod eliminasyonu ozeti
        total_twins = sum(len(twin_log.get(hi,[])) for hi in twin_log)
        f.write("SECTION 0: TWIN MODE ELIMINATION  (tolerance: {0:.3f} Hz)\n".format(twin_hz))
        f.write(SEP + "\n\n")
        if total_twins == 0:
            f.write("  No twin modes found.\n\n")
        else:
            f.write("  {0:>6}  {1:>8}  {2:>8}  {3:>10}  {4}\n".format(
                "HI","Kept","Elim","Diff(Hz)","Note"))
            f.write("  {0}\n".format(SEP2[:55]))
            for hi in sorted(twin_log.keys()):
                for (kept, elim, diff) in twin_log[hi]:
                    f.write("  {0:>6}  {1:>8}  {2:>8}  {3:>10.4f}  lower EM eliminated\n".format(
                        hi, kept, elim, diff))
            f.write("\n  Total eliminated: {0}\n\n".format(total_twins))

        # BOLUM 1: Effective Mass
        f.write("SECTION 1: EFFECTIVE MASS  (all harmonic indices)\n")
        if em_enabled:
            f.write("(*) values exceeding threshold ({0:.2f}%)\n".format(em_thr))
        else:
            f.write("(EM filter disabled -- all modes shown, no threshold applied)\n")
        f.write(SEP + "\n\n")

        for block in parsed:
            hi    = block["hi"]
            modes = block["modes"]
            f.write("  Harmonic Index {0}\n".format(hi))
            f.write("  {0}\n".format(SEP2[:64]))
            f.write("  {0:>6}  {1:>12}  {2:>11}  {3:>11}  {4:>11}  {5}\n".format(
                "Mode", "Freq (Hz)", "X (%)", "Y (%)", "Z (%)", "Max"))
            f.write("  {0}\n".format(SEP2[:76]))
            for mode in sorted(modes.keys()):
                rec  = modes[mode]
                xp   = rec["X"] * 100
                yp   = rec["Y"] * 100
                zp   = rec["Z"] * 100
                best = max(["X","Y","Z"], key=lambda d: rec[d])
                def _fm(v):
                    s = "{0:>10.3f}".format(v)
                    return s + "*" if (em_enabled and v >= em_thr) else s + " "
                f.write("  {0:>6}  {1:>12.2f}  {2}  {3}  {4}  {5}({6:.2f}%)\n".format(
                    mode, rec["freq"], _fm(xp), _fm(yp), _fm(zp), best, rec[best]*100))
            f.write("\n")

        if energy is None:
            f.write(SEP + "\n")
            f.write("SECTION 2-4: STRAIN ENERGY SECTIONS SKIPPED\n")
            f.write(SEP + "\n\n")
            f.write("  Strain energy computation was disabled by the user "
                    "(Section 2, Strain Energy Threshold checkbox unchecked).\n")
            f.write("  Only effective mass results (Section 1 above) were computed.\n\n")
            return

        # BOLUM 2: Strain Energy
        f.write(SEP + "\n")
        f.write("SECTION 2: STRAIN ENERGY RATIO  (RST direct, stiffness_matrix_energy)\n")
        if em_enabled:
            f.write("Modes passing effective mass threshold ({0:.2f}%)\n".format(em_thr))
        else:
            f.write("All modes (EM filter disabled)\n")
        f.write("Ratio = body / all_bodies * 100\n")
        f.write("(**) any body ratio >= SE threshold ({0:.2f}%)\n".format(se_thr))
        f.write(SEP + "\n\n")

        col_w = max(14, max(len(p) for p in parts) + 2)
        # Baslik: body isimleri
        hdr = "   {0:>6}  {1:>10}  {2:>16}  {3}  {4}".format(
            "Mode", "Freq(Hz)", "All Bodies",
            "  ".join(p.ljust(col_w) for p in parts), "Dominant")
        # Birim satiri
        hdr_unit = "   {0:>6}  {1:>10}  {2:>16}  {3}".format(
            "", "", "[mJ]",
            "  ".join("[mJ]".ljust(col_w) for p in parts))
        # Oran satiri
        hdr_r = "   {0:>6}  {1:>10}  {2:>16}  {3}".format(
            "", "", "(raw)",
            "  ".join("ratio(%)".ljust(col_w) for p in parts))

        for hi in sorted(energy.keys()):
            f.write("  Harmonic Index {0}\n".format(hi))
            f.write("  {0}\n".format(SEP2[:64]))
            f.write(hdr + "\n")
            f.write(hdr_unit + "\n")
            f.write(hdr_r + "\n")
            f.write("  {0}\n".format(SEP2[:min(len(hdr), 76)]))
            for mode in sorted(energy[hi].keys()):
                rec    = energy[hi][mode]
                ratios = rec["ratio"]
                best   = max(parts, key=lambda p: ratios[p]) if parts else "-"
                flag   = "**" if (hi, mode) in se_flagged else "  "
                em_str = "{0:.1f}% [{1}]".format(
                    selected[hi][mode]["maxval"]*100, selected[hi][mode]["maxdir"])
                # Deger satirlari: sadece sayilar, hizalama bozulmuyor
                body_cols  = "  ".join("{0:<{1}.4E}".format(rec["bodies"][p], col_w) for p in parts)
                ratio_cols = "  ".join("{0:<{1}}".format(
                    "{0:.2f}%".format(ratios[p]), col_w) for p in parts)
                f.write("{0} {1:>6}  {2:>10.2f}  {3:>16.4E}  {4}  {5}\n".format(
                    flag, mode, selected[hi][mode]["freq"], rec["all"], body_cols, best))
                f.write("   {0:>6}  {1:>10}  {2:>16}  {3}  Eff. Mass Ratio: {4}\n\n".format(
                    "", "", "", ratio_cols, em_str))
            f.write("\n")

        # BOLUM 3: SE esigini gecenler
        f.write(SEP + "\n")
        f.write("SECTION 3: MODES EXCEEDING STRAIN ENERGY THRESHOLD ({0:.2f}%)\n".format(se_thr))
        f.write(SEP + "\n\n")
        if not se_flagged:
            f.write("  No modes exceed the strain energy threshold.\n\n")
        else:
            f.write("  {0:>6}  {1:>6}  {2:>10}  {3:>12}  {4}\n".format(
                "HI","Mode","Freq(Hz)","MaxRatio(%)","Body@Max"))
            f.write("  {0}\n".format(SEP2[:60]))
            for (hi, mode) in sorted(se_flagged):
                ratios = energy[hi][mode]["ratio"]
                best   = max(parts, key=lambda p: ratios[p])
                f.write("  {0:>6}  {1:>6}  {2:>10.2f}  {3:>12.2f}%  {4}\n".format(
                    hi, mode, selected[hi][mode]["freq"], ratios[best], best))
            f.write("\n")

        # BOLUM 4: Her body icin dominant oldugu modlar (frekans + oran detayli)
        # NOT: "Dominant" = bu modda en yuksek SE oranina sahip body.
        # (**) = bu body'nin orani SE esigini ({0:.2f}%) de geciyor -- yani sadece
        # dominant degil, ayni zamanda guclu/belirleyici bir katkisi var.
        f.write(SEP + "\n")
        f.write("SECTION 4: DOMINANT MODES PER BODY  (** marks ratio >= SE threshold {0:.2f}%)\n".format(se_thr))
        f.write(SEP + "\n\n")
        for p in parts:
            # Bu body'nin dominant oldugu (hi, mode) ciftlerini frekans bilgisiyle topla
            rows = []
            for hi in sorted(energy.keys()):
                for mode in sorted(energy[hi].keys()):
                    ratios = energy[hi][mode]["ratio"]
                    best = max(parts, key=lambda pp: ratios[pp]) if parts else None
                    if best == p:
                        rows.append((hi, mode, selected[hi][mode]["freq"], ratios[p]))
            f.write("  {0}\n".format(p))
            if not rows:
                f.write("    -> No dominant modes\n\n")
                continue
            f.write("    {0:>6}  {1:>6}  {2:>10}  {3:>10}\n".format(
                "HI", "Mode", "Freq(Hz)", "Ratio(%)"))
            f.write("    {0}\n".format(SEP2[:44]))
            for (hi, mode, freq, ratio) in rows:
                flag = "**" if ratio >= se_thr else "  "
                f.write("  {0}{1:>6}  {2:>6}  {3:>10.2f}  {4:>9.2f}%\n".format(
                    flag, hi, mode, freq, ratio))
            f.write("    ({0} total)\n\n".format(len(rows)))

    def _on_mac_toggle(self, s, a):
        if self.chk_mac.Checked:
            try:
                twin_hz = float(self.txt_twin.Text.strip().replace(",", "."))
                assert twin_hz >= 0
            except Exception:
                MessageBox.Show(
                    "Please enter a valid Twin Mode Tolerance (Section 3) "
                    "before configuring MAC.", "Warning")
                self.chk_mac.Checked = False
                return
            # MAC artik kendi RST alanini sormuyor -- Section 1'de secilen
            # RST dosyasini kullanir (iki ayri RST secimi kafa karistiriciydi).
            rst_path = self.txt_se_rst.Text.strip()
            if not rst_path or not os.path.exists(rst_path):
                MessageBox.Show(
                    "Please choose a valid RST file in Section 1 "
                    "before configuring MAC.", "Warning")
                self.chk_mac.Checked = False
                return
            dlg = MacPopupDialog(twin_hz, rst_path)
            if dlg.ShowDialog() == DialogResult.OK:
                self._mac_dlg = dlg
            else:
                self.chk_mac.Checked = False

            self._mac_dlg = None

    def _load_group(self, s, a):

            MessageBox.Show("Please select an NS Group first.", "Warning"); return
        gname  = str(self.combo_group.SelectedItem)
        ns_lst = self.ns_groups.get(gname, [])
        if not ns_lst:

        while self.part_rows:
            row = self.part_rows.pop()
            for c in row: self.grp_parts.Controls.Remove(c)

                if str(combo.Items[j]) == ns_name:
                    combo.SelectedIndex = j; break
        if not self.part_rows: self._add_part_row()
        self._relayout()

    def _run(self, s, a):
        errs = []
        sel = list(dict.fromkeys(
            str(r[1].SelectedItem) for r in self.part_rows if r[1].SelectedIndex > 0))
        want_mac   = self.chk_mac.Checked
        want_safe  = self.chk_safe.Checked
        want_campbell = self.chk_campbell.Checked

        # .OUT dosyasi sadece body-bazli akis (Section 2), SAFE diyagrami
        # veya Campbell (genel diyagram icin TUM HI/mod/freq verisini
        # solve.out'tan okur) icin gereklidir.
        need_out = (len(sel) >= 1) or want_safe or want_campbell
        of = self.txt_out.Text.strip()
        if need_out and (not of or not os.path.exists(of)):
            errs.append("Please select a valid solve.out file.")

        # Body secimi sadece strain-energy akisi icin sart. MAC, SAFE
        # veya Campbell tek basina secilirse body gerekmez.
        if len(sel) < 1 and not want_mac and not want_safe and not want_campbell:
            errs.append("Please select at least 1 body (Named Selection), "
                       "or enable MAC / SAFE / Campbell below.")

        if want_mac and self._mac_dlg is None:
            errs.append("MAC settings are incomplete. Please re-check the "
                       "'Generate MAC matrix' box and configure it.")
        if want_safe and self._safe_dlg is None:
            errs.append("SAFE settings are incomplete. Please re-check the "
                       "'Generate SAFE diagram' box and configure it.")
        if want_campbell and self._campbell_dlg is None:
            errs.append("Campbell settings are incomplete. Please re-check the "
                       "'Generate Campbell Diagram' box and configure it.")

        want_em = self.chk_em.Checked
        em_mode = "info" if (want_em and self.rad_em_info.Checked) else "filter"
        want_se = self.chk_se.Checked

        em = 0.0
        if want_em:
            try:
                em = float(self.txt_em.Text.strip().replace(",",".")); assert 0<=em<=100
            except:
                errs.append("Effective mass threshold must be 0-100."); em = 5.0

        se = 0.0
        if want_se:
            try:
                se = float(self.txt_se.Text.strip().replace(",",".")); assert 0<=se<=100
            except:
                errs.append("Strain energy threshold must be 0-100."); se = 10.0

        # Strain energy ARTIK SADECE RST'den okunur (UDR kaldirildi) --
        # bu yuzden Strain Energy Threshold acikken RST dosyasi ZORUNLUDUR.
        se_rst_path = self.txt_se_rst.Text.strip()
        if want_se:
            if not se_rst_path or not os.path.exists(se_rst_path):
                errs.append("Please choose a valid RST file (strain energy is now "
                           "always read directly from RST).")

        try:
            tw = float(self.txt_twin.Text.strip().replace(",",".")); assert tw >= 0
        except:
            errs.append("Twin mode tolerance must be >= 0 Hz."); tw = 2.0

        # Cikti klasoru Section 1 (body-bazli akis), MAC, SAFE ve Campbell
        # icin gereklidir (hepsi dlg.out_dir'e dosya yazar).
        need_out_dir = (len(sel) >= 1) or want_mac or want_safe or want_campbell
        od = self.txt_dir.Text.strip()
        if need_out_dir and not od:
            errs.append("Please select an output folder.")

        if errs:
            MessageBox.Show("Please fix:\n\n" + "\n".join(errs), "Warning"); return

        self.parts      = sel
        self.em_thr     = em
        self.se_thr     = se
        self.want_em    = want_em
        self.em_mode    = em_mode
        self.want_se    = want_se
        self.se_rst_path = se_rst_path
        self.twin_hz    = tw
        self.want_mac   = want_mac
        self.want_safe  = want_safe
        self.want_campbell = want_campbell
        self.out_file = of; self.out_dir = od
        _reg_set("LastOutFile", of); _reg_set("LastDir", od)
        if se_rst_path:
            _reg_set("LastSeRst", se_rst_path)
        self.DialogResult = DialogResult.OK; self.Close()

def _fmt_eta(seconds):
    """Kalan sureyi okunabilir sekilde bicimlendirir."""
    s = int(max(0, seconds))
    if s < 60:
        return "{0} sn".format(s)
    m, s = divmod(s, 60)
    if m < 60:
        return "{0} dk {1} sn".format(m, s)
    h, m = divmod(m, 60)
    return "{0} sa {1} dk".format(h, m)

        for hi in sorted(sel_set):
            self.lst_sel.Items.Add(str(hi))

        to_add = [self._parse_hi(self.lst_avail.Items[i])
                  for i in range(self.lst_avail.Items.Count)
                  if self.lst_avail.GetSelected(i)]
        for hi in to_add:
            if hi not in self.selected_his:
                self.selected_his.append(hi)
        self.selected_his.sort()
        self._populate()

    def _on_add_all(self, sender, args):
        self.selected_his = list(self._available_his)
        self._populate()

    def _on_del(self, sender, args):
        to_del = set(self._parse_hi(self.lst_sel.Items[i])
                     for i in range(self.lst_sel.Items.Count)
                     if self.lst_sel.GetSelected(i))
        self.selected_his = [hi for hi in self.selected_his if hi not in to_del]
        self._populate()

    def _on_ok(self, sender, args):
        if not self.selected_his:
            MessageBox.Show("Please select at least one Harmonic Index.", "Warning")
            return
        self.DialogResult = DialogResult.OK
        self.Close()

# ---- MAC hesap yardimcilari ----
def _mac_qty_to_float(q):
    for attr in ("Value", "Magnitude"):
        try: return float(getattr(q, attr))
        except: pass
    try: return float(str(q).split()[0].replace(",", "."))
    except: return 0.0

def _mac_dot(a, b):
    return sum(a[i] * b[i] for i in range(len(a)))

def _mac_parse_mode_no(label):
    """[YENI -- MOD AILESI TAKIBI] 'HI 5 Mod 13' gibi bir etiketten mod
    numarasini (13) cikarir -- family tablosunu/chart'ini artik frekansa
    degil, (HI, Mod) numarasina gore siralamak icin kullanilir (kucuk
    HI/mod once gelsin, ornegin 'HI 0 Mod 3' 'HI 0 Mod 5'ten once)."""
    try:
        return int(str(label).rsplit("Mod", 1)[-1].strip())
    except Exception:
        return 0

def _mac_col_letter(n):
    result = ""
    while n > 0:
        n, r = divmod(n - 1, 26)
        result = chr(65 + r) + result
    return result

def _mac_xml_escape(s):
    s = str(s)
    return s.replace("&","&amp;").replace("<","&lt;").replace(">","&gt;").replace('"',"&quot;")

def _mac_cs(col, row, text, style):
    return ('<c r="' + str(col) + str(row) + '" t="inlineStr" s="' + str(style) +
            '"><is><t>' + _mac_xml_escape(str(text)) + '</t></is></c>')

def _mac_cn(col, row, value, style, decimals=3):
    fmt = "{:." + str(decimals) + "f}"
    return ('<c r="' + str(col) + str(row) + '" t="n" s="' + str(style) +
            '"><v>' + fmt.format(float(value)) + '</v></c>')

def _mac_dpf_import():
    import mech_dpf
    import Ans.DataProcessing as dpf
    try: mech_dpf.setExtAPI(ExtAPI)
    except: pass
    return dpf

def _mac_dpf_load(rst_path):
    dpf = _mac_dpf_import()
    ds = dpf.DataSources(rst_path)
    model = dpf.Model(ds)
    return dpf, ds, model

def _mac_dpf_discover_modes(dpf, ds, model):
    """TimeFreqSupport'tan (set_id, hi, freq) okur."""
    tfs = model.TimeFreqSupport
    try: nsets = int(tfs.NumberSets)
    except: nsets = 0
    freqs = []
    try:
        ff = tfs.TimeFreqs
        freqs = [_mac_qty_to_float(x) for x in list(ff.Data)]
    except: freqs = []
    his = []
    try:
        hf = tfs.GetHarmonicIndices()
        his = [int(round(_mac_qty_to_float(x))) for x in list(hf.Data)]
    except:
        try:
            hf = tfs.HarmonicIndices
            his = [int(round(_mac_qty_to_float(x))) for x in list(hf.Data)]
        except: his = []
    if nsets == 0 and freqs: nsets = len(freqs)
    results = []
    for i in range(nsets):
        sid  = i + 1
        freq = freqs[i] if i < len(freqs) else 0.0
        hi   = abs(his[i]) if i < len(his) else 0
        results.append((sid, hi, freq))
    return results

def _mac_dpf_extract_phi_bulk(dpf, ds, set_ids, prog=None):
    """TOPLU OKUMA: tum setleri TEK operator cagrisiyla RST'den ceker ve
    HER SET ICIN HAZIR phi VEKTORU (ref_ids sirasinda, duz liste) uretir.

    ONEMLI - KARMASIK (COMPLEX) MOD SEKILLERI:
    HI != 0 (ve Nyquist disi) cyclic modlar icin mod sekli KARMASIKTIR;
    DPF her istenen set icin REAL ve IMAGINARY olmak uzere 2 AYRI field
    dondurur (ayni 'time' etiketi, farkli 'complex' etiketiyle: 0=real,
    1=imag). Bunlardan sadece birini kullanmak YANLIS sonuc verir. Bu
    fonksiyon sqrt(real^2+imag^2) esdegerligini DOF bazinda uygulayarak
    gercek-degerli (real-valued) bir buyukluk vektoru uretir; ardindan
    ux,uy,uz'yi de node basina TEK bir toplam buyuklukte (sqrt toplam
    kareler) birlestirerek MAC hesabinin uzunlugunu 3'te 1'e indirir --
    NODE SAYISI/COZUNURLUGU DEGISMEZ.

    PERFORMANS (ASIL DARBOGAZ BURADAYDI):
    RST'deki tum sonuc setleri ayni mesh/node scoping'e (ScopingIds)
    sahipse -- ki modal sonuclarda normalde boyledir -- per-node dict
    kurma adimi TAMAMEN ATLANIR ve field.Data DOGRUDAN kullanilir.

    Donus: data_by_set = {set_id: phi_list}  (phi_list uzunlugu = n_nodes,
           node basina TEK buyukluk degeri), ref_ids (node ID sirasi)
    """
    import time as _time
    _t0 = _time.time()

    def _diag(msg, detail=""):
        print("  MAC (bulk) DIAG: " + msg + ((" -- " + detail) if detail else ""))
        if prog: prog.update("MAC: " + msg, detail)

    _diag("setting up operator, connecting inputs...", "{0} set(s)".format(len(set_ids)))

    op = dpf.operators.result.displacement()
    op.inputs.data_sources.Connect(ds)
    op.inputs.time_scoping.Connect(list(set_ids))
    try:
        op.inputs.read_cyclic.Connect(1)
        _diag("read_cyclic=1 (BASE SECTOR ONLY) connected successfully.")
    except Exception as ex:
        _diag("WARNING: read_cyclic FAILED TO CONNECT! May be performing a "
              "full 360-degree expansion (VERY SLOW).", str(ex))

    _diag("Reading mode shapes from RST...", "{0} set(s)".format(len(set_ids)))
    _t1 = _time.time()
    fc = op.outputs.fields_container.GetData()
    _t2 = _time.time()
    print("  MAC (bulk) DIAG: RST read complete -- {0:.1f}s".format(_t2 - _t1))

    try: nf = fc.FieldCount
    except:
        try: nf = len(fc)
        except: nf = 0

    _diag("operator returned {0} field(s) (requested set count: {1}).".format(nf, len(set_ids)))
    if nf > len(set_ids):
        _diag("INFO: field count exceeds requested set count -- for HI!=0 "
              "modes, a REAL+IMAGINARY field pair may have been "
              "returned (complex mode shape). This is NORMAL, the "
              "code will combine them per-DOF via sqrt(re^2+im^2).")

    # Referans node sirasi -- ilk field'dan. MAC sonucu node siralamasindan
    # bagimsizdir (dot product invariant), yeter ki TUM modlarda AYNI sira
    # kullanilsin -- bu yuzden sort() bile gerekmiyor.
    try: field0 = fc[0]
    except: field0 = fc.GetFieldByIndex(0)
    ref_ids = list(field0.ScopingIds)
    try: comp0 = int(field0.ComponentCount)
    except: comp0 = 3
    n_nodes = len(ref_ids)
    _diag("reference node order obtained.", "{0} node(s), comp={1}".format(n_nodes, comp0))

    # sid -> {0: ("flat"/"dict", payload, comp), 1: (...)}   (0=real, 1=imag)
    raw = {}
    fast_hits = [0]
    slow_hits = [0]

    for fi in range(nf):
        try: field = fc[fi]
        except: field = fc.GetFieldByIndex(fi)

        # Bu field hangi set/time'a ait ve real mi imaginary mi?
        # Once metadata'dan dene (GetLabelSpace -> "time"/"complex"),
        # olmazsa TimeId, o da olmazsa pozisyonel eslestir.
        sid = None
        cplx = 0
        try:
            ls = fc.GetLabelSpace(fi)
            if "time" in ls: sid = int(ls["time"])
            if "complex" in ls: cplx = int(ls["complex"])
        except:
            pass
        if sid is None:
            try: sid = int(field.TimeId)
            except: sid = set_ids[fi] if fi < len(set_ids) else (fi + 1)
        # Guvenlik: "complex" etiketi okunamadiysa ama ayni sid icin bu
        # slot zaten doluysa (real+imag ikilisi label'siz geldi demektir),
        # ikinciyi otomatik "imag" (1) olarak isaretle -- veri KAYBOLMASIN.
        if cplx in raw.get(sid, {}):
            cplx = 1 if cplx == 0 else 0

        try: ids = list(field.ScopingIds)
        except: ids = list(field.Scoping.Ids)
        data = list(field.Data)
        try: comp = int(field.ComponentCount)
        except: comp = 3

        if comp == comp0 and ids == ref_ids:
            payload = ("flat", data, comp)
            fast_hits[0] += 1
        else:
            node_vals = {}
            for k in range(len(ids)):
                nid = ids[k]
                ux = data[k*comp + 0] if comp > 0 else 0.0
                uy = data[k*comp + 1] if comp > 1 else 0.0
                uz = data[k*comp + 2] if comp > 2 else 0.0
                node_vals[nid] = (ux, uy, uz)
            payload = ("dict", node_vals, comp)
            slow_hits[0] += 1

        raw.setdefault(sid, {})[cplx] = payload

        if prog and (fi % 5 == 0 or fi == nf - 1):
            pct = 100.0 * (fi + 1) / nf
            prog.update("MAC: Reading mode shapes... {0:.0f}%".format(pct),
                       "{0}/{1}".format(fi + 1, nf))

    print("  MAC (bulk) DIAG: all fields processed -- "
         "fast path: {0}, slow path: {1}".format(fast_hits[0], slow_hits[0]))

    def _to_flat(payload):
        kind, val, comp = payload
        if kind == "flat":
            if comp == 3:
                return val
            flat = [0.0] * (n_nodes * 3)
            for k in range(n_nodes):
                if comp > 0: flat[k*3]     = val[k*comp]
                if comp > 1: flat[k*3 + 1] = val[k*comp + 1]
                if comp > 2: flat[k*3 + 2] = val[k*comp + 2]
            return flat
        else:
            flat = [0.0] * (n_nodes * 3)
            for k in range(n_nodes):
                u = val.get(ref_ids[k], (0.0, 0.0, 0.0))
                flat[k*3] = u[0]; flat[k*3 + 1] = u[1]; flat[k*3 + 2] = u[2]
            return flat

    _diag("Combining mode shape data...", "{0} mode(s)".format(len(raw)))
    data_by_set = {}
    n_combined = 0
    for sid, parts in raw.items():
        real_flat = _to_flat(parts[0]) if 0 in parts else None
        imag_flat = _to_flat(parts[1]) if 1 in parts else None

        node_mag = [0.0] * n_nodes
        if real_flat is not None and imag_flat is not None:
            for k in range(n_nodes):
                base = k * comp0
                s = 0.0
                for c in range(comp0):
                    re = real_flat[base + c]; im = imag_flat[base + c]
                    m = (re*re + im*im) ** 0.5   # bu DOF'un buyuklugu (real+imag)
                    s += m * m
                node_mag[k] = s ** 0.5           # node'un TOPLAM yer degistirme buyuklugu
            n_combined += 1
        elif real_flat is not None or imag_flat is not None:
            src = real_flat if real_flat is not None else imag_flat
            for k in range(n_nodes):
                base = k * comp0
                s = 0.0
                for c in range(comp0):
                    v = src[base + c]
                    s += v * v
                node_mag[k] = s ** 0.5
        else:
            continue
        data_by_set[sid] = node_mag

    print("  MAC (bulk) DIAG: combination complete -- {0} mode(s), "
         "phi length {1} (was {2})".format(n_combined, n_nodes, n_nodes * comp0))
    print("  MAC (bulk) DIAG: total time {0:.1f}s (including RST read).".format(
        _time.time() - _t0))
    return data_by_set, ref_ids

def _mac_build_phi_matrix_bulk(dpf, ds, model, mode_list, prog=None, labels=None):
    """TOPLU OKUMA: RST tek operator cagrisiyla acilir, real+imaginary
    birlesimi ve node-hizli-yol islemesi ARTIK _mac_dpf_extract_phi_bulk
    icinde yapiliyor (data_by_set = {set_id: hazir phi listesi}). Burada
    sadece mode_list sirasina gore diziliyor -- ucuz bir islem, yine de
    Parallel.For ile yapiliyor (zararsiz, hizli)."""
    n = len(mode_list)
    set_ids = [m[0] for m in mode_list]

    if prog:
        prog.update("MAC (bulk): reading ALL mode shapes - single call...",
                    "{0} sets".format(n))
    print("  MAC (bulk): {0} mode(s), fetching from RST with a SINGLE operator call...".format(n))

    data_by_set, node_ids = _mac_dpf_extract_phi_bulk(dpf, ds, set_ids, prog=prog)

    n_nodes = len(node_ids)
    freqs   = [m[2] for m in mode_list]
    phis    = [None] * n   # her worker sadece kendi idx'ine yazar

    if prog:
        prog.update("MAC (bulk): finalizing phi vectors...", "{0} modes".format(n))

    _missing_flags = [False] * n

    def _build_one(idx):
        sid = mode_list[idx][0]
        phi = data_by_set.get(sid)
        if phi is None:
            phi = [0.0] * n_nodes
            _missing_flags[idx] = True
        phis[idx] = phi

    Parallel.For(0, n, System.Action[int](_build_one))

    for idx in range(n):
        if _missing_flags[idx]:
            print("    MAC (bulk) WARNING: No data found for Set {0}.".format(mode_list[idx][0]))

    if prog:
        prog.update("MAC (bulk): phi vectors ready.", "{0} modes".format(n))

    return phis, freqs, node_ids

def _mac_compute(phis, prog=None):
    """MAC matrisi PARALEL hesaplanir (System.Threading.Tasks.Parallel.For).
    IronPython'da GIL olmadigi icin bu gercek coklu-cekirdek paralellik
    saglar. Optimizasyon 1: her modun kendi-ile-nokta-carpimi (self dot)
    once bir kez hesaplanip cache'lenir.
    Optimizasyon 2 (IS YUKU DENGELEME): satir bazli paralellestirme
    (Parallel.For(0,n,_row)) UCGEN bir is yuku uretir -- i=0 gorevi (n-1)
    dot product yaparken i=n-1 gorevi 0 tane yapar; .NET'in varsayilan
    statik/blok bolme mantigi bu durumda cekirdeklerin cogunu erken
    bosa cikarir (dusuk CPU kullanimi). Bunu onlemek icin TUM (i,j)
    ciftleri esit-maliyetli TEK BOYUTLU bir listeye cikarilir ve
    Parallel.For bu liste uzerinde calisir -- artik her paralel birim
    esit is yaptigi icin cekirdekler gercekten dengeli calisir.

    ILERLEME/ETA (prog verilirse): Parallel.For, cagiran (UI) thread'i
    tamamen bloke eder -- bu sirada Application.DoEvents() cagrilamaz,
    yani progress penceresi guncellenemez. Bunu asmak icin asil
    hesaplama System.Threading.Tasks.Task.Run ile AYRI bir arka plan
    thread'inde baslatilir; UI thread'i serbest kalir ve ~250ms'de bir
    tamamlanan cift sayisina bakip yuzde + gecen sure + tahmini kalan
    sureyi (dogrusal ekstrapolasyonla) progress penceresine yazar."""
    n = len(phis)
    mac = [[0.0]*n for _ in range(n)]

    self_dots = [0.0] * n
    def _self(i):
        self_dots[i] = _mac_dot(phis[i], phis[i])
    Parallel.For(0, n, System.Action[int](_self))

    for i in range(n):
        mac[i][i] = 1.0 if self_dots[i] > 1e-30 else 0.0

    pairs = []
    for i in range(n):
        for j in range(i + 1, n):
            pairs.append((i, j))
    n_pairs = len(pairs)

    if n_pairs == 0:
        return mac

    completed = [0]
    _lock = threading.Lock()

    def _pair(k):
        i, j = pairs[k]
        num = _mac_dot(phis[i], phis[j])
        den = self_dots[i] * self_dots[j]
        v = 0.0 if den < 1e-30 else (num * num) / den
        mac[i][j] = v
        mac[j][i] = v
        with _lock:
            completed[0] += 1

    if prog is None:
        # Progress penceresi verilmediyse eskisi gibi dogrudan blokla
        # (ilerleme raporlanmaz, ama daha basit/dusuk overhead'li yol).
        Parallel.For(0, n_pairs, System.Action[int](_pair))
        return mac

    import time as _pt
    _t_start = _pt.time()

    def _run_all():
        Parallel.For(0, n_pairs, System.Action[int](_pair))

    task = System.Threading.Tasks.Task.Run(System.Action(_run_all))

    while not task.IsCompleted:
        Application.DoEvents()
        System.Threading.Thread.Sleep(250)
        done = completed[0]
        pct = (100.0 * done / n_pairs) if n_pairs else 100.0
        elapsed = _pt.time() - _t_start
        if done > 0:
            eta_str = _fmt_eta(elapsed * (n_pairs - done) / float(done))
        else:
            eta_str = "calculating..."
        prog.update(
            "MAC: computing MAC matrix (parallel, balanced)...  %{0:.1f}".format(pct),
            "{0}/{1} pairs  |  elapsed: {2}  |  estimated remaining: {3}".format(
                done, n_pairs, _fmt_eta(elapsed), eta_str))

    if task.IsFaulted:
        raise Exception(str(task.Exception))

    prog.update("MAC: computing MAC matrix (parallel, balanced)...  %100.0",
                "{0}/{0} pairs  |  done -- {1}".format(
                    n_pairs, _fmt_eta(_pt.time() - _t_start)))
    return mac

def _mac_make_labels(mode_list, sid_to_rawmode=None):
    """Etiketleri olusturur. sid_to_rawmode verilirse (sid -> (hi, raw_mode))
    GERCEK/HAM mod numarasi kullanilir (twin elenmis olsa bile orijinal
    numaralandirma korunur, ornegin '1,3,5,7' -- ardisik '1,2,3,4' DEGIL).
    Verilmezse (eski davranis) mode_list icindeki sirayla 1'den baslayan
    bir sayac kullanilir."""
    labels = []
    if sid_to_rawmode is not None:
        for (sid, hi, fr) in mode_list:
            raw = sid_to_rawmode.get(sid)
            mode_no = raw[1] if raw is not None else "?"
            labels.append("HI {0} Mod {1}".format(hi, mode_no))
        return labels

    counter = {}
    for (sid, hi, fr) in mode_list:
        counter[hi] = counter.get(hi, 0) + 1
        labels.append("HI {0} Mod {1}".format(hi, counter[hi]))
    return labels

def _mac_eliminate_twins_simple(mode_list, twin_hz):
    """mode_list: [(set_id, hi, freq), ...]. Ayni HI icinde frekansi yakin
    olan komsu ciftlerden dusuk set_id'yi tutar (MAC icin EM verisi yok).
    Donus: (kept_list, twin_log)"""
    by_hi = {}
    for (sid, hi, fr) in mode_list:
        by_hi.setdefault(hi, []).append((sid, fr))
    for hi in by_hi:
        by_hi[hi].sort(key=lambda x: x[1])

    twin_log = {}
    kept = []
    for hi in sorted(by_hi.keys()):
        items = by_hi[hi]
        eliminated = set()
        twin_log[hi] = []
        for idx in range(len(items) - 1):
            sid1, f1 = items[idx]
            sid2, f2 = items[idx + 1]
            if sid1 in eliminated or sid2 in eliminated: continue
            if abs(f2 - f1) <= twin_hz:
                eliminated.add(sid2)
                twin_log[hi].append((sid1, sid2, abs(f2-f1)))
                print("  MAC Twin: HI{0} Set{1}({2:.3f}Hz) ~ Set{3}({4:.3f}Hz) -> kept Set{5}".format(
                    hi, sid1, f1, sid2, f2, sid1))
        for (sid, fr) in items:
            if sid not in eliminated:
                kept.append((sid, hi, fr))
    # HI'ye gore grupla, sonra HER HI icinde frekansa gore sirala -- boylece
    # etiketler 'HI 0 Mod 1, HI 0 Mod 2, ... HI 1 Mod 1, ...' seklinde
    # HI'ye gore ardisik cikar (frekansa gore GLOBAL siralamada farkli
    # HI'lerin modlari birbirine karisiyordu).
    kept.sort(key=lambda x: (x[1], x[2]))
    return kept, twin_log

def _mac_sheet_name(base, used):
    """Excel sayfa adi kurallarina uygun, benzersiz bir isim uretir
    (max 31 karakter, : \\ / ? * [ ] karakterleri yasak)."""
    bad = set(':\\/?*[]')
    s = "".join(c for c in str(base) if c not in bad).strip()
    if not s: s = "Sheet"
    s = s[:31]
    orig = s
    i = 1
    while s in used:
        suffix = "_{0}".format(i)
        s = (orig[:max(1, 31 - len(suffix))] + suffix)
        i += 1
    used.add(s)
    return s

def _mac_find_mode_families(mac, mode_list, seed_indices, threshold):
    """[YENI -- MOD AILESI TAKIBI] Bir body'nin kendi (dominant + SE esigini
    gecen) modlarini 'tohum' (seed) olarak alir, her tohumdan HEM ILERI HEM
    GERI yonde, HI ekseninde ADIM ADIM (komsu HI'ye) TUM mode_list icinde
    (sadece bu body'ye ozel degil) en yuksek MAC degerine sahip modu arar.
    Bu deger threshold'u gecerse aileye eklenir (ODUNC/borrowed -- baska
    bir body'de dominant olabilir) ve takip oradan devam eder; gecmezse o
    yonde durur (bosluk -- mode veering/crossing sinyali sayilir, zorla
    doldurulmaz).

    ONEMLI: Bu algoritma 'bagli bilesen' (transitive/union-find) DEGILDIR
    -- A-B ve B-C eslesse bile A-C dogrudan eslesmiyorsa A ve C ayni
    aileye ZORLA sokulmaz, cunku takip HER ADIMDA sadece bir ONCEKI
    komsuyla kiyaslar (kademeli drift'i doğal olarak takip eder, zincirleme
    hatasi riskini ortadan kaldirir).

    Parametreler:
      mac          : tam NxN MAC matrisi (liste-liste)
      mode_list    : [(sid, hi, freq), ...] -- mac ile AYNI sirada
      seed_indices : bu body'nin dominant+esik-gecen mode_list indeksleri
      threshold    : MAC esigi (ana MAC esigiyle AYNI kullanilir)

    Donus: [ [(hi, freq, idx, is_own), ...HI'ye gore sirali], ... ]
           is_own=True  -> bu nokta body'nin KENDI tohumu (dolu marker)
           is_own=False -> takip sirasinda tam matristen bulunan, baska
                           bir body'de dominant olabilecek nokta (soluk marker)
    """
    seed_set = set(seed_indices)
    if not seed_set:
        return []

    by_hi = {}
    for i, (sid, hi, fr) in enumerate(mode_list):
        by_hi.setdefault(hi, []).append(i)
    hi_values = sorted(by_hi.keys())

    visited = set()
    families = []

    for seed in sorted(seed_set, key=lambda i: mode_list[i][1]):
        if seed in visited:
            continue
        family = {seed: True}
        visited.add(seed)
        seed_hi = mode_list[seed][1]

        # ILERI yonde takip (HI artan)
        cur = seed
        for hi in [h for h in hi_values if h > mode_list[cur][1]]:
            candidates = [i for i in by_hi[hi] if i not in family]
            if not candidates:
                continue
            best = max(candidates, key=lambda i: mac[cur][i])
            if mac[cur][best] < threshold:
                break
            family[best] = (best in seed_set)
            visited.add(best)
            cur = best

        # GERI yonde takip (HI azalan), tohumdan baslayarak
        cur = seed
        for hi in [h for h in reversed(hi_values) if h < mode_list[cur][1]]:
            candidates = [i for i in by_hi[hi] if i not in family]
            if not candidates:
                continue
            best = max(candidates, key=lambda i: mac[cur][i])
            if mac[cur][best] < threshold:
                break
            family[best] = (best in seed_set)
            visited.add(best)
            cur = best

        pts = sorted(
            [(mode_list[i][1], mode_list[i][2], i, is_own) for i, is_own in family.items()],
            key=lambda t: t[0])
        families.append(pts)

    return families

def _mac_family_sheet_xml(mac, labels, threshold, colw, families, full_labels):
    """[YENI -- MOD AILESI TAKIBI] _mac_matrix_sheet_xml() ile AYNI MAC
    alt-matrisini uretir, GRAFIGIN YANINA (matrisin degil -- grafik
    row 27'den basliyor, bkz. _mac_family_drawing_xml) aile veri
    tablosunu ekler.

    [DEGISIKLIK -- TABLO FORMATI SADELESTIRILDI] Eskiden her aile KENDI
    sutun blogunda YAN YANA duruyordu (Family1 | Family2 | Family3 ...),
    bu da parca parca, tutarsiz yukseklikte kutular halinde 'karisik'
    goruntu veriyordu. Artik TEK BIR DIKEY TABLO var: 4 sabit sutun
    (Family, Freq (Hz), HI, Mode), TUM ailelerin noktalari ALT ALTA
    (satir satir) siralaniyor. Her ailenin 'Family N' etiketi, o ailenin
    kapladigi satir araligi boyunca DIKEY OLARAK BIRLESTIRILIYOR
    (mergeCell) -- standart, temiz bir tablo gorunumu.

    'Mode' sutunu her noktanin ADINI (ornegin 'HI 5 Mod 13') gosterir;
    bu ozellikle BORROWED (odunc) noktalar icin kritik.

    [DUZELTME] Matrisin KENDI baslik hucresi (row1, 'MAC MATRIX
    Threshold:...') bu fonksiyonda _mac_matrix_rows() ile geliyor ama
    eskiden BIRLESTIRILMEMISTI (sadece _mac_matrix_sheet_xml'de
    birlestiriliyordu, bu fonksiyon kendi sheet XML'ini ayri kurdugu
    icin o duzeltme buraya YANSIMAMISTI) -- artik burada da matris
    genisligi kadar birlestiriliyor.

    full_labels: TAM (tum mode_list) etiket listesi -- family noktalari
    baska bir body'den (borrowed) gelebildigi icin, bu body'nin KENDI
    'labels' listesi (sadece bu body'nin modlari) YETMEZ, tam listeye
    ihtiyac var.

    Satir cakismasi Excel'i bozdugu icin, matris ve aile tablosu
    hucreleri TEK BIR birlesik satir sozlugunde toplanir.

    Donus: (sheet_xml, col_freq, col_hi, base_row, table_top_row) --
    chart'in hucre referanslarini olusturmasi icin gereken sutun/satir
    bilgileridir (artik her aile AYNI iki sutunu (Freq, HI) FARKLI satir
    araliklarinda kullanir)."""
    n = len(labels)
    rows = _mac_matrix_rows(mac, labels, threshold)   # {satir: {sutun: xml}} -- matris hucreleri

    # [DUZELTME] Matrisin kendi baslik hucresini de (row1) matris
    # genisligi kadar birlestir -- ONCEDEN SADECE _mac_matrix_sheet_xml
    # icinde yapiliyordu, bu fonksiyon (family'li sayfalar) icin
    # UNUTULMUSTU. [BUYUTME] Artik 2 SATIR boyunca birlestiriliyor (daha
    # once sadece 1 satirdi), boylece daha buyuk fontla da rahat sigar.
    _merges = [("R", 1, max(n, 1) + 1, 1, 2)]   # (etiket, col0, col1, row0, row1)

    # Grafik (bkz. _mac_family_drawing_xml) row 27'den (0-based row 26)
    # basliyor -- tablo da AYNI satirdan baslar, boylece grafigin TAM
    # YANINDA gorunur. Sutun ise matris genisliginden bagimsiz, sabit
    # bir minimumla (17) baslar -- grafigin sag kenariyla (col O=15)
    # arasinda bosluk kalsin, buyuk matrislerle CAKISMASIN.
    base_row = 27
    col_family = max(17, n + 3)
    col_freq   = col_family + 1
    col_hi     = col_family + 2
    col_mode   = col_family + 3

    def put(r, c, xml):
        rows.setdefault(r, {})[c] = xml

    put(base_row, col_family, _mac_cs(_mac_col_letter(col_family), base_row,
        "MODE FAMILY TRACKING (MAC >= {0:.2f})".format(threshold), 1))
    put(base_row + 1, col_family, _mac_cs(_mac_col_letter(col_family), base_row + 1,
        "RED = this body's OWN mode (>= SE threshold)   "
        "BLUE = borrowed point, dominant in a DIFFERENT body   "
        "ORANGE = isolated mode (no correlated neighbor)", 1))
    _merges.append((base_row,     col_family, col_mode))
    _merges.append((base_row + 1, col_family, col_mode))

    header_row = base_row + 2
    for off, hdr in enumerate(["Family", "Freq (Hz)", "HI", "Harmonic Index and Mode"]):
        put(header_row, col_family + off, _mac_cs(_mac_col_letter(col_family + off), header_row, hdr, 1))

    r = header_row + 1
    for k, fam in enumerate(families):
        n_pts = len(fam)
        if n_pts == 0:
            continue
        note = ""
        if n_pts == 1:
            note = "  (isolated)"
        fam_start_row = r
        # [DEGISIKLIK] Tablo (ve dolayisiyla chart'in okudugu veri) artik
        # frekansa gore DEGIL, KUCUK HI'den BUYUK HI'ye, ayni HI icinde de
        # KUCUK MOD numarasindan BUYUGE dogru siralanir (ornegin 'HI 0
        # Mod 3', 'HI 0 Mod 5'ten ONCE gelir).
        fam_sorted = sorted(fam, key=lambda t: (
            t[0], _mac_parse_mode_no(full_labels[t[2]] if 0 <= t[2] < len(full_labels) else "")))
        is_isolated = (n_pts == 1)
        for (hi, fr, idx, is_own) in fam_sorted:
            mode_name = full_labels[idx] if 0 <= idx < len(full_labels) else "?"
            # [YENI] Tek nokta (izole) aileler KIRMIZI DEGIL, ayirt edici
            # bir TURUNCU renk alir -- kirmizi/mavi SADECE gercek bir
            # aile/trend icindeki own/borrowed ayrimi icin kullanilir.
            if is_isolated:
                row_style = 7   # 7=turuncu (izole mod)
            else:
                row_style = 3 if is_own else 6   # 3=kirmizi(own), 6=mavi(borrowed)
            put(r, col_freq, _mac_cn(_mac_col_letter(col_freq), r, fr, row_style))
            put(r, col_hi,   _mac_cn(_mac_col_letter(col_hi),   r, hi, row_style))
            put(r, col_mode, _mac_cs(_mac_col_letter(col_mode), r, mode_name, row_style))
            r += 1
        fam_end_row = r - 1
        # 'Family N (M data points)' etiketi, o ailenin kapladigi TUM
        # satir araligi boyunca DIKEY OLARAK birlestirilir.
        put(fam_start_row, col_family, _mac_cs(_mac_col_letter(col_family), fam_start_row,
            "Family {0} ({1} data points){2}".format(k + 1, n_pts, note), 1))
        if fam_end_row > fam_start_row:
            _merges.append((None, None, None))   # placeholder, asagida duzeltiliyor
            _merges[-1] = ("V", col_family, fam_start_row, fam_end_row)   # dikey birlesim isareti

    table_bottom_row = r - 1

    _merge_cells_xml_parts = []
    for m in _merges:
        if len(m) == 5:
            (_tag, c0, c1, r0, r1) = m
            _merge_cells_xml_parts.append('<mergeCell ref="{0}{1}:{2}{3}"/>'.format(
                _mac_col_letter(c0), r0, _mac_col_letter(c1), r1))
        elif len(m) == 3:
            (mrow, c0, c1) = m
            _merge_cells_xml_parts.append('<mergeCell ref="{0}{1}:{2}{1}"/>'.format(
                _mac_col_letter(c0), mrow, _mac_col_letter(c1)))
        else:
            (_tag, mcol, r0, r1) = m
            _merge_cells_xml_parts.append('<mergeCell ref="{0}{1}:{0}{2}"/>'.format(
                _mac_col_letter(mcol), r0, r1))
    _merge_cells_xml = ""
    if _merge_cells_xml_parts:
        _merge_cells_xml = ('<mergeCells count="{0}">'.format(len(_merge_cells_xml_parts)) +
                           "".join(_merge_cells_xml_parts) + '</mergeCells>')

    # [DEGISIKLIK] HI sutunu artik GORUNMUYOR (hidden="1") -- kullanicinin
    # istegi uzerine, bu bilgi zaten 'Harmonic Index and Mode' sutununda
    # (ornegin 'HI 5 Mod 15') metin olarak mevcut. HI sutunu SILINMEDI --
    # grafigin X ekseni (Harmonic Index) hala bu sutundan okuyor, sadece
    # kullaniciya gorunmez yapildi.
    _fam_cols_xml = (
        '<col min="{0}" max="{0}" width="20" customWidth="1"/>'.format(col_family) +
        '<col min="{0}" max="{0}" width="14" customWidth="1"/>'.format(col_freq) +
        '<col min="{0}" max="{0}" width="8" customWidth="1" hidden="1"/>'.format(col_hi) +
        '<col min="{0}" max="{0}" width="24" customWidth="1"/>'.format(col_mode)
    )

    sheet_xml = (
        '<?xml version="1.0" encoding="UTF-8" standalone="yes"?>' +
        '<worksheet xmlns="http://schemas.openxmlformats.org/spreadsheetml/2006/main"' +
        ' xmlns:r="http://schemas.openxmlformats.org/officeDocument/2006/relationships">' +
        '<cols>' +
        '<col min="1" max="1" width="' + str(colw) + '" customWidth="1"/>' +
        '<col min="2" max="' + str(max(n, 1) + 1) + '" width="' + str(colw) + '" customWidth="1"/>' +
        _fam_cols_xml +
        '</cols>' +
        '<sheetData>' + _mac_rows_dict_to_xml(rows) + '</sheetData>' +
        _merge_cells_xml +
        '<drawing r:id="rId1"/>' +
        '</worksheet>'
    )
    return sheet_xml, col_freq, col_hi, header_row

def _mac_family_chart_xml(sheet_name, families, col_freq_num, col_hi_num, full_labels, header_row=29):
    """[YENI -- MOD AILESI TAKIBI] X=Harmonic Index, Y=Frekans(Hz) scatter
    chart'i uretir. Her aile icin TEK seri:
      - Cizgi rengi: aileyi ayirt eden palet rengi (kirmizi/mavi ASLA
        kullanilmaz, karisiklik olmasin diye -- kullanicinin talebi).
      - Her NOKTA <c:dPt> ile ayri ayri renklendirilir:
          own=True  (body'nin kendi modu, SE esigini gecmis) -> KIRMIZI
          own=False (baska body'den odunc, takip icin bulunmus) -> MAVI

    [DEGISIKLIK -- TABLO FORMATI SADELESTIRILDI] Artik her aile KENDI
    sutun blogunda degil, SABIT IKI sutunda (Freq, HI) FARKLI SATIR
    araliklarinda duruyor (bkz. _mac_family_sheet_xml) -- bu fonksiyon
    da AYNI satir-aralik hesaplamasini (header_row+1'den baslayarak,
    her aile n_pts kadar satir kaplar) birebir tekrarlar, boylece
    referanslar TABLO ILE HER ZAMAN TUTARLI kalir.

    [DUZELTME] Az sayida nokta (ozellikle 1 aile x 1 nokta gibi) oldugunda
    Excel'in OTOMATIK eksen olceklemesi bozuk/degenerate bir aralik
    secebiliyor. Bunu onlemek icin eksen min/max degerleri VERIDEN
    MANUEL hesaplanip chart'a yaziliyor.

    [DUZELTME] Tek noktali aileler (izole mod) icin seri-seviyesi marker
    'none' birakilirsa legend'da HICBIR SEY gorunmuyordu -- artik
    n_pts==1 ise seri seviyesinde GORUNUR bir marker kullaniliyor.

    [BUYUTME] Kullanicinin istegi uzerine nokta boyutu 7'den 11'e
    cikarildi -- grafikte kirmizi/mavi/turuncu noktalar artik daha
    belirgin."""
    sn = "'{0}'".format(sheet_name.replace("'", "''"))
    MARKER_SIZE = 11   # eskiden 7 idi -- noktalar daha buyuk/belirgin olsun
    col_freq = _mac_col_letter(col_freq_num)
    col_hi   = _mac_col_letter(col_hi_num)
    # Aile CIZGI paleti -- KIRMIZI (FF0000/D62728 vb.) ve MAVI (0070C0 vb.)
    # BILEREK YOK, cunku bu iki renk artik SADECE nokta durumunu (own/
    # borrowed) gostermek icin ayrilmis; cizgiyle karismasinlar diye.
    palette = ["2CA02C", "9467BD", "FF7F0E", "17BECF", "8C564B",
              "E377C2", "7F7F7F", "BCBD22", "556B2F", "8B4513"]
    COLOR_OWN = "FF0000"       # kirmizi -- body'nin kendi modu (SE esigi ustu)
    COLOR_BORROWED = "0070C0"  # mavi -- odunc/takip noktasi

    all_his = [hi for fam in families for (hi, fr, idx, is_own) in fam]
    all_freqs = [fr for fam in families for (hi, fr, idx, is_own) in fam]
    if all_his and all_freqs:
        # [DUZELTME] min()/max() int HI degerleri uzerinde calisirsa
        # sonuc INT olarak kalabilir; asagidaki .2f formatlamasi
        # IronPython'da int'e uygulanirsa "Precision not allowed in
        # integer format specifier" hatasi verir (daha once sum() ile
        # ayni hatayi gormustuk) -- bu yuzden HER ADIMDA float() ile
        # ACIKCA float'a cevriliyor.
        hi_min, hi_max = float(min(all_his)), float(max(all_his))
        fr_min, fr_max = float(min(all_freqs)), float(max(all_freqs))
        hi_pad = max(1.0, (hi_max - hi_min) * 0.15)
        fr_pad = max(10.0, (fr_max - fr_min) * 0.10)
        axis_hi_min = max(0.0, hi_min - hi_pad)
        axis_hi_max = hi_max + hi_pad
        axis_fr_min = max(0.0, fr_min - fr_pad)
        axis_fr_max = fr_max + fr_pad
    else:
        axis_hi_min, axis_hi_max = 0.0, 10.0
        axis_fr_min, axis_fr_max = 0.0, 1000.0

    COLOR_ISOLATED = "E67E22"  # turuncu -- tek noktali (izole) aileler icin, kirmizi/mavi DEGIL

    series_xml = ""
    si = 0
    r = header_row + 1   # _mac_family_sheet_xml'deki AYNI baslangic satiri
    for k, fam in enumerate(families):
        n_pts = len(fam)
        if n_pts == 0:
            continue
        line_color = palette[k % len(palette)]
        # [DEGISIKLIK] Tablo (ve dolayisiyla bu chart'in okudugu hucreler)
        # artik frekansa DEGIL, KUCUK HI'den BUYUGE, ayni HI icinde de
        # KUCUK MOD numarasindan BUYUGE siralanir -- dPt (nokta rengi)
        # indeksleri bu SIRALI listeye gore hesaplanmali, yoksa renkler
        # yanlis satirlara denk gelir.
        fam_sorted = sorted(fam, key=lambda t: (
            t[0], _mac_parse_mode_no(full_labels[t[2]] if 0 <= t[2] < len(full_labels) else "")))
        is_isolated = (n_pts == 1)
        r0, r1 = r, r + n_pts - 1
        r += n_pts

        line_xml = ('<a:ln w="19050"><a:solidFill><a:srgbClr val="{0}"/></a:solidFill></a:ln>'.format(line_color)
                   if n_pts >= 2 else '<a:ln><a:noFill/></a:ln>')

        # Her nokta icin ayri renk -- own=kirmizi, borrowed=mavi; AMA
        # aile TEK NOKTALI (izole) ise kirmizi/mavi yerine TURUNCU
        # kullanilir (bu, gercek bir aile/trend icindeki own/borrowed
        # ayrimindan FARKLI bir anlam tasir -- 'bu mod hicbir komsu
        # HI'de eslesme bulamadi', kullanicinin talebi).
        dpts_xml = ""
        for pidx, (hi, fr, idx, is_own) in enumerate(fam_sorted):
            if is_isolated:
                pt_color = COLOR_ISOLATED
            else:
                pt_color = COLOR_OWN if is_own else COLOR_BORROWED
            dpts_xml += (
                '<c:dPt><c:idx val="{i}"/><c:invertIfNegative val="0"/>'
                '<c:marker><c:symbol val="circle"/><c:size val="{sz}"/>'
                '<c:spPr><a:solidFill><a:srgbClr val="{c}"/></a:solidFill>'
                '<a:ln w="9525"><a:solidFill><a:srgbClr val="000000"/></a:solidFill></a:ln>'
                '</c:spPr></c:marker></c:dPt>'
            ).format(i=pidx, c=pt_color, sz=MARKER_SIZE)

        # [DUZELTME] Seri SEVIYESINDE marker, aile 2+ nokta iceriyorsa
        # BILEREK 'none' -- gercek noktalar zaten yukaridaki <c:dPt> ile

def _mac_matrix_rows(mac, labels, threshold):
    """[DUZELTME] Matris hucrelerini {satir_no: {sutun_no: cell_xml}}
    sozlugu olarak dondurur -- hem duz MAC matris sayfasi hem de aile
    grafikli body sayfasi bu OTAK sozlugu kullanir, boylece iki farkli
    icerik (matris + aile tablosu) AYNI satir numarasini paylassa bile
    TEK bir <row> elemaninda birlesir (Excel'in AYNI r degerine sahip
    IKI ayri <row> elemanini gecersiz sayip dosyayi bozuk isaretlemesini
    onler -- eski hata buydu)."""
    n = len(labels)
    rows = {}
    def put(r, c, xml):
        rows.setdefault(r, {})[c] = xml

    esik = "MAC MATRIX  Threshold:{0:.2f}  Red>=threshold  Yellow>=0.5  Green=Diagonal".format(threshold)
    put(1, 1, _mac_cs("A", 1, esik, 8))

    put(3, 1, _mac_cs("A", 3, "MODE", 1))
    for j in range(n):
        put(3, j + 2, _mac_cs(_mac_col_letter(j + 2), 3, labels[j], 1))

    for i in range(n):
        r = i + 4
        put(r, 1, _mac_cs("A", r, labels[i], 1))
        for j in range(n):
            v = mac[i][j]
            if i == j:          s = 2
            elif v >= threshold: s = 3
            elif v >= 0.50:      s = 4
            else:                s = 5
            put(r, j + 2, _mac_cn(_mac_col_letter(j + 2), r, v, s))

    if n == 0:
        put(4, 1, _mac_cs("A", 4, "No modes meet the criteria for this sheet.", 5))

    return rows

def _mac_rows_dict_to_xml(rows):
    """{satir_no: {sutun_no: cell_xml}} sozlugunu <row> elemanlarina
    cevirir -- her satir numarasi icin TEK VE SADECE TEK <row> uretir."""
    rows_xml = []
    for r in sorted(rows.keys()):
        cells = rows[r]
        rows_xml.append('<row r="{0}">{1}</row>'.format(
            r, "".join(cells[c] for c in sorted(cells.keys()))))
    return "".join(rows_xml)

def _mac_matrix_sheet_xml(mac, labels, threshold, colw):
    """Bir MAC matrisi (tam veya alt-kume) icin sheetData XML'ini uretir --
    hem ana 'MAC Matrix' sayfasi hem de body-bazli alt-matris sayfalari
    icin ORTAK olarak kullanilir.

    [DUZELTME] Baslik hucresi (row1) eskiden TEK bir dar sutunda (A1)
    duruyordu; uzun aciklama metni bu dar sutuna sigmayip dikey olarak
    akiyordu (cok uzun/dar bir hucre gibi gorunuyordu). Artik matrisin
    TUM genisligi kadar BIRLESTIRILIYOR (mergeCells) -- metin normal
    genislikte, birkac satirda okunakli sekilde goruntuleniyor."""
    n = len(labels)
    rows = _mac_matrix_rows(mac, labels, threshold)
    last_col_letter = _mac_col_letter(max(n, 1) + 1)
    merge_xml = '<mergeCells count="1"><mergeCell ref="A1:{0}2"/></mergeCells>'.format(last_col_letter)
    sheet_xml = (
        '<?xml version="1.0" encoding="UTF-8" standalone="yes"?>' +
        '<worksheet xmlns="http://schemas.openxmlformats.org/spreadsheetml/2006/main"' +
        ' xmlns:r="http://schemas.openxmlformats.org/officeDocument/2006/relationships">' +
        '<cols>' +
        '<col min="1" max="1" width="' + str(colw) + '" customWidth="1"/>' +
        '<col min="2" max="' + str(max(n, 1) + 1) + '" width="' + str(colw) + '" customWidth="1"/>' +
        '</cols>' +
        '<sheetData>' + _mac_rows_dict_to_xml(rows) + '</sheetData>' +
        merge_xml +
        '</worksheet>'
    )
    return sheet_xml

    _warnings = []

                        # indeksleriyle calisan aile verisini SUB-matris
                        # etiketlerine gore yeniden kur -- idx degerleri
                        # dogrudan labels[]/mode_list[] icin GECERLIDIR.
                        sxml, col_freq, col_hi, header_row = _mac_family_sheet_xml(
                            sub_mac, sub_labels, threshold, _colw, fams, labels)
                        sheet_idx = 2 + len(_extra_sheet_xmls) + 1
                        chart_xml = _mac_family_chart_xml(sname, fams, col_freq, col_hi, labels, header_row=header_row)
                        drawing_xml = _mac_family_drawing_xml()
                        drawing_rels_xml = _mac_family_drawing_rels_xml("chart{0}.xml".format(sheet_idx))
                        sheet_rels_xml = _mac_family_sheet_rels_xml("drawing{0}.xml".format(sheet_idx))
                        chart_info = (sheet_idx, chart_xml, drawing_xml, drawing_rels_xml, sheet_rels_xml)
                    except Exception as ex:
                        # [ONEMLI] Mod ailesi grafigi bu body icin herhangi
                        # bir sebeple basarisiz olursa, TUM MAC dosyasinin
                        # uretilmesini ENGELLEMEZ -- bu body icin sadece
                        # grafiksiz duz matrise GERI DUSULUR. Kok neden hala
                        # arastiriliyor olsa bile kullanici en azindan
                        # calisan bir MAC_results.xlsx alir.
                        print("  MAC WARNING: mode family chart FAILED for body "
                             "'{0}', falling back to plain matrix (no chart) "
                             "for this sheet -- {1}".format(body_name, ex))
                        sxml = None
                        chart_info = None

                if sxml is None:
                    sxml = _mac_matrix_sheet_xml(sub_mac, sub_labels, threshold, _colw)

                return sname, sxml, chart_info

            sname, sxml, chart_info = _sub_phase(
                "building body sheet '{0}'".format(body_name), _build_one_body_sheet)
            if chart_info is not None:
                sheet_idx, chart_xml, drawing_xml, drawing_rels_xml, sheet_rels_xml = chart_info
                _chart_parts[sheet_idx] = (chart_xml, drawing_xml, drawing_rels_xml, sheet_rels_xml)
            _extra_sheet_xmls.append((sname, sxml))

    # ---- OOXML iskeletini DINAMIK sayfa sayisina gore olustur ----
    def _build_skeleton():
        all_sheet_names = ["MAC Matrix", "Warnings"] + [nm for (nm, _) in _extra_sheet_xmls]
        n_sheets = len(all_sheet_names)

        _CT_overrides = "".join(
            '<Override PartName="/xl/worksheets/sheet{0}.xml" '
            'ContentType="application/vnd.openxmlformats-officedocument.spreadsheetml.worksheet+xml"/>'.format(k + 1)
            for k in range(n_sheets))
        # [YENI] her grafikli sayfa icin drawing + chart Content_Types girisi
        _CT_chart_overrides = "".join(
            '<Override PartName="/xl/drawings/drawing{0}.xml" '
            'ContentType="application/vnd.openxmlformats-officedocument.drawing+xml"/>'
            '<Override PartName="/xl/charts/chart{0}.xml" '
            'ContentType="application/vnd.openxmlformats-officedocument.drawingml.chart+xml"/>'.format(k)
            for k in sorted(_chart_parts.keys()))
        _CT = (
            '<?xml version="1.0" encoding="UTF-8" standalone="yes"?>' +
            '<Types xmlns="http://schemas.openxmlformats.org/package/2006/content-types">' +
            '<Default Extension="rels" ContentType="application/vnd.openxmlformats-package.relationships+xml"/>' +
            '<Default Extension="xml" ContentType="application/xml"/>' +
            '<Override PartName="/xl/workbook.xml" ContentType="application/vnd.openxmlformats-officedocument.spreadsheetml.sheet.main+xml"/>' +
            _CT_overrides + _CT_chart_overrides +
            '<Override PartName="/xl/styles.xml" ContentType="application/vnd.openxmlformats-officedocument.spreadsheetml.styles+xml"/>' +
            '</Types>'
        )
        _RELS = (
            '<?xml version="1.0" encoding="UTF-8" standalone="yes"?>' +
            '<Relationships xmlns="http://schemas.openxmlformats.org/package/2006/relationships">' +
            '<Relationship Id="rId1" Type="http://schemas.openxmlformats.org/officeDocument/2006/relationships/officeDocument" Target="xl/workbook.xml"/>' +
            '</Relationships>'
        )
        _wbrels_parts = "".join(
            '<Relationship Id="rId{0}" Type="http://schemas.openxmlformats.org/officeDocument/2006/relationships/worksheet" '
            'Target="worksheets/sheet{0}.xml"/>'.format(k + 1)
            for k in range(n_sheets))
        _styles_rid = n_sheets + 1
        _WBRELS = (
            '<?xml version="1.0" encoding="UTF-8" standalone="yes"?>' +
            '<Relationships xmlns="http://schemas.openxmlformats.org/package/2006/relationships">' +
            _wbrels_parts +
            '<Relationship Id="rId{0}" Type="http://schemas.openxmlformats.org/officeDocument/2006/relationships/styles" '
            'Target="styles.xml"/>'.format(_styles_rid) +
            '</Relationships>'
        )
        _sheets_xml = "".join(
            '<sheet name="{0}" sheetId="{1}" r:id="rId{1}"/>'.format(_mac_xml_escape(nm), k + 1)
            for k, nm in enumerate(all_sheet_names))
        _WB = (
            '<?xml version="1.0" encoding="UTF-8" standalone="yes"?>' +
            '<workbook xmlns="http://schemas.openxmlformats.org/spreadsheetml/2006/main"' +
            ' xmlns:r="http://schemas.openxmlformats.org/officeDocument/2006/relationships">' +
            '<sheets>' + _sheets_xml + '</sheets></workbook>'
        )
        return all_sheet_names, n_sheets, _CT, _RELS, _WBRELS, _WB

    all_sheet_names, n_sheets, _CT, _RELS, _WBRELS, _WB = _sub_phase(
        "building workbook skeleton (Content_Types/rels/workbook.xml)", _build_skeleton)

    def _ensure_out_dir():
        _out_dir = os.path.dirname(out_path)
        if _out_dir and not os.path.exists(_out_dir):
            os.makedirs(_out_dir)

    _sub_phase("creating output directory", _ensure_out_dir)

    all_sheet_xmls = [_s1, _s2] + [xml for (_, xml) in _extra_sheet_xmls]

    def _clean(s):
        """[SAVUNMA] Bazi kaynaklardan (ornegin registry okumalari, .NET
        string donusleri) beklenmedik bir NULL byte (\\x00) sizabiliyor;
        bu, .NET'in zip/deflate yazicisinda dusuk seviyeli, okunaksiz bir
        hataya ("unknown", "\\x00", 0, 1, "") yol acabiliyor. Yazmadan
        ONCE her XML govdesinden NULL byte'lari temizleriz -- bu hem
        olasi kok nedeni ortadan kaldirir hem de zararsizdir (gecerli
        XML icerigi hicbir zaman NULL byte icermemeli)."""
        return s.replace("\x00", "") if isinstance(s, str) else s

    def _write_part(zf, name, content_str):
        try:
            zf.writestr(name, _clean(content_str).encode("utf-8"))
        except Exception as ex:
            raise Exception("Failed writing zip part '{0}': {1}".format(name, ex))

    with _zipfile.ZipFile(out_path, "w", _zipfile.ZIP_DEFLATED) as _zf:
        _write_part(_zf, "[Content_Types].xml",        _CT)
        _write_part(_zf, "_rels/.rels",                _RELS)
        _write_part(_zf, "xl/workbook.xml",            _WB)
        _write_part(_zf, "xl/_rels/workbook.xml.rels", _WBRELS)

            _write_part(_zf, "xl/worksheets/sheet{0}.xml".format(k + 1), sxml)
        # [YENI] her grafikli sayfa icin: sheet rels + drawing + drawing rels + chart
        for sheet_idx, (chart_xml, drawing_xml, drawing_rels_xml, sheet_rels_xml) in _chart_parts.items():
            _write_part(_zf, "xl/worksheets/_rels/sheet{0}.xml.rels".format(sheet_idx), sheet_rels_xml)
            _write_part(_zf, "xl/drawings/drawing{0}.xml".format(sheet_idx), drawing_xml)
            _write_part(_zf, "xl/drawings/_rels/drawing{0}.xml.rels".format(sheet_idx), drawing_rels_xml)
            _write_part(_zf, "xl/charts/chart{0}.xml".format(sheet_idx), chart_xml)
        _write_part(_zf, "xl/styles.xml", _styles)

    print("  MAC Excel written: {0}  ({1} sheet(s): {2})".format(
        out_path, n_sheets, ", ".join(all_sheet_names)))

def run_mac_module(rst_path, threshold, twin_hz, out_dir, hi_filter=None,
                   prog=None, dpf=None, ds=None, model=None,
                   energy=None, se_thr=None, parts=None):
    """Ana akistan cagrilir. RST'den DPF ile mode shape okur (TOPLU - tek
    operator cagrisi), sadece hi_filter icindeki Harmonic Index'lere ait
    modlari kullanir, MAC'i paralel hesaplar, MAC_results.xlsx yazar.
    Kendi twin elemesini yapar (EM verisi yoktur).

    hi_filter: None ise tum HI'ler dahil edilir; aksi halde sadece
    verilen HI degerlerine sahip modlar taramaya dahil edilir.

    energy/se_thr/parts: VERILIRSE (strain energy hesaplanmissa), MAC
    Excel'ine ana sayfalarin (MAC Matrix + Warnings) yaninda HER BODY icin
    ayri bir sayfa eklenir -- o sayfada SADECE o body'nin DOMINANT oldugu
    (en yuksek SE oranina sahip oldugu) VE o oranin se_thr'yi ASTIGI
    modlari iceren kucultulmus bir MAC alt-matrisi bulunur. energy None
    ise (strain energy hesaplanmadiysa) body sayfalari atlanir, sadece
    eski 2 sayfa (MAC Matrix + Warnings) uretilir.

    dpf/ds/model: onceden yuklenmisse (ornegin MacPopupDialog icinde HI
    listesi gosterilirken RST zaten acildiysa) tekrar RST acmamak icin
    disaridan verilebilir."""
    out_path = os.path.join(out_dir, "MAC_results.xlsx")

    def _phase(name, fn):
        """Bir asamayi calistirir; hata olursa HANGI asamada oldugunu
        acikca belirten bir mesajla yeniden firlatir. Bazi .NET/COM
        kaynakli hatalar Python'da anlamsiz, okunaksiz bir tuple metni
        olarak gorunur (ornegin "('unknown', '\\x00', 0, 1, '')") --
        boyle durumlarda EN AZINDAN hangi adimda oldugunu bilmek,
        kor tahmin yerine gercek teshis yapmayi mumkun kilar."""
        try:
            return fn()
        except Exception as ex:
            raise Exception("[MAC phase: {0}] {1}".format(name, ex))

    if dpf is None or ds is None or model is None:
        if prog: prog.update("MAC: loading RST (DPF)...", os.path.basename(rst_path))
        dpf, ds, model = _phase("loading RST (DPF)", lambda: _mac_dpf_load(rst_path))

    if prog: prog.update("MAC: reading modes from TimeFreqSupport...")
    raw_modes_all = _phase("reading modes from TimeFreqSupport",
                           lambda: _mac_dpf_discover_modes(dpf, ds, model))
    print("MAC: number of sets found: {0}".format(len(raw_modes_all)))
    if not raw_modes_all:
        raise Exception("No modes found in RST.")

    # sid -> (hi, raw_mode_local) eslemesi -- TUM setler, HICBIR filtre/
    # siralama uygulanmadan ONCE, natural (sid artan) sirada kuruluyor.
    # Bu, build_hi_mode_to_setid() ile AYNI
    # numaralandirmayi verir (HI blogu icinde 1'den baslayan sirali
    # sayac) -- body-bazli MAC sayfalari icin 'energy' sozlugundeki
    # (hi, mode) anahtarlariyla eslesmesi bunu GEREKTIRIR.
    _sid_to_rawmode = {}
    _counters = {}
    for (sid, hi, fr) in raw_modes_all:
        _counters[hi] = _counters.get(hi, 0) + 1
        _sid_to_rawmode[sid] = (hi, _counters[hi])

    raw_modes = raw_modes_all
    if hi_filter is not None:
        hi_set = set(hi_filter)
        raw_modes = [m for m in raw_modes if m[1] in hi_set]
        print("MAC: set count remaining after HI filter: {0}  (HI: {1})".format(
            len(raw_modes), sorted(hi_set)))
        if not raw_modes:
            raise Exception("No modes match the selected Harmonic Indices.")

    raw_modes = list(raw_modes)
    raw_modes.sort(key=lambda x: x[2])

    if prog: prog.update("MAC: eliminating twin modes...", "tolerance {0} Hz".format(twin_hz))
    mode_list, twin_log = _phase("eliminating twin modes",
                                 lambda: _mac_eliminate_twins_simple(raw_modes, twin_hz))
    print("MAC: modes remaining after twin elimination: {0}".format(len(mode_list)))
    labels = _phase("building labels",
                    lambda: _mac_make_labels(mode_list, sid_to_rawmode=_sid_to_rawmode))

    if prog: prog.update("MAC: reading ALL mode shapes (bulk, single call)...", "{0} modes".format(len(mode_list)))
    phis, freqs, node_ids = _phase(
        "reading mode shapes (displacement) from RST",
        lambda: _mac_build_phi_matrix_bulk(dpf, ds, model, mode_list, prog=prog, labels=labels))
    if not node_ids:
        raise Exception("Node list is empty.")

    n_mac = len(mode_list)
    n_pairs_mac = n_mac * (n_mac - 1) // 2
    if prog: prog.update("MAC: computing MAC matrix (parallel, balanced)...",
                         "{0}x{0}  ({1} pairs)".format(n_mac, n_pairs_mac))
    import time as _mac_time
    _t_mac0 = _mac_time.time()
    mac = _phase("computing MAC matrix (parallel)", lambda: _mac_compute(phis, prog=prog))
    _t_mac1 = _mac_time.time()
    print("MAC: matrix computation time: {0:.1f} sec  ({1} pairs)".format(
        _t_mac1 - _t_mac0, n_pairs_mac))

    # ---- Body-bazli sayfalar icin dominant-mod eslemesi ----
    body_sheets = None
    if energy is not None and parts:
        if prog: prog.update("MAC: matching modes to dominant bodies (strain energy)...")

        def _match_bodies():
            body_mode_indices = {p: [] for p in parts}
            n_matched = 0
            for idx, (sid, hi, fr) in enumerate(mode_list):
                raw = _sid_to_rawmode.get(sid)
                if raw is None:
                    continue
                rhi, rmode = raw
                erec = energy.get(rhi, {}).get(rmode)
                if erec is None:
                    continue
                ratios = erec.get("ratio", {})
                if not ratios:
                    continue
                dominant = max(parts, key=lambda p: ratios.get(p, 0.0))
                if ratios.get(dominant, 0.0) >= se_thr:
                    body_mode_indices[dominant].append(idx)
                    n_matched += 1
            print("MAC: {0} mode(s) matched to a dominant body >= SE threshold ({1:.1f}%).".format(
                n_matched, se_thr))
            return [(p, body_mode_indices[p]) for p in parts]

        body_sheets = _phase("matching modes to dominant bodies", _match_bodies)

    # [YENI -- MOD AILESI TAKIBI] Her body'nin kendi (dominant+esik-gecen)
    # modlarini 'tohum' alarak, tam MAC matrisi uzerinden HI ekseninde
    # ileri/geri takip edip mod ailelerini (frekans-vs-HI egrileri) bulur.
    # body_sheets yoksa (strain energy hesaplanmadiysa) body_families de
    # None kalir -- _mac_build_xlsx_multi() bu durumda ONCEKI davranisla
    # (grafiksiz) AYNI sekilde calisir.
    body_families = None
    if body_sheets:
        if prog: prog.update("MAC: tracking mode families across HI...")

        def _track_families():
            fams = {}
            for (p, indices) in body_sheets:
                fams[p] = _mac_find_mode_families(mac, mode_list, set(indices), threshold)
                print("MAC: '{0}' -> {1} mode family/families tracked.".format(p, len(fams[p])))
            return fams

        body_families = _phase("tracking mode families", _track_families)

    if prog: prog.update("MAC: writing Excel...", os.path.basename(out_path))
    _phase("writing MAC Excel file", lambda: _mac_build_xlsx_multi(
        mac, mode_list, threshold, out_path, labels,
        body_sheets=body_sheets, body_families=body_families))

    n_warn = sum(1 for i in range(len(mac)) for j in range(i+1, len(mac)) if mac[i][j] >= threshold)
    return out_path, len(mode_list), n_warn

def _safe_esc(s):
    return s.replace("&", "&amp;").replace("<", "&lt;").replace(">", "&gt;")

def _safe_col_letter(n):
    s = ""
    while n > 0:
        n, r = divmod(n - 1, 26)
        s = chr(65 + r) + s
    return s

def _safe_cellref(row, col):
    return "%s%d" % (_safe_col_letter(col), row)

def _safe_fold(eo, h2):
    period = 2.0 * h2
    m = eo % period
    if m > h2:
        return period - m
    return m

def _safe_parse_out_all(modes_by_hi_input):
    """parse_out() ana koddan gelen formati [{"hi":n,"modes":{m:{"freq":f,...}}}]
    -> SAFE'in bekledigi {hi: [(mode, freq), ...]} formatina cevirir."""
    out = {}
    for block in modes_by_hi_input:
        hi = block["hi"]
        modes = block["modes"]
        out[hi] = [(m, modes[m]["freq"]) for m in sorted(modes.keys())]
    return out

        sel_set = set(self.selected_eos)
        self.lst_avail.Items.Clear()
        self.lst_sel.Items.Clear()
        for eo in range(1, self._max_eo + 1):
            if eo not in sel_set:
                self.lst_avail.Items.Add(str(eo))
        for eo in sorted(sel_set):
            self.lst_sel.Items.Add(str(eo))

    def _parse_eo(self, item_str):
        return int(str(item_str).strip())

    def _on_add(self, sender, args):
        to_add = [self._parse_eo(self.lst_avail.Items[i])
                  for i in range(self.lst_avail.Items.Count)
                  if self.lst_avail.GetSelected(i)]
        for eo in to_add:
            if eo not in self.selected_eos:
                self.selected_eos.append(eo)
        self.selected_eos.sort()
        self._populate()

        to_del = set(self._parse_eo(self.lst_sel.Items[i])
                     for i in range(self.lst_sel.Items.Count)
                     if self.lst_sel.GetSelected(i))
        self.selected_eos = [eo for eo in self.selected_eos if eo not in to_del]
        self._populate()

    def _on_ok(self, sender, args):
        if not self.selected_eos:
            MessageBox.Show("Please select at least one Engine Order.", "Warning")
            return
        self.DialogResult = DialogResult.OK
        self.Close()

        errs = []
        try:
            blade_val = int(self.txt_blade.Text.strip())
            assert blade_val > 0
        except Exception:
            errs.append("Blade count must be a positive integer."); blade_val = 0
        try:
            sector_val = int(self.txt_sector.Text.strip())
            assert sector_val > 0
        except Exception:
            errs.append("Sector must be a positive integer."); sector_val = 0
        try:
            maxhi_val = int(self.txt_maxhi.Text.strip())

        except Exception:
            errs.append("Max Harmonic Index must be a positive integer."); maxhi_val = 0

            errs.append("Margin must be a number between 0 and 100."); margin_val = 0.

        self.sector = sector_val

        self.idle_rpm = idle_val
        self.rl_rpm = rl_val
        self.margin_pct = margin_val

        self.Close()

def run_safe_module(modes_by_hi_raw, blade_count, sector, max_hi,
                    idle_rpm, rl_rpm, margin_pct, critical_eos, out_path):
    """Ana akistan cagrilir. modes_by_hi_raw: ana koddan gelen parsed_clean
    (twin elenmis, [{"hi":n,"modes":{m:{"freq":f,...}}}] formatinda).
    SAFE_Diagram.xlsx yazar."""
    MODES_BY_HI = _safe_parse_out_all(modes_by_hi_raw)

    BLADE_COUNT = blade_count
    SECTOR = sector
    MAX_HI = max_hi
    IDLE_RPM = idle_rpm
    RL_RPM = rl_rpm
    MARGIN_PCT = margin_pct
    CRITICAL_EOS = critical_eos

    H2 = float(MAX_HI)
    MARGIN_FRAC = MARGIN_PCT / 100.
    IDLE_MARGIN_RPM = IDLE_RPM * (1. - MARGIN_FRAC)
    RL_MARGIN_RPM = RL_RPM * (1. + MARGIN_FRAC)

    eo_info = []
    for eo in CRITICAL_EOS:
        nd = _safe_fold(float(eo), H2)
        t = eo * IDLE_MARGIN_RPM / 60.
        u = eo * RL_MARGIN_RPM / 60.
        eo_info.append((eo, nd, t, u))

    max_eo = max(CRITICAL_EOS)
    N_ZZ = max(6, int(_safe_math.ceil(max_eo / H2)) + 1)

    def build_zigzag(rpm):
        pts = [(0., 0.)]
        for n in range(1, N_ZZ + 1):
            nd = 0. if (n % 2 == 0) else H2
            f = n * H2 * rpm / 60.
            pts.append((nd, f))
        return pts

    ZZ_IDLE = build_zigzag(IDLE_RPM)
    ZZ_RL = build_zigzag(RL_RPM)
    ZZ_IDLE_MARGIN = build_zigzag(IDLE_MARGIN_RPM)
    ZZ_RL_MARGIN = build_zigzag(RL_MARGIN_RPM)

    def crossing_eos_for(hi, freq):
        matches = []
        for (eo, nd, t, u) in eo_info:
            if abs(float(hi) - nd) > 0.5:
                continue
            lo, hif = (t, u) if t <= u else (u, t)
            if lo <= freq <= hif:
                matches.append(eo)
        return matches

    HI_LIST = sorted(MODES_BY_HI.keys())
    mode_points = []
    crossing_points = []
    crossing_details = []
    all_mode_freqs = []
    for hi in HI_LIST:
        for (mode_no, freq) in MODES_BY_HI[hi]:
            all_mode_freqs.append(freq)
            matches = crossing_eos_for(hi, freq)
            if matches:
                crossing_points.append((hi, mode_no, freq))
                crossing_details.append((hi, mode_no, freq, matches))
            else:
                mode_points.append((hi, mode_no, freq))

    MODE_MAX_FREQ = max(all_mode_freqs) if all_mode_freqs else 1000.
    AXIS_MAX_FREQ = MODE_MAX_FREQ + 2000.
    AXIS_MAX_ND = H2 + 1.

# CAMPBELL DIAGRAM MODULU
#
# Iki farkli diyagram uretir:
#   1) GENEL (combined) diyagram -- TUM Harmonic Index'lerden gelen,
#      secili EO listesi ile RPM +/- Margin bandi icinde GERCEKTEN
#      kesisen (crossing) modlari TEK bir grafikte birlestirir (her
#      cizgi 'HI x Mod y' etiketli). Kesisen mod yoksa sadece EO+RPM
#      sinir cizgileri gosterilir.
#   2) HER BODY icin AYRI bir sayfa, o sayfada IKI grafik:
#      - 'Dominant + SE Threshold ustu' modlar (mevcut MAC/aile takibi
#        ile AYNI kriter: bu body TUM body'ler arasinda en yuksek SE
#        oranina sahip VE bu oran se_thr'yi geciyor)
#      - 'Tum Dominant' modlar (esik SARTI YOK -- sadece bu body en
#        yuksek SE oranina sahipse yeterli)
#
# OOXML yaklasimi (sharedStrings tabanli) kullanicinin verdigi orijinal
# 'Campbell Diagram Excel Generator' script'inden UYARLANMISTIR; sutun/
# satir harfi ve XML kacis fonksiyonlari MAC modulundekiyle AYNI oldugu
# icin (_mac_col_letter, _mac_xml_escape) TEKRAR YAZILMADI, dogrudan
# kullanilir.
# =============================================================================

def _camp_find_crossing(freq, eos, rpm, rpm_lo, rpm_hi):
    """Tek bir mod frekansi icin, secili EO listesindeki HANGI EO'nun
    RPM +/- Margin bandi (rpm_lo..rpm_hi) icinde kesistigini bulur.
    Birden fazla EO kesisiyorsa, nominal RPM'e EN YAKIN olani secilir.
    Donus: (eo, rpm_cross) veya (None, None)."""
    best = None
    for eo in eos:
        if eo <= 0:
            continue
        rpm_cross = freq * 60.0 / eo
        if rpm_lo <= rpm_cross <= rpm_hi:
            key = abs(rpm_cross - rpm)
            if best is None or key < best[0]:
                best = (key, eo, rpm_cross)
    if best is None:
        return None, None
    return best[1], best[2]

def _camp_build_chart_data(mode_entries, eos, rpm, margin):
    """mode_entries: [(label, freq), ...] -- zaten YUKARIDA (cagiran
    tarafta) hangi modlarin bu grafige dahil olacagi belirlenmis olmali
    (genel/kesisen filtre veya body-bazli dominant filtre).

    _mac_col_letter/_mac_cs/_mac_cn KULLANILMAZ -- Campbell'in kendi
    OOXML yaklasimi (sharedStrings + numeric <c r><v>) farkli oldugu
    icin ham hucre XML'i burada dogrudan uretilir.

    Donus: (rows, meta, crossings) -- rows: {satir_no: {sutun_no: deger}}
    (SADECE SAYISAL degerler, hucre XML'i degil -- cagiran taraf bunlari
    <c r><v> olarak yazar), meta: chart'in ihtiyac duydugu sutun/satir
    indeksleri, crossings: [(label, freq, eo, rpm_cross), ...]."""
    rpm_lo  = rpm * (1.0 - margin)
    rpm_hi  = rpm * (1.0 + margin)
    rpm_max = rpm_hi * 1.15

    n_modes = len(mode_entries)
    n_eos   = len(eos)
    freqs   = [f for (_, f) in mode_entries]
    max_freq = max(freqs) * 1.15 if freqs else 100.0

    C_X     = 1
    C_M1    = 2
    C_EOX   = n_modes + 2
    C_EO1   = n_modes + 3
    C_VL    = n_modes + 3 + n_eos
    C_CROSS = C_VL + 6

    R_DATA1, R_DATA2     = 1, 2
    R_VL_LO1, R_VL_LO2   = 3, 4
    R_VL_NOM1, R_VL_NOM2 = 5, 6
    R_VL_HI1, R_VL_HI2   = 7, 8
    R_CROSS_START        = 9

    crossings = []
    for (label, freq) in mode_entries:
        eo, rpm_cross = _camp_find_crossing(freq, eos, rpm, rpm_lo, rpm_hi)
        if eo is not None:
            crossings.append((label, freq, eo, rpm_cross))

    R_CROSS_END = R_CROSS_START + len(crossings) - 1 if crossings else R_CROSS_START - 1
    max_row = max(R_VL_HI2, R_CROSS_END if crossings else R_DATA2)

    rows = {}
    def setv(r, c, v):
        rows.setdefault(r, {})[c] = v

    setv(R_DATA1, C_X, 0.0); setv(R_DATA1, C_EOX, 0.0)
    for i, (label, f) in enumerate(mode_entries):
        setv(R_DATA1, C_M1 + i, f)
    for i in range(n_eos):
        setv(R_DATA1, C_EO1 + i, 0.0)

    setv(R_DATA2, C_X, rpm_max); setv(R_DATA2, C_EOX, rpm_max)
    for i, (label, f) in enumerate(mode_entries):
        setv(R_DATA2, C_M1 + i, f)
    for i, eo in enumerate(eos):
        setv(R_DATA2, C_EO1 + i, round(eo * rpm_max / 60.0, 2))

    for rpm_v, col_x, col_y, r1, r2 in [
        (rpm_lo, C_VL,   C_VL + 1, R_VL_LO1,  R_VL_LO2),
        (rpm,    C_VL+2, C_VL + 3, R_VL_NOM1, R_VL_NOM2),
        (rpm_hi, C_VL+4, C_VL + 5, R_VL_HI1,  R_VL_HI2),
    ]:
        setv(r1, col_x, rpm_v); setv(r1, col_y, 0.0)
        setv(r2, col_x, rpm_v); setv(r2, col_y, round(max_freq, 0))

    for i, (label, freq, eo, rpm_cross) in enumerate(crossings):
        r = R_CROSS_START + i
        setv(r, C_CROSS,     rpm_cross)
        setv(r, C_CROSS + 1, freq)

    meta = dict(
        n_modes=n_modes, n_eos=n_eos, max_freq=max_freq, rpm_max=rpm_max,
        C_X=C_X, C_M1=C_M1, C_EOX=C_EOX, C_EO1=C_EO1, C_VL=C_VL, C_CROSS=C_CROSS,
        R_DATA1=R_DATA1, R_DATA2=R_DATA2,
        R_VL_LO1=R_VL_LO1, R_VL_LO2=R_VL_LO2,
        R_VL_NOM1=R_VL_NOM1, R_VL_NOM2=R_VL_NOM2,
        R_VL_HI1=R_VL_HI1, R_VL_HI2=R_VL_HI2,
        R_CROSS_START=R_CROSS_START, R_CROSS_END=R_CROSS_END,
        max_row=max_row,
    )
    return rows, meta, crossings

_CAMP_MODE_COLORS = ["1F3864", "C00000", "00B050", "7030A0", "B45F06",

    x = '<c:ser><c:idx val="{0}"/><c:order val="{0}"/>'.format(idx)
    if title is not None:

        x += (
            '<c:dLbls><c:dLbl><c:idx val="1"/>'
            '<c:tx><c:rich><a:bodyPr/><a:lstStyle/><a:p><a:pPr>'
            '<a:defRPr sz="900" b="1"><a:solidFill><a:srgbClr val="{clr}"/></a:solidFill></a:defRPr>'
            '</a:pPr><a:r><a:rPr lang="en-US" b="1"/><a:t>{lbl}</a:t></a:r></a:p></c:rich></c:tx>'

            '<c:showLegendKey val="0"/><c:showVal val="0"/><c:showCatName val="0"/>'
            '<c:showSerName val="0"/><c:showPercent val="0"/><c:showBubbleSize val="0"/></c:dLbls>'
        ).format(lbl=_mac_xml_escape(label_text), clr=color)
    x += '<c:spPr><a:ln w="{0}"><a:solidFill><a:srgbClr val="{1}"/></a:solidFill>'.format(width, color)
    if dash:
        x += '<a:prstDash val="{0}"/>'.format(dash)
    x += '</a:ln></c:spPr>'
    x += '<c:marker><c:symbol val="none"/></c:marker><c:smooth val="0"/>'
    x += '<c:xVal><c:numRef><c:f>{0}</c:f></c:numRef></c:xVal>'.format(xf)
    x += '<c:yVal><c:numRef><c:f>{0}</c:f></c:numRef></c:yVal>'.format(yf)

def _camp_build_chart_xml(title, sheet_name, meta, mode_entries, eos, rpm, margin, crossings):
    sn = "'{0}'".format(sheet_name.replace("'", "''"))
    def ref(col, r1, r2):
        return "{0}!${1}${2}:${1}${3}".format(sn, _mac_col_letter(col), r1, r2)

    rpm_lo = rpm * (1.0 - margin)
    rpm_hi = rpm * (1.0 + margin)

    si = 0
    series_xml = ""
    for i, (label, f) in enumerate(mode_entries):
        col = meta["C_M1"] + i
        clr = _CAMP_MODE_COLORS[i % len(_CAMP_MODE_COLORS)]
        series_xml += _camp_ser_line(si, label,
            ref(meta["C_X"], meta["R_DATA1"], meta["R_DATA2"]),
            ref(col,         meta["R_DATA1"], meta["R_DATA2"]),
            clr, width=28000)
        si += 1

    for i, eo in enumerate(eos):
        col = meta["C_EO1"] + i
        series_xml += _camp_ser_line(si, "EO{0}".format(eo),
            ref(meta["C_EOX"], meta["R_DATA1"], meta["R_DATA2"]),
            ref(col,           meta["R_DATA1"], meta["R_DATA2"]),
            _CAMP_EO_COLOR, width=10000, label_text="EO{0}".format(eo))
        si += 1

    C_VL = meta["C_VL"]
    for lbl, col_x, col_y, r1_, r2_, clr, dash in [
        ("RPM lo {0:.0f}".format(rpm_lo),  C_VL,   C_VL+1, meta["R_VL_LO1"],  meta["R_VL_LO2"],  "808080", "dash"),
        ("RPM nom {0:.0f}".format(rpm),    C_VL+2, C_VL+3, meta["R_VL_NOM1"], meta["R_VL_NOM2"], "000000", None),
        ("RPM hi {0:.0f}".format(rpm_hi),  C_VL+4, C_VL+5, meta["R_VL_HI1"],  meta["R_VL_HI2"],  "808080", "dash"),
    ]:
        series_xml += _camp_ser_line(si, lbl, ref(col_x, r1_, r2_), ref(col_y, r1_, r2_),
                                     clr, width=32000, dash=dash)
        si += 1

    if crossings:
        series_xml += _camp_ser_scatter(si, "CROSSING",
            ref(meta["C_CROSS"],   meta["R_CROSS_START"], meta["R_CROSS_END"]),
            ref(meta["C_CROSS"]+1, meta["R_CROSS_START"], meta["R_CROSS_END"]),
            "FF0000", sz=12)
        si += 1

    chart_xml = (
        '<?xml version="1.0" encoding="UTF-8" standalone="yes"?>' +
        '<c:chartSpace xmlns:c="http://schemas.openxmlformats.org/drawingml/2006/chart" ' +
        'xmlns:a="http://schemas.openxmlformats.org/drawingml/2006/main" ' +
        'xmlns:r="http://schemas.openxmlformats.org/officeDocument/2006/relationships">' +
        '<c:chart>' +
        '<c:title><c:tx><c:rich><a:bodyPr/><a:lstStyle/><a:p><a:pPr><a:defRPr sz="1200" b="1"/></a:pPr>' +
        '<a:r><a:rPr lang="en-US" b="1"/><a:t>{0}</a:t></a:r></a:p></c:rich></c:tx><c:overlay val="0"/></c:title>' +
        '<c:autoTitleDeleted val="0"/>' +
        '<c:plotArea><c:layout/><c:scatterChart><c:scatterStyle val="lineMarker"/><c:varyColors val="0"/>' +
        series_xml +
        '<c:axId val="10"/><c:axId val="20"/></c:scatterChart>' +
        _camp_val_ax(10, 20, "b", "Speed (RPM)", mn=0, mx=meta["rpm_max"]) +
        _camp_val_ax(20, 10, "l", "Frequency (Hz)", mn=0, mx=round(meta["max_freq"], 0)) +
        '</c:plotArea>' +
        '<c:legend><c:legendPos val="b"/><c:overlay val="0"/></c:legend>' +
        '<c:plotVisOnly val="1"/>' +
        '</c:chart></c:chartSpace>'
    ).format(_mac_xml_escape(title))
    return chart_xml

def _camp_raw_sheet_xml(rows, max_row):
    """SADECE SAYISAL hucreler icin (sharedStrings YOK) basit bir
    worksheet XML'i uretir -- Campbell'in hidden veri sayfalari icin."""
    rows_xml = []
    for r in sorted(rows.keys()):
        cells = rows[r]
        cells_xml = "".join(
            '<c r="{0}"><v>{1}</v></c>'.format(_mac_col_letter(c) + str(r), float(v))
            for c, v in sorted(cells.items()))
        rows_xml.append('<row r="{0}">{1}</row>'.format(r, cells_xml))
    return (
        '<?xml version="1.0" encoding="UTF-8" standalone="yes"?>' +
        '<worksheet xmlns="http://schemas.openxmlformats.org/spreadsheetml/2006/main">' +
        '<sheetData>' + "".join(rows_xml) + '</sheetData>' +
        '</worksheet>'
    )

def run_campbell_module(parsed_clean, energy, parts, se_thr, rpm, margin_pct, eos, out_dir, prog=None):
    """Ana akistan cagrilir.
    parsed_clean : [{"hi": n, "modes": {mode: {"freq": f, ...}}}, ...]  --
                    twin-elenmis .OUT verisi (ana koddan gelir).
    energy       : energy[hi][mode] = {"all","bodies","ratio"} veya None
                    (strain energy hesaplanmadiysa body sayfalari
                    ATLANIR, sadece genel diyagram uretilir).
    parts        : body (Named Selection) isimleri listesi.
    se_thr       : Strain Energy Threshold (%).
    rpm, margin_pct, eos : Campbell popup'undan gelen degerler.

    Uretilen sayfalar:
      1) 'General'          -- TUM HI'lerden gelen, secili EO'larla
                                KESISEN modlari TEK grafikte birlestirir.
      2) Her body icin BIR sayfa, ikisi de AYNI sayfada IKI grafik:
                                'Dominant + SE Threshold' ve 'All Dominant
                                (no threshold)'.

    Donus: (out_path, n_general_crossings, body_summary) --
    body_summary = {body_name: (n_filtered_modes, n_unfiltered_modes)}."""
    if prog: prog.update("Campbell: preparing data...")
    margin = margin_pct / 100.0

    # ---- 1) GENEL diyagram icin: TUM HI/mod/freq'leri topla ----
    all_modes = []   # [(label, hi, mode, freq), ...]
    for block in parsed_clean:
        hi = block["hi"]
        for mode, rec in block["modes"].items():
            all_modes.append(("HI{0} Mod{1}".format(hi, mode), hi, mode, rec["freq"]))

    rpm_lo = rpm * (1.0 - margin)
    rpm_hi = rpm * (1.0 + margin)
    general_entries = []   # [(label, freq), ...] -- SADECE kesisenler
    for (label, hi, mode, freq) in all_modes:
        eo, rpm_cross = _camp_find_crossing(freq, eos, rpm, rpm_lo, rpm_hi)
        if eo is not None:
            general_entries.append((label, freq))
    print("Campbell: {0} mode(s) found across all HI, {1} crossing(s) in general diagram.".format(
        len(all_modes), len(general_entries)))

    # ---- 2) Body-bazli veriler (SADECE strain energy hesaplandiysa) ----
    body_data = {}   # {body_name: (filtered_entries, unfiltered_entries)}
    if energy is not None and parts:
        # (hi,mode) -> freq lookup (parsed_clean'den, energy dict frekans
        # tutmuyor)
        freq_lookup = {}
        for block in parsed_clean:
            hi = block["hi"]
            for mode, rec in block["modes"].items():
                freq_lookup[(hi, mode)] = rec["freq"]

        for p in parts:
            filtered = []
            unfiltered = []
            for hi in sorted(energy.keys()):
                for mode in sorted(energy[hi].keys()):
                    ratios = energy[hi][mode].get("ratio", {})
                    if not ratios:
                        continue
                    dominant = max(parts, key=lambda pp: ratios.get(pp, 0.0))
                    if dominant != p:
                        continue
                    freq = freq_lookup.get((hi, mode))
                    if freq is None:
                        continue
                    label = "HI{0} Mod{1}".format(hi, mode)
                    unfiltered.append((label, freq))
                    if ratios.get(p, 0.0) >= se_thr:
                        filtered.append((label, freq))
            body_data[p] = (filtered, unfiltered)
            print("Campbell: '{0}' -> {1} dominant+threshold mode(s), {2} all-dominant mode(s).".format(
                p, len(filtered), len(unfiltered)))

    # ---- OOXML iskeleti ----
    if prog: prog.update("Campbell: building Excel...")
    sheets = []          # [(name, xml, hidden_bool), ...]
    drawings = {}         # {drawing_idx: xml}
    drawing_rels = {}     # {drawing_idx: xml}
    charts = {}           # {chart_idx: xml}
    sheet_rels = {}       # {sheet_idx(1-based): xml}
    used_names = set()

    SUMMARY_COL = 22   # grafik (col 0-20) sonrasi bir bosluk (col 21) birakip col 22'den (V) baslar

    def _camp_txt(col, row, text):
        return '<c r="{0}{1}" t="inlineStr"><is><t xml:space="preserve">{2}</t></is></c>'.format(
            _mac_col_letter(col), row, _mac_xml_escape(text))

    def _add_chart_sheet(view_name, raw_name, rows, max_row, chart_specs):
        """chart_specs: [(title, meta, mode_entries, crossings, anchor_row0, anchor_row1), ...]
        Bir gizli veri sayfasi + bir gorunur (1 veya 2 grafikli) sayfa ekler.
        [YENI] Her grafigin YANINA (SAFE modulundeki gibi), o araliktaki
        gercek kesisen (crossing) modlarin bir OZETI yazilir -- kesisen
        mod yoksa 'No crossing modes found' notu."""
        raw_idx = len(sheets) + 1
        sheets.append((raw_name, _camp_raw_sheet_xml(rows, max_row), True))

        view_idx = len(sheets) + 1
        summary_rows = {}   # {excel_row: {col: cell_xml}} -- grafik basina ozet metni
        drawing_anchors = ""
        rels_chart_parts = ""
        for k, (title, meta, mode_entries, crossings, r0, r1) in enumerate(chart_specs):
            drawing_idx = len(charts) + 1
            chart_xml = _camp_build_chart_xml(title, raw_name, meta, mode_entries, eos, rpm, margin, crossings)
            charts[drawing_idx] = chart_xml
            rels_chart_parts += ('<Relationship Id="rId{0}" '
                                'Type="http://schemas.openxmlformats.org/officeDocument/2006/relationships/chart" '
                                'Target="../charts/chart{0}.xml"/>').format(drawing_idx)
            drawing_anchors += (
                '<xdr:twoCellAnchor moveWithCells="0" sizeWithCells="0">' +
                '<xdr:from><xdr:col>0</xdr:col><xdr:colOff>0</xdr:colOff>' +
                '<xdr:row>{0}</xdr:row><xdr:rowOff>0</xdr:rowOff></xdr:from>'.format(r0) +
                '<xdr:to><xdr:col>20</xdr:col><xdr:colOff>0</xdr:colOff>' +
                '<xdr:row>{0}</xdr:row><xdr:rowOff>0</xdr:rowOff></xdr:to>'.format(r1) +
                '<xdr:graphicFrame macro="">' +
                '<xdr:nvGraphicFramePr><xdr:cNvPr id="{0}" name="Chart {0}"/>'.format(drawing_idx) +
                '<xdr:cNvGraphicFramePr/></xdr:nvGraphicFramePr>' +
                '<xdr:xfrm><a:off x="0" y="0"/><a:ext cx="0" cy="0"/></xdr:xfrm>' +
                '<a:graphic><a:graphicData uri="http://schemas.openxmlformats.org/drawingml/2006/chart">' +
                '<c:chart xmlns:c="http://schemas.openxmlformats.org/drawingml/2006/chart" r:id="rId{0}"/>'.format(drawing_idx) +
                '</a:graphicData></a:graphic></xdr:graphicFrame><xdr:clientData/></xdr:twoCellAnchor>'
            )

            # ---- Kesisen mod ozeti (SAFE modulundeki gibi, grafigin yanina) ----
            sr = r0 + 1   # 0-based anchor row -> 1-based Excel satiri (grafigin ust kenariyla hizali)
            summary_rows.setdefault(sr, {})[SUMMARY_COL] = _camp_txt(
                SUMMARY_COL, sr, "{0} -- Crossing Summary".format(title))
            sr += 1
            if crossings:
                for (label, freq, eo, rpm_cross) in crossings:
                    summary_rows.setdefault(sr, {})[SUMMARY_COL] = _camp_txt(
                        SUMMARY_COL, sr, "{0} ({1:.0f} Hz) -> CROSSES EO{2} at {3:.0f} RPM".format(
                            label, freq, eo, rpm_cross))
                    sr += 1
            else:
                summary_rows.setdefault(sr, {})[SUMMARY_COL] = _camp_txt(
                    SUMMARY_COL, sr, "No crossing modes found in this range.")

        drawing_idx_for_sheet = len(drawings) + 1
        drawing_xml = (
            '<?xml version="1.0" encoding="UTF-8" standalone="yes"?>' +
            '<xdr:wsDr xmlns:xdr="http://schemas.openxmlformats.org/drawingml/2006/spreadsheetDrawing" ' +
            'xmlns:a="http://schemas.openxmlformats.org/drawingml/2006/main" ' +
            'xmlns:r="http://schemas.openxmlformats.org/officeDocument/2006/relationships">' +
            drawing_anchors + '</xdr:wsDr>'
        )
        drawings[drawing_idx_for_sheet] = drawing_xml
        drawing_rels[drawing_idx_for_sheet] = (
            '<?xml version="1.0" encoding="UTF-8" standalone="yes"?>' +
            '<Relationships xmlns="http://schemas.openxmlformats.org/package/2006/relationships">' +
            rels_chart_parts + '</Relationships>'
        )

        summary_rows_xml = "".join(
            '<row r="{0}">{1}</row>'.format(r, "".join(cells[c] for c in sorted(cells.keys())))
            for r, cells in sorted(summary_rows.items())
        )
        ws_view = (
            '<?xml version="1.0" encoding="UTF-8" standalone="yes"?>' +
            '<worksheet xmlns="http://schemas.openxmlformats.org/spreadsheetml/2006/main"' +
            ' xmlns:r="http://schemas.openxmlformats.org/officeDocument/2006/relationships">' +
            '<cols><col min="{0}" max="{0}" width="46" customWidth="1"/></cols>'.format(SUMMARY_COL) +
            '<sheetData>' + summary_rows_xml + '</sheetData>' +
            '<drawing r:id="rId1"/></worksheet>'
        )
        sheets.append((view_name, ws_view, False))
        sheet_rels[view_idx] = (
            '<?xml version="1.0" encoding="UTF-8" standalone="yes"?>' +
            '<Relationships xmlns="http://schemas.openxmlformats.org/package/2006/relationships">' +
            '<Relationship Id="rId1" Type="http://schemas.openxmlformats.org/officeDocument/2006/relationships/drawing" ' +
            'Target="../drawings/drawing{0}.xml"/></Relationships>'.format(drawing_idx_for_sheet)
        )

    # ---- Sayfa: General ----
    gen_rows, gen_meta, gen_crossings = _camp_build_chart_data(general_entries, eos, rpm, margin)
    gen_title = "Campbell Diagram - General (all HI, crossing modes)"
    if not general_entries:
        gen_title = "Campbell Diagram - General (no crossing modes found)"
    _add_chart_sheet("General", "General_Raw", gen_rows, gen_meta["max_row"],
                     [(gen_title, gen_meta, general_entries, gen_crossings, 0, 40)])

    # ---- Sayfalar: her body icin (SADECE strain energy varsa) ----
    body_summary = {}
    for p in parts if (energy is not None and parts) else []:
        filtered, unfiltered = body_data.get(p, ([], []))
        body_summary[p] = (len(filtered), len(unfiltered))

        sname = _mac_sheet_name(p, used_names)
        raw_name = _mac_sheet_name(p + "_Raw", used_names)

        f_rows, f_meta, f_cross = _camp_build_chart_data(filtered, eos, rpm, margin)
        u_rows, u_meta, u_cross = _camp_build_chart_data(unfiltered, eos, rpm, margin)

        # Iki veri blogunu AYNI gizli sayfada, farkli sutun araliklarinda
        # tutmak icin ikinci blogu saga kaydiriyoruz.
        u_shift = f_meta["C_CROSS"] + 2 + 4
        u_rows_shifted = {}
        for r, cols in u_rows.items():
            u_rows_shifted[r] = {}
            for c, v in cols.items():
                u_rows_shifted[r][c + u_shift] = v
        combined_rows = {}
        for r, cols in f_rows.items():
            combined_rows.setdefault(r, {}).update(cols)
        for r, cols in u_rows_shifted.items():
            combined_rows.setdefault(r, {}).update(cols)
        combined_max_row = max(f_meta["max_row"], u_meta["max_row"])

        # u_meta'nin sutun indekslerini de kaydir (chart referanslari icin)
        u_meta_shifted = dict(u_meta)
        for key in ["C_X", "C_M1", "C_EOX", "C_EO1", "C_VL", "C_CROSS"]:
            u_meta_shifted[key] = u_meta[key] + u_shift

        title_f = "{0} - Dominant & >= SE Threshold ({1:.1f}%)".format(p, se_thr)
        title_u = "{0} - All Dominant Modes (no threshold)".format(p)

        _add_chart_sheet(sname, raw_name, combined_rows, combined_max_row, [
            (title_f, f_meta,         filtered,   f_cross, 0,  40),
            (title_u, u_meta_shifted, unfiltered, u_cross, 42, 82),
        ])

    # ---- Workbook iskeleti ----
    active_tab = 0
    for i, (_n, _x, hidden) in enumerate(sheets):
        if not hidden:
            active_tab = i
            break

    ct = ('<?xml version="1.0" encoding="UTF-8" standalone="yes"?>' +
         '<Types xmlns="http://schemas.openxmlformats.org/package/2006/content-types">' +
         '<Default Extension="rels" ContentType="application/vnd.openxmlformats-package.relationships+xml"/>' +
         '<Default Extension="xml" ContentType="application/xml"/>' +
         '<Override PartName="/xl/workbook.xml" ContentType="application/vnd.openxmlformats-officedocument.spreadsheetml.sheet.main+xml"/>')
    for i in range(1, len(sheets) + 1):
        ct += ('<Override PartName="/xl/worksheets/sheet{0}.xml" '
              'ContentType="application/vnd.openxmlformats-officedocument.spreadsheetml.worksheet+xml"/>').format(i)
    for di in charts:
        ct += ('<Override PartName="/xl/charts/chart{0}.xml" '
              'ContentType="application/vnd.openxmlformats-officedocument.drawingml.chart+xml"/>').format(di)
    for di in drawings:
        ct += ('<Override PartName="/xl/drawings/drawing{0}.xml" '
              'ContentType="application/vnd.openxmlformats-officedocument.drawing+xml"/>').format(di)
    ct += ('<Override PartName="/xl/styles.xml" '
          'ContentType="application/vnd.openxmlformats-officedocument.spreadsheetml.styles+xml"/></Types>')

    wb = ('<?xml version="1.0" encoding="UTF-8" standalone="yes"?>' +
         '<workbook xmlns="http://schemas.openxmlformats.org/spreadsheetml/2006/main" ' +
         'xmlns:r="http://schemas.openxmlformats.org/officeDocument/2006/relationships">' +
         '<bookViews><workbookView activeTab="{0}"/></bookViews><sheets>'.format(active_tab))
    for i, (name, _xml, hidden) in enumerate(sheets, start=1):
        state = ' state="hidden"' if hidden else ''
        wb += '<sheet name="{0}" sheetId="{1}" r:id="rId{1}"{2}/>'.format(_mac_xml_escape(name), i, state)
    wb += '</sheets></workbook>'

    n_sheets = len(sheets)
    wb_rels = ('<?xml version="1.0" encoding="UTF-8" standalone="yes"?>' +
              '<Relationships xmlns="http://schemas.openxmlformats.org/package/2006/relationships">')
    for i in range(1, n_sheets + 1):
        wb_rels += ('<Relationship Id="rId{0}" '
                   'Type="http://schemas.openxmlformats.org/officeDocument/2006/relationships/worksheet" '
                   'Target="worksheets/sheet{0}.xml"/>').format(i)
    wb_rels += ('<Relationship Id="rId{0}" '
               'Type="http://schemas.openxmlformats.org/officeDocument/2006/relationships/styles" '
               'Target="styles.xml"/></Relationships>').format(n_sheets + 1)

    styles = (
        '<?xml version="1.0" encoding="UTF-8" standalone="yes"?>' +
        '<styleSheet xmlns="http://schemas.openxmlformats.org/spreadsheetml/2006/main">' +
        '<fonts count="1"><font><sz val="10"/><name val="Calibri"/></font></fonts>' +
        '<fills count="2"><fill><patternFill patternType="none"/></fill>' +
        '<fill><patternFill patternType="gray125"/></fill></fills>' +
        '<borders count="1"><border><left/><right/><top/><bottom/><diagonal/></border></borders>' +
        '<cellStyleXfs count="1"><xf numFmtId="0" fontId="0" fillId="0" borderId="0"/></cellStyleXfs>' +
        '<cellXfs count="1"><xf numFmtId="0" fontId="0" fillId="0" borderId="0" xfId="0"/></cellXfs>' +
        '<cellStyles count="1"><cellStyle name="Normal" xfId="0" builtinId="0"/></cellStyles>' +
        '</styleSheet>'
    )

    root_rels = ('<?xml version="1.0" encoding="UTF-8" standalone="yes"?>' +
                '<Relationships xmlns="http://schemas.openxmlformats.org/package/2006/relationships">' +
                '<Relationship Id="rId1" Type="http://schemas.openxmlformats.org/officeDocument/2006/relationships/officeDocument" ' +
                'Target="xl/workbook.xml"/></Relationships>')

    out_path = os.path.join(out_dir, "Campbell_Diagram.xlsx")
    _out_dir = os.path.dirname(out_path)
    if _out_dir and not os.path.exists(_out_dir):
        os.makedirs(_out_dir)

    if prog: prog.update("Campbell: writing Excel...", os.path.basename(out_path))
    with _zipfile.ZipFile(out_path, "w", _zipfile.ZIP_DEFLATED) as _zf:
        _zf.writestr("[Content_Types].xml", ct.encode("utf-8"))
        _zf.writestr("_rels/.rels", root_rels.encode("utf-8"))
        _zf.writestr("xl/workbook.xml", wb.encode("utf-8"))
        _zf.writestr("xl/_rels/workbook.xml.rels", wb_rels.encode("utf-8"))
        _zf.writestr("xl/styles.xml", styles.encode("utf-8"))
        for i, (name, xml, hidden) in enumerate(sheets, start=1):
            _zf.writestr("xl/worksheets/sheet{0}.xml".format(i), xml.encode("utf-8"))
            if i in sheet_rels:
                _zf.writestr("xl/worksheets/_rels/sheet{0}.xml.rels".format(i), sheet_rels[i].encode("utf-8"))
        for di, xml in drawings.items():
            _zf.writestr("xl/drawings/drawing{0}.xml".format(di), xml.encode("utf-8"))
            _zf.writestr("xl/drawings/_rels/drawing{0}.xml.rels".format(di), drawing_rels[di].encode("utf-8"))
        for ci, xml in charts.items():
            _zf.writestr("xl/charts/chart{0}.xml".format(ci), xml.encode("utf-8"))

    print("  Campbell Excel written: {0}".format(out_path))
    return out_path, len(general_entries), body_summary

# MAIN
# =============================================================================
def main():
    global FONT_TITLE
    _CANCEL[0] = False   # her calistirmada sifirla

    model = ExtAPI.DataModel.Project.Model
    base  = Form().Font
    FONT_TITLE = Font(base, FontStyle.Bold)

    analyses = list(model.Analyses)
    modal    = [a for a in analyses if "modal" in str(a.AnalysisType).lower()]
    cands    = modal if modal else analyses
    if not cands:
        MessageBox.Show("No analysis found.", "Error"); return

    if len(cands) == 1:
        analysis = cands[0]
    else:
        d = AnalysisSelectDialog(cands)
        if d.ShowDialog() != DialogResult.OK: return
        analysis = d.selected_analysis

    # NS isimleri ve gruplar
    ns_groups = get_ns_groups(model)
    ns_names  = get_ns_names(model)
    if not ns_names:
        MessageBox.Show("At least 1 Named Selection is required.", "Error"); return

    dlg = ModalIdDialog(ns_names, ns_groups)
    if dlg.ShowDialog() != DialogResult.OK: return

    # MAC/SAFE checkbox'lari isaretlendigi anda kendi popup'larini acip
    # OK'lanan ayarlari ModalIdDialog._mac_dlg / _safe_dlg icinde sakladi
    # (bkz. _on_mac_toggle / _on_safe_toggle). Burada tekrar acmaya gerek yok.
    mac_dlg  = dlg._mac_dlg  if dlg.want_mac  else None
    safe_dlg = dlg._safe_dlg if dlg.want_safe else None
    campbell_dlg = dlg._campbell_dlg if dlg.want_campbell else None

    # solve.out oku (sadece gerekliyse: body secili, Full Sweep veya SAFE
    # istenmisse. Sadece MAC calistiriliyorsa solve.out sart degildir.)
    parsed = []
    parsed_clean = []
    twin_log = {}
    have_out_file = bool(dlg.out_file) and os.path.exists(dlg.out_file)
    if have_out_file:
        try:
            parsed = parse_out(dlg.out_file)
        except Exception as ex:
            MessageBox.Show("Failed to read solve.out:\n" + str(ex), "Error"); return
        if not parsed:
            MessageBox.Show("No participation factor table found in solve.out.", "Error"); return

        print("HI blocks found:", [b["hi"] for b in parsed])

        # Twin mod eliminasyonu (her iki akis icin de ortak)
        print("Twin elimination (tolerance: {0} Hz)...".format(dlg.twin_hz))
        parsed_clean, twin_log = eliminate_twins(parsed, dlg.twin_hz)
        total_twins = sum(len(twin_log[hi]) for hi in twin_log)
        if total_twins > 0:
            print("Total twins eliminated: {0}".format(total_twins))
        else:
            print("No twin modes found.")
    else:
        print("No .OUT file selected -- skipping .OUT-based flow (MAC-only run).")

    # ---- Section 1-4 akisi: effective mass filtreli (VEYA sadece bilgi) ----
    run_main_flow = len(dlg.parts) > 0
    selected = {}
    nsel = 0
    if run_main_flow:
        if dlg.em_mode == "info":
            # BILGI MODU: hicbir mod ELENMEZ -- twin-elenmis TUM modlar
            # icin EM orani hesaplanir (select_modes negatif esikle
            # cagrilarak TUM modlar "gecmis" sayilir). EM esigi yine de
            # raporda (*) isaretlemek icin kullanilir (write_txt zaten
            # deger>=esik kontrolu yapiyor, elemeden bagimsiz).
            selected = select_modes(parsed_clean, -1.0)
            nsel = sum(len(selected[hi]) for hi in selected)
            print("Effective mass: INFO MODE (no elimination) -- {0} modes total "
                 "(twin-filtered), EM threshold {1}% used only for report flags.".format(
                     nsel, dlg.em_thr))
        else:
            selected = select_modes(parsed_clean, dlg.em_thr)
            nsel = sum(len(selected[hi]) for hi in selected)
            if nsel == 0:
                MessageBox.Show(
                    "No modes exceed the effective mass threshold ({0:.1f}%).".format(dlg.em_thr),
                    "Info")
                run_main_flow = False
            else:
                print("Selected modes (EM {0}%):".format(dlg.em_thr))
                for hi in sorted(selected.keys()):
                    print("  HI {0}: modes {1}".format(hi, sorted(selected[hi].keys())))

    if not run_main_flow and not dlg.want_mac and not dlg.want_safe and not dlg.want_campbell:
        return

    # progress popup
    import System
    _t_start = System.DateTime.Now
    prog = ProgressDialog()
    prog.Show()
    prog.update("Starting...", "{0} modes".format(nsel))

    energy = None
    if run_main_flow:
        if dlg.want_se:
            try:
                energy = compute_se_from_rst(analysis, dlg.parts, selected,
                                             dlg.se_rst_path, prog)
            except Exception as ex:
                prog.close_ok()
                MessageBox.Show("Computation error:\n" + str(ex), "Error"); return

            if _CANCEL[0]:
                prog.close_ok()
                MessageBox.Show("Cancelled by user.", "Cancelled"); return

            if energy is None:
                prog.close_ok(); return
        else:
            print("Strain energy computation SKIPPED (disabled by user) -- "
                  "continuing with effective mass selection only.")

        # Mode shapes - sadece main flow'daki secili modlar icin (SE
        # acik/kapali olmasindan bagimsiz, sadece EM secimine baglidir)
        prog.update("Creating mode shapes (main selection)...")
        add_mode_shapes(analysis.Solution, selected, prog, name_prefix="_MS")
        if not _CANCEL[0]:
            group_results(analysis.Solution, ExtAPI.DataModel.Project.Model,
                          ms_prefix="_MS_", ms_group_name="Mode Shape Results")

    if _CANCEL[0]:
        prog.close_ok()
        MessageBox.Show("Cancelled by user.", "Cancelled"); return

    txt_path = None
    if run_main_flow:
        prog.update("Writing report...")
        if not os.path.exists(dlg.out_dir): os.makedirs(dlg.out_dir)
        txt_path = os.path.join(dlg.out_dir,
            "ModalPartID_{0}.txt".format(sanitize(analysis.Name)))
        write_txt(txt_path, dlg.parts, dlg.em_thr, dlg.se_thr, dlg.twin_hz,
                  parsed_clean, twin_log, selected, energy,
                  em_enabled=dlg.want_em, se_enabled=dlg.want_se, em_mode=dlg.em_mode)

    # ---- MAC (secildiyse) ----
    mac_path = None; mac_n = 0; mac_warn = 0; mac_error = None
    if dlg.want_mac and mac_dlg is not None:
        try:
            prog.update("MAC: starting...", "")
            # HI listesi secilirken RST zaten yuklenmis olabilir
            # (mac_dlg._dpf/_ds/_model); varsa tekrar RST acmamak icin
            # bunlari run_mac_module'e aktar.
            mac_path, mac_n, mac_warn = run_mac_module(
                mac_dlg.rst_path, mac_dlg.threshold, dlg.twin_hz, dlg.out_dir,
                hi_filter=mac_dlg.hi_list, prog=prog,
                dpf=mac_dlg._dpf, ds=mac_dlg._ds, model=mac_dlg._model,
                energy=energy, se_thr=dlg.se_thr, parts=dlg.parts)
        except Exception as ex:
            mac_error = str(ex)
            # Bazen .NET/COM tarafindan firlatilan hatalar Python'da
            # anlamsiz bir tuple metni olarak gorunur (ornegin
            # "('unknown', '\x00', 0, 1, '')") -- bu durumda kullaniciya
            # HANGI ASAMADA oldugunu bilerek daha isabetli bir ipucu
            # ekleriz (RST okuma ile Excel yazma FARKLI sebeplerdendir).
            if len(mac_error) < 60 or "\\x00" in repr(mac_error):
                if "writing MAC Excel" in mac_error or "writing zip part" in mac_error:
                    mac_error += ("\n(This looks like a low-level file-write error while "
                                 "creating MAC_results.xlsx. Please verify the output folder "
                                 "in Section 1 is writable, the file isn't already open "
                                 "elsewhere, and there is enough disk space.)")
                elif "RST" in mac_error or "loading RST" in mac_error or "mode shapes" in mac_error:
                    mac_error += ("\n(This looks like a low-level RST read error. Please verify "
                                 "the RST file chosen in Section 1 is complete, not locked/open "
                                 "elsewhere, and matches the analysis that was actually solved.)")
                else:
                    mac_error += ("\n(This is a low-level .NET/DPF error without a clear message. "
                                 "Please check the console output above for more context.)")
            print("MAC ERROR:", mac_error)

    # ---- SAFE (secildiyse) ----
    safe_path = None; safe_cross = 0; safe_error = None
    if dlg.want_safe and safe_dlg is not None:
        try:
            prog.update("SAFE: writing diagram...", "")
            safe_out_path = os.path.join(dlg.out_dir, "SAFE_Diagram.xlsx")
            safe_path, safe_cross = run_safe_module(
                parsed_clean, safe_dlg.blade_count, safe_dlg.sector, safe_dlg.max_hi,
                safe_dlg.idle_rpm, safe_dlg.rl_rpm, safe_dlg.margin_pct,
                safe_dlg.critical_eos, safe_out_path)
        except Exception as ex:
            safe_error = str(ex)
            print("SAFE ERROR:", safe_error)

    # ---- Campbell Diagram (secildiyse) ----
    campbell_path = None; campbell_n_general = 0; campbell_body_summary = {}
    campbell_error = None
    if dlg.want_campbell and campbell_dlg is not None:
        try:
            prog.update("Campbell: starting...", "")
            campbell_path, campbell_n_general, campbell_body_summary = run_campbell_module(
                parsed_clean, energy, dlg.parts, dlg.se_thr,
                campbell_dlg.rpm, campbell_dlg.margin, campbell_dlg.eos,
                dlg.out_dir, prog=prog)
        except Exception as ex:
            campbell_error = str(ex)
            print("CAMPBELL ERROR:", campbell_error)

    prog.done(); prog.close_ok()

    # Sure hesapla
    _t_end     = System.DateTime.Now
    _elapsed   = int((_t_end - _t_start).TotalSeconds)
    if _elapsed < 60:
        _dur_str = "{0} sec".format(_elapsed)
    else:
        _dur_str = "{0} min {1} sec".format(_elapsed // 60, _elapsed % 60)

    # ozet popup
    lines = []
    if run_main_flow and energy is not None:
        lines.append("Done.  {0} modes processed (main selection, strain energy "
                     "read directly from RST).\n".format(nsel))
        for hi in sorted(energy.keys()):
            lines.append("HI {0}:".format(hi))
            for mode in sorted(energy[hi].keys()):
                ratios = energy[hi][mode]["ratio"]
                best   = max(dlg.parts, key=lambda p: ratios[p]) if dlg.parts else "-"
                flag   = "**" if any(ratios[p] >= dlg.se_thr for p in dlg.parts) else "  "
                lines.append("{0}  Mode {1}  f={2:.0f}Hz -> {3} ({4:.1f}%)".format(
                    flag, mode, selected[hi][mode]["freq"], best, ratios[best]))
        if len(lines) > 45: lines = lines[:45] + ["... (see report)"]
    elif run_main_flow and energy is None:
        lines.append("Done.  {0} modes processed (effective mass only -- "
                     "strain energy was disabled).\n".format(nsel))

    if dlg.want_mac:
        lines.append("")
        if mac_error:
            lines.append("MAC: FAILED -- " + mac_error)
        else:
            hi_str = ", ".join(str(h) for h in mac_dlg.hi_list) if mac_dlg is not None else "?"
            lines.append("MAC: HI [{0}] -- {1} modes, {2} warnings >= threshold.".format(
                hi_str, mac_n, mac_warn))

    if dlg.want_safe:
        lines.append("")
        if safe_error:
            lines.append("SAFE: FAILED -- " + safe_error)
        else:
            lines.append("SAFE: {0} modes crossing a Critical Engine Order.".format(safe_cross))

    if dlg.want_campbell:
        lines.append("")
        if campbell_error:
            lines.append("Campbell: FAILED -- " + campbell_error)
        else:
            lines.append("Campbell: {0} crossing mode(s) in the general diagram.".format(campbell_n_general))
            for p in dlg.parts:
                if p in campbell_body_summary:
                    n_f, n_u = campbell_body_summary[p]
                    lines.append("  {0}: {1} dominant+threshold, {2} all-dominant.".format(p, n_f, n_u))

    msg  = "\n".join(lines) if lines else "Done."
    if txt_path:
        msg += "\n\nReport: " + txt_path
    if mac_path:
        msg += "\nMAC: " + mac_path
    if safe_path:
        msg += "\nSAFE: " + safe_path
    if campbell_path:
        msg += "\nCampbell: " + campbell_path
    msg += "\nDuration: " + _dur_str
    MessageBox.Show(msg, "Modal Part Identifier - Summary")
    return analysis, dlg, parsed_clean, twin_log
