REG_KEY = r"Software\RSTSETest"

def _reg_get(name):
    try:
        key = Win32.Registry.CurrentUser.OpenSubKey(REG_KEY)
        if key is None: return ""
        val = key.GetValue(name); key.Close()
        return str(val) if val else ""
    except: return ""

def _reg_set(name, val):
    try:
        key = Win32.Registry.CurrentUser.CreateSubKey(REG_KEY)
        key.SetValue(name, val); key.Close()
    except: pass

def _diag(msg, detail=""):
    _log("TEST: " + msg + ((" -- " + detail) if detail else ""))

def _qty_to_float(q):
    for attr in ("Value", "Magnitude"):
        try: return float(getattr(q, attr))
        except: pass
    try: return float(str(q).split()[0].replace(",", "."))
    except: return 0.0

def find_ns(model, ns_name):
    try:
        for item in model.NamedSelections.Children:
            try:
                sub = list(item.Children)
                if len(sub) > 0:
                    for ns in sub:
                        try:
                            if ns.Name == ns_name: return ns
                        except: pass
                else:
                    if item.Name == ns_name: return item
            except:
                if item.Name == ns_name: return item
    except: pass
    return None

def get_ns_names(model):
    names = []
    try:
        for item in model.NamedSelections.Children:
            try:
                sub = list(item.Children)
                if len(sub) > 0:
                    for ns in sub:
                        try: names.append(ns.Name)
                        except: pass
                else:
                    names.append(item.Name)
            except:
                names.append(item.Name)
    except: pass
    return names

def _flatten_region_elements(raw):
    """GetElementIdsFromRegionIds(geo_ids) TEK bir flat element ID listesi
    DEGIL, her geometri ID'si icin ayri bir KeyValuePair<int, IList<int>>
    (region_id -> element_id_listesi) dondurur (test ciktisiyla teyit
    edildi). Bu fonksiyon hepsini TEK bir flat, tekrarsiz (set) element
    ID listesine acar.

    ONEMLI (hata duzeltmesi): 'raw' bazen None donebilir (ornegin gecersiz
    veya bos bir geo_id verildiginde) -- bu durumda 'for item in raw'
    dogrudan 'NoneType is not iterable' hatasi verir. Bu yuzden EN BASTA
    None kontrolu yapiyoruz."""
    elem_ids = set()
    if raw is None:
        _diag("UYARI: GetElementIdsFromRegionIds None dondurdu (element ID cozulemedi).")
        return elem_ids
    try:
        raw_list = list(raw)
    except Exception as ex:
        _diag("UYARI: donen deger iterable degil (list() basarisiz)", str(ex))
        return elem_ids

    for item in raw_list:
        # Beklenen: KeyValuePair -- .Key (geo_id) / .Value (element listesi)
        try:
            values = item.Value
            if values is not None:
                for v in values:
                    elem_ids.add(int(v))
            continue
        except AttributeError:
            pass
        # Alternatif: (geo_id, [elem_ids]) tuple/list seklinde gelirse
        try:
            values = item[1]
            if values is not None:
                for v in values:
                    elem_ids.add(int(v))
            continue
        except (TypeError, IndexError):
            pass
        # Son care: zaten duz bir int ise (baska bir Ansys versiyonunda
        # API flat liste donduruyor olabilir)
        try:
            elem_ids.add(int(item))
        except (TypeError, ValueError):
            pass
    return elem_ids

def step_A_resolve_elements(model, body_names):
    """A) Body -> element ID cozumu (Mechanical'in kendi mesh API'si ile,
    DPF'e hic girmeden)."""
    _step("ADIM A: Body -> element ID cozumu (GetElementIdsFromRegionIds)")

    mesh_data = ExtAPI.DataModel.MeshDataByName("Global")
    if mesh_data is None:
        _diag("BASARISIZ: MeshDataByName('Global') None dondurdu.")
        return None
    _diag("MeshDataByName('Global') BASARILI.")

    body_elem_ids = {}
    for name in body_names:
        if _CANCEL[0]: return body_elem_ids
        ns = find_ns(model, name)
        if ns is None:
            _diag("UYARI: Named Selection bulunamadi: '{0}'".format(name))
            continue
        try:
            geo_ids_raw = ns.Location.Ids
            geo_ids = list(geo_ids_raw) if geo_ids_raw is not None else []
        except Exception as ex:
            _diag("UYARI: '{0}' icin ns.Location.Ids okunamadi".format(name), str(ex))
            continue
        if not geo_ids:
            _diag("UYARI: '{0}' icin geometri ID listesi BOS.".format(name))
            body_elem_ids[name] = set()
            continue
        _diag("'{0}': geometri ID'leri".format(name), str(geo_ids))
        try:
            raw = mesh_data.GetElementIdsFromRegionIds(geo_ids)
        except Exception as ex:
            _diag("BASARISIZ: '{0}' icin GetElementIdsFromRegionIds hata verdi".format(name), str(ex))
            continue

        elem_ids = _flatten_region_elements(raw)
        body_elem_ids[name] = elem_ids
        if elem_ids:
            sample = list(elem_ids)[:5]
            _diag("BASARILI: '{0}' -> {1} geometri bolgesi -> TOPLAM {2} element ID  (ilk 5: {3})".format(
                name, len(geo_ids), len(elem_ids), sample))
        else:
            _diag("UYARI: '{0}' -> flatten sonrasi 0 element ID (BOS, bir sorun var).".format(name))

    return body_elem_ids

def _get_result_info(dpf_model):
    """result_info'yu birden fazla olasi isim/casing ile dener (bu ortam
    dusuk seviye .NET DPF API'si kullaniyor -- pydpf-core'daki
    'model.metadata.result_info' burada CALISMAYABILIR, PascalCase
    varyasyonlari denemek gerekiyor)."""
    attempts = [
        ("model.metadata.result_info", lambda: dpf_model.metadata.result_info),
        ("model.result_info",          lambda: dpf_model.result_info),
        ("model.ResultInfo",           lambda: dpf_model.ResultInfo),
        ("model.Metadata.ResultInfo",  lambda: dpf_model.Metadata.ResultInfo),
    ]
    for name, fn in attempts:
        try:
            ri = fn()
            if ri is None:
                _diag("'{0}' None dondurdu, sonraki deneniyor.".format(name))
                continue
            _diag("result_info BULUNDU ('{0}' ile).".format(name))
            return ri
        except Exception as ex:
            _diag("'{0}' basarisiz".format(name), str(ex))
    return None

def _result_info_names(dpf_model):
    """result_info'dan sonuc adlari listesini GUVENLI sekilde cikarir
    (available_results None donerse BOS liste ile devam eder, crash etmez)."""
    ri_cands = []
    ri = _get_result_info(dpf_model)
    if ri is None:
        _diag("UYARI: result_info HICBIR yontemle bulunamadi.")
        return ri_cands
    avail = None
    try:
        avail = ri.available_results
    except Exception as ex:
        _diag("UYARI: available_results okunamadi (kucuk harf)", str(ex))
    if avail is None:
        try:
            avail = ri.AvailableResults
        except Exception as ex:
            _diag("UYARI: AvailableResults okunamadi (buyuk harf)", str(ex))
    if avail is None:
        _diag("UYARI: available_results/AvailableResults None -- sonuc adi listesi cikarilamadi.")
        return ri_cands
    try:
        avail_list = list(avail)
    except Exception as ex:
        _diag("UYARI: available_results iterable degil", str(ex))
        return ri_cands
    for r in avail_list:
        try:
            rname = (getattr(r, "operator_name", None) or getattr(r, "OperatorName", None)
                    or getattr(r, "name", None) or getattr(r, "Name", None))
            if rname: ri_cands.append(str(rname))
        except: pass
    return ri_cands

def step_B_discover_modes(rst_path):
    """DPF ile RST'yi ac, mevcut sonuc adlarini ve HI/mode/freq bilgisini oku."""
    _step("ADIM B: RST acma + mevcut sonuc adlarini kesfetme")

    import mech_dpf
    import Ans.DataProcessing as dpf
    try: mech_dpf.setExtAPI(ExtAPI)
    except: pass

    if not os.path.exists(rst_path):
        _diag("BASARISIZ: RST dosyasi bulunamadi: " + rst_path)
        return None, None, None, None

    ds = dpf.DataSources(rst_path)
    dpf_model = dpf.Model(ds)
    _diag("RST yuklendi (DPF Model olusturuldu).")

    ri_cands = _result_info_names(dpf_model)
    if ri_cands:
        _diag("RST'deki TUM sonuc adlari:", str(ri_cands))

    energy_like = [c for c in ri_cands if ("energ" in c.lower() or "strain" in c.lower())]
    if energy_like:
        _diag("Enerji/strain iceren aday adlar (bunlar denenecek):", str(energy_like))
    else:
        _diag("UYARI: result_info icinde enerji/strain iceren isim bulunamadi -- "
             "yine de bilinen isimler (ENG_SE vb.) denenecek.")

    tfs = dpf_model.TimeFreqSupport
    try: nsets = int(tfs.NumberSets)
    except: nsets = 0

    freqs = []
    try:
        ff = tfs.TimeFreqs
        ff_data = ff.Data if ff is not None else None
        if ff_data is not None:
            freqs = [_qty_to_float(x) for x in list(ff_data)]
    except Exception as ex:
        _diag("UYARI: TimeFreqs okunamadi", str(ex))
        freqs = []

    his = []
    try:
        hf = tfs.GetHarmonicIndices()
        hf_data = hf.Data if hf is not None else None
        if hf_data is not None:
            his = [int(round(_qty_to_float(x))) for x in list(hf_data)]
    except Exception as ex:
        _diag("GetHarmonicIndices basarisiz, HarmonicIndices deneniyor", str(ex))
        try:
            hf = tfs.HarmonicIndices
            hf_data = hf.Data if hf is not None else None
            if hf_data is not None:
                his = [int(round(_qty_to_float(x))) for x in list(hf_data)]
        except Exception as ex2:
            _diag("UYARI: HarmonicIndices de okunamadi", str(ex2))
            his = []

    _diag("Toplam set sayisi (NumberSets): {0}".format(nsets))
    mode_list = []
    counters = {}
    for i in range(nsets):
        sid = i + 1
        hi = abs(his[i]) if i < len(his) else 0
        fr = freqs[i] if i < len(freqs) else 0.0
        counters[hi] = counters.get(hi, 0) + 1
        mode_list.append((sid, hi, counters[hi], fr))

    _log("  Set  HI  Mode(local)  Freq(Hz)")
    for (sid, hi, mode_local, fr) in mode_list[:15]:
        _log("  {0:>4} {1:>4} {2:>10}  {3:>10.2f}".format(sid, hi, mode_local, fr))
    if len(mode_list) > 15:
        _log("  ... ({0} set daha var)".format(len(mode_list) - 15))

    return dpf, ds, dpf_model, mode_list

def step_C_read_energy(dpf, ds, dpf_model, set_ids):
    """Verilen set_id'ler icin element strain energy'sini TEK cagriyla oku.
    Iki farkli API tarzini dener:
      - YUKSEK SEVIYE: dpf.operators.result.<isim>() -- varsa .inputs.x.Connect()
      - DUSUK SEVIYE:  dpf.Operator(<isim>) -- .Connect(pin, deger) / .GetOutput(pin)"""
    _step("ADIM C: Strain energy okuma (set_ids={0})".format(set_ids))

    ri_cands = _result_info_names(dpf_model)
    ri_energy_cands = [c for c in ri_cands if ("energ" in c.lower() or "strain" in c.lower()
                                                or c.upper() == "ENG_SE")]
    candidates = (ri_energy_cands + ["ENG_SE", "elastic_strain_energy_density",
                                     "stiffness_matrix_energy", "cyclic_strain_energy"])
    seen = set(); candidates_unique = []
    for c in candidates:
        if c not in seen:
            seen.add(c); candidates_unique.append(c)

    _diag("Denenecek adaylar (sirayla):", str(candidates_unique))

    op = None; used_name = None; fc = None; nf = 0
    for cand in candidates_unique:
        if _CANCEL[0]: return None, None
        _diag("Deneniyor: '{0}'".format(cand))

        op = None; op_is_highlevel = False
        try:
            factory = getattr(dpf.operators.result, cand, None)
            if factory is not None:
                op = factory()
                op_is_highlevel = True
                _diag("  -> YUKSEK SEVIYE operator olusturuldu (dpf.operators.result.{0}())".format(cand))
        except Exception as ex:
            _diag("  -> yuksek seviye olusturma basarisiz", str(ex))

        if op is None:
            try:
                op = dpf.Operator(cand)
                op_is_highlevel = False
                _diag("  -> DUSUK SEVIYE Operator('{0}') olusturuldu.".format(cand))
            except Exception as ex:
                _diag("  -> Operator('{0}') OLUSTURULAMADI".format(cand), str(ex))
                continue

        try:
            if op_is_highlevel:
                op.inputs.data_sources.Connect(ds)
                op.inputs.time_scoping.Connect(list(set_ids))
                try:
                    op.inputs.read_cyclic.Connect(1)
                    _diag("  -> read_cyclic=1 baglandi (BASE SEKTOR).")
                except:
                    _diag("  -> UYARI: read_cyclic input yok, base sektor garantisi YOK.")
                fc = op.outputs.fields_container.GetData()
            else:
                # Dusuk seviye API: pin 0 = time_scoping, pin 4 = data_sources,
                # cikti pin 0 = fields_container.
                op.Connect(0, list(set_ids))
                op.Connect(4, ds)
                fc = op.GetOutput(0)
        except Exception as ex:
            _diag("  -> BAGLANTI/OKUMA BASARISIZ", str(ex))
            op = None
            continue

        if fc is None:
            _diag("  -> '{0}' fields_container None dondu, sonrakine geciliyor.".format(cand))
            op = None
            continue

        try: nf = fc.FieldCount
        except:
            try: nf = len(fc)
            except: nf = 0
        if nf == 0:
            _diag("  -> '{0}' 0 field dondurdu (bu RST'de yok), sonrakine geciliyor.".format(cand))
            op = None
            continue
        used_name = cand
        _diag("BASARILI: '{0}' -> {1} field dondu.".format(cand, nf))
        break

    if op is None or used_name is None:
        _diag("TAMAMEN BASARISIZ: Hicbir aday veri dondurmedi.")
        return None, None

    result = {}
    for fi in range(nf):
        if _CANCEL[0]: break
        try: field = fc[fi]
        except: field = fc.GetFieldByIndex(fi)
        try: sid = int(fc.GetLabelSpace(fi)["time"])
        except:
            try: sid = int(field.TimeId)
            except: sid = set_ids[fi] if fi < len(set_ids) else (fi + 1)
        try:
            ids_raw = field.ScopingIds
            ids = list(ids_raw) if ids_raw is not None else []
        except:
            try:
                ids_raw = field.Scoping.Ids
                ids = list(ids_raw) if ids_raw is not None else []
            except:
                ids = []
        try:
            data_raw = field.Data
            data = list(data_raw) if data_raw is not None else []
        except:
            data = []
        result[sid] = dict(zip(ids, data))
        _diag("  Set {0}: {1} element, ornek deger[0]={2}".format(
            sid, len(ids), data[0] if data else "yok"))

    return result, used_name

    def _on_rst_browse(self, s, a):
        dlg = OpenFileDialog()
        dlg.Title  = "Select RST result file"
        dlg.Filter = "Ansys Result (*.rst)|*.rst|All files (*.*)|*.*"
        init = self.txt_rst.Text.strip()
        if init and os.path.exists(os.path.dirname(init)):
            dlg.InitialDirectory = os.path.dirname(init)
        if dlg.ShowDialog() == DialogResult.OK:
            self.txt_rst.Text = dlg.FileName
            if not self.txt_dir.Text.strip():
                self.txt_dir.Text = os.path.dirname(dlg.FileName)

    def _on_dir_browse(self, s, a):
        dlg = OpenFileDialog()
        dlg.Title = "Select output folder"; dlg.Filter = "Folder|*.none"
        dlg.CheckFileExists = False; dlg.CheckPathExists = True
        dlg.ValidateNames = False; dlg.FileName = "Select folder"
        if self.txt_dir.Text and os.path.exists(self.txt_dir.Text):
            dlg.InitialDirectory = self.txt_dir.Text
        if dlg.ShowDialog() == DialogResult.OK:
            folder = os.path.dirname(dlg.FileName)
            if not folder: folder = dlg.FileName
            self.txt_dir.Text = folder

    def _on_run(self, s, a):
        errs = []
        rst = self.txt_rst.Text.strip()
        if not rst or not os.path.exists(rst):
            errs.append("Please select a valid RST file.")

        bodies = []
        if self.combo_b1.SelectedIndex > 0:
            bodies.append(str(self.combo_b1.SelectedItem))
        if self.combo_b2.SelectedIndex > 0:
            bodies.append(str(self.combo_b2.SelectedItem))
        if not bodies:
            errs.append("Please select at least 1 body (Named Selection).")

        try:
            nsets = int(self.txt_nsets.Text.strip()); assert nsets > 0
        except:
            errs.append("Number of sets to test must be a positive integer."); nsets = 3

        od = self.txt_dir.Text.strip()
        if not od:
            errs.append("Please select an output folder.")

        if errs:
            MessageBox.Show("\n".join(errs), "Warning"); return

        self.rst_path   = rst
        self.body_names = bodies
        self.n_sets     = nsets
        self.out_dir    = od
        _reg_set("LastRst", rst); _reg_set("LastDir", od)
        self.DialogResult = DialogResult.OK; self.Close()

def main():
    global FONT_TITLE
    _CANCEL[0] = False
    base = Form().Font
    FONT_TITLE = Font(base, FontStyle.Bold)

    model = ExtAPI.DataModel.Project.Model
    ns_names = get_ns_names(model)
    if not ns_names:
        MessageBox.Show("At least 1 Named Selection is required.", "Error"); return

    dlg = RstSeTestDialog(ns_names)
    if dlg.ShowDialog() != DialogResult.OK:
        return

    prog = ProgressDialog()
    _PROG[0] = prog
    prog.Show()
    prog.set_step("Starting...")

    _log("#" * 70)
    _log("# RST STRAIN ENERGY TEST SCRIPTI")
    _log("#" * 70)

    body_elem_ids = None
    mode_list = None
    energy_by_set = None
    op_used = None

    try:
        # ---- ADIM A ----
        body_elem_ids = step_A_resolve_elements(model, dlg.body_names)
        if _CANCEL[0]:
            _log(""); _log("IPTAL EDILDI (ADIM A sirasinda).")
        elif not body_elem_ids:
            _log(""); _log("DURDURULDU: ADIM A basarisiz oldu, devam edilemiyor.")
        else:
            # ---- ADIM B ----
            dpf, ds, dpf_model, mode_list = step_B_discover_modes(dlg.rst_path)
            if _CANCEL[0]:
                _log(""); _log("IPTAL EDILDI (ADIM B sirasinda).")
            elif mode_list is None:
                _log(""); _log("DURDURULDU: ADIM B basarisiz oldu, devam edilemiyor.")
            else:
                # ---- ADIM C ----
                test_sets = [m[0] for m in mode_list[:dlg.n_sets]]
                energy_by_set, op_used = step_C_read_energy(dpf, ds, dpf_model, test_sets)
                if _CANCEL[0]:
                    _log(""); _log("IPTAL EDILDI (ADIM C sirasinda).")
                elif energy_by_set is None:
                    _log(""); _log("DURDURULDU: ADIM C basarisiz oldu.")
                else:
                    # ---- SONUC TABLOSU ----
                    _step("SONUC TABLOSU  (operator: {0})".format(op_used))
                    for (sid, hi, mode_local, fr) in mode_list[:dlg.n_sets]:
                        elem_vals = energy_by_set.get(sid, {})
                        all_val = sum(elem_vals.values())
                        line = "HI{0} M{1} (Set{2}, {3:.1f}Hz): ALL={4:.6e}".format(
                            hi, mode_local, sid, fr, all_val)
                        for name in dlg.body_names:
                            ids = body_elem_ids.get(name, set())
                            body_val = sum(elem_vals.get(eid, 0.0) for eid in ids)
                            ratio = (body_val / all_val * 100.0) if all_val != 0 else 0.0
                            line += "  |  {0}={1:.6e} ({2:.2f}%)".format(name, body_val, ratio)
                        _log(line)

                    _log("")
                    _log("TEST TAMAMLANDI. Yukaridaki tablo mantikli gorunuyorsa "
                        "(oranlar toplamda ~%100'u makul sekilde paylasiyorsa), "
                        "ana koda entegre edebiliriz.")
    except Exception as ex:
        _log("")
        _log("BEKLENMEYEN HATA: " + str(ex))
        import traceback
        _log(traceback.format_exc())

    prog.done()
    prog.close_ok()
    _PROG[0] = None

    report_path = _write_report(dlg.out_dir)

    if _CANCEL[0]:
        MessageBox.Show("Islem iptal edildi.\n\nRapor (o ana kadar): " + report_path, "Iptal Edildi")
    else:
        MessageBox.Show("Test tamamlandi.\n\nRapor: " + report_path, "RST SE Test - Tamamlandi")

def _write_report(out_dir):
    if not os.path.exists(out_dir):
        os.makedirs(out_dir)
    report_path = os.path.join(out_dir, "RST_SE_Test_Report.txt")
    with open(report_path, "w") as f:
        f.write("\n".join(_REPORT_LINES))
    print("Rapor yazildi: " + report_path)
    return report_path

main()
