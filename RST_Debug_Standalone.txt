REG_KEY = r"Software\SEExperiment"

def _reg_get(name):
    try:
        key = Win32.Registry.CurrentUser.OpenSubKey(REG_KEY)
        if key is None: return ""
        val = key.GetValue(name); key.Close()
        return str(val) if val else ""
    except: return ""

def _reg_set(name, val):
    try:
        key = Win32.Registry.CurrentUser.CreateSubKey(REG_KEY)
        key.SetValue(name, val); key.Close()
    except: pass

    def _on_rst_browse(self, s, a):
        dlg = OpenFileDialog()
        dlg.Title = "Select RST result file"
        dlg.Filter = "Ansys Result (*.rst)|*.rst|All files (*.*)|*.*"
        init = self.txt_rst.Text.strip()
        if init and os.path.exists(os.path.dirname(init)):
            dlg.InitialDirectory = os.path.dirname(init)
        if dlg.ShowDialog() == DialogResult.OK:
            self.txt_rst.Text = dlg.FileName
            if not self.txt_dir.Text.strip():
                self.txt_dir.Text = os.path.dirname(dlg.FileName)

    def _on_dir_browse(self, s, a):
        dlg = OpenFileDialog()
        dlg.Title = "Select output folder"; dlg.Filter = "Folder|*.none"
        dlg.CheckFileExists = False; dlg.CheckPathExists = True
        dlg.ValidateNames = False; dlg.FileName = "Select folder"
        if self.txt_dir.Text and os.path.exists(self.txt_dir.Text):
            dlg.InitialDirectory = self.txt_dir.Text
        if dlg.ShowDialog() == DialogResult.OK:
            folder = os.path.dirname(dlg.FileName)
            if not folder: folder = dlg.FileName
            self.txt_dir.Text = folder

    def _on_run(self, s, a):
        errs = []
        rst = self.txt_rst.Text.strip()
        if not rst or not os.path.exists(rst):
            errs.append("Please select a valid RST file.")
        od = self.txt_dir.Text.strip()
        if not od: errs.append("Please select an output folder.")
        if errs:
            MessageBox.Show("\n".join(errs), "Warning"); return
        self.rst_path = rst
        self.out_dir = od
        _reg_set("LastRst", rst); _reg_set("LastDir", od)
        self.DialogResult = DialogResult.OK; self.Close()

def _dpf_import():
    import mech_dpf
    import Ans.DataProcessing as dpf
    try: mech_dpf.setExtAPI(ExtAPI)
    except: pass
    return dpf

CANDIDATES = [
    "cyclic_strain_energy", "elastic_strain_energy_density",
    "stiffness_matrix_energy", "part_internal_energy",
    "creep_strain_energy_density", "plastic_strain_energy_density",
    "thermal_strain_energy_density", "co_energy", "total_strain",
    "elastic_strain", "strain_eqv_as_mechanical",
]

def run_debug(rst_path, out_dir, prog=None):
    L = []
    def log(msg):
        print(msg)
        L.append(msg)
        if prog: prog.update("RST Debug running...", msg[:120])
        Application.DoEvents()

    log("=== RST DEBUG START ===")
    log("RST: " + rst_path)

    dpf = _dpf_import()
    ds = dpf.DataSources(rst_path)
    model = dpf.Model(ds)
    log("Model olusturuldu.")

    log("")
    log("=== dir(model) -- TUM attribute'lar (alt cizgiyle baslamayanlar) ===")
    try:
        attrs = sorted([a for a in dir(model) if not a.startswith("_")])
        log(str(attrs))
    except Exception as ex:
        log("dir(model) BASARISIZ: " + str(ex))

    log("")
    log("=== Olasi property denemeleri (model.X) ===")
    for attr in ["metadata", "result_info", "available_named_selections",
                "meshed_region", "Results", "elements", "Elements",
                "MeshedRegion", "ResultInfo", "AvailableNamedSelections"]:
        try:
            val = getattr(model, attr)
            log("model.{0} -> BASARILI, tip={1}, deger(kisa)={2}".format(
                attr, type(val), str(val)[:200]))
        except Exception as ex:
            log("model.{0} -> BASARISIZ: {1}".format(attr, ex))

    log("")
    log("=== TimeFreqSupport (bilinen calisan) ===")
    try:
        tfs = model.TimeFreqSupport
        log("NumberSets = {0}".format(tfs.NumberSets))
    except Exception as ex:
        log("TimeFreqSupport BASARISIZ: " + str(ex))

    log("")
    log("=== dir(dpf.operators.result) icinde enerji/strain iceren TUM adlar ===")
    try:
        all_res_ops = sorted([a for a in dir(dpf.operators.result) if not a.startswith("_")])
        log("TOPLAM operator sayisi: {0}".format(len(all_res_ops)))
        energy_like = [a for a in all_res_ops if ("energ" in a.lower() or "strain" in a.lower())]
        log("enerji/strain-benzeri: {0}".format(energy_like))
    except Exception as ex:
        log("dir(dpf.operators.result) BASARISIZ: " + str(ex))

    log("")
    log("=== Aday operatorlerin TAM testi (set_id=1) ===")
    for cand in CANDIDATES:
        log("--- '{0}' ---".format(cand))
        try:
            factory = getattr(dpf.operators.result, cand, None)
            if factory is None:
                log("  factory YOK (dpf.operators.result.{0} attribute'u bulunamadi)".format(cand))
                continue
            op = factory()
            log("  operator olusturuldu.")
        except Exception as ex:
            log("  operator olusturma BASARISIZ: {0}".format(ex))
            continue

        try:
            op.inputs.data_sources.Connect(ds)
            op.inputs.time_scoping.Connect([1])
        except Exception as ex:
            log("  Connect BASARISIZ: {0}".format(ex))
            continue

        try:
            fc = op.outputs.fields_container.GetData()
        except Exception as ex:
            log("  GetData() BASARISIZ: {0}".format(ex))
            continue

        try: nf = fc.FieldCount
        except:
            try: nf = len(fc)
            except: nf = -1
        log("  field count = {0}".format(nf))
        if nf is None or nf <= 0:
            log("  -> field yok, sonraki adaya geciliyor.")
            continue

        try: field = fc[0]
        except Exception as ex:
            log("  fc[0] BASARISIZ: {0}".format(ex))
            continue

        try:
            ids = list(field.ScopingIds)
            log("  ScopingIds count = {0}, ornek = {1}".format(len(ids), ids[:5]))
        except Exception as ex:
            log("  ScopingIds BASARISIZ: {0}".format(ex))
            ids = None

        try:
            raw_data = field.Data
            if raw_data is None:
                log("  field.Data = None (!!)")
            else:
                data = list(raw_data)
                log("  Data count = {0}, ornek deger = {1}".format(len(data), data[:5]))
        except Exception as ex:
            log("  field.Data erisimi BASARISIZ: {0}".format(ex))

        try:
            loc = field.Location
            log("  field.Location = {0}".format(loc))
        except Exception as ex:
            log("  field.Location erisimi BASARISIZ: {0}".format(ex))

    log("")
    log("=== RST DEBUG END ===")

    out_path = os.path.join(out_dir, "RST_Debug_Log.txt")
    if not os.path.exists(out_dir): os.makedirs(out_dir)
    with open(out_path, "w") as f:
        f.write("\n".join(L))
    return out_path

def main():
    global FONT_TITLE
    base = Form().Font
    FONT_TITLE = Font(base, FontStyle.Bold)

    dlg = DebugDialog()
    if dlg.ShowDialog() != DialogResult.OK:
        return

    prog = ProgressDialog()
    prog.Show()
    prog.update("Starting RST debug...", "")

    try:
        out_path = run_debug(dlg.rst_path, dlg.out_dir, prog=prog)
    except Exception as ex:
        prog.close_ok()
        MessageBox.Show("RST debug failed:\n" + str(ex), "Error")
        return

    prog.close_ok()
    MessageBox.Show(
        "RST debug complete.\n\nLog file: {0}\n\n"
        "Please share the console output or this log file.".format(out_path),
        "RST Debug - Done")

main()
